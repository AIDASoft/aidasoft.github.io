70 Vector3D TiledLayerGridXY::position(const CellID& cID) const {
71  _decoder->setValue(cID);
72  unsigned int _layerIndex;
73  Vector3D cellPosition;
74 
75  // AHcal: _layerIndex is [1,48], _layerOffsetX is [0,47]
76  _layerIndex = (*_decoder)[_identifierLayer];
77 
78  if ( _layerOffsetX.size() != 0 && _layerIndex <=_layerOffsetX.size() ) {
79  cellPosition.X = binToPosition((*_decoder)[_xId].value(), _gridSizeX, _layerOffsetX[_layerIndex - 1]*_gridSizeX/2.);
80  // check the integer cell boundary in x,
81  if ( ( _layerDimX.size() != 0 && _layerIndex <= _layerDimX.size() )
82  &&( _fractCellSizeXPerLayer.size() != 0 && _layerIndex <= _fractCellSizeXPerLayer.size() )
83  &&(std::fabs( cellPosition.X ) > ( _layerDimX.at(_layerIndex - 1) - _fractCellSizeXPerLayer.at(_layerIndex - 1) ))
84  )
85  {
86  // and correct the fraction cell center position in x.
87  cellPosition.X = cellPosition.X/std::fabs(cellPosition.X)
88  *(_layerDimX.at(_layerIndex - 1) - _fractCellSizeXPerLayer.at(_layerIndex - 1)/2.0) ;
89  }
90  } else {
91  cellPosition.X = binToPosition((*_decoder)[_xId].value(), _gridSizeX, _offsetX);
92  }
93  cellPosition.Y = binToPosition((*_decoder)[_yId].value(), _gridSizeY, _offsetY);
94  return cellPosition;
95 }
