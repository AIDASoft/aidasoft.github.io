25 static Ref_t create_detector(LCDD& lcdd, xml_h e, SensitiveDetector sens) {
26  xml_det_t x_det = e;
27  xml_dim_t dim = x_det.dimensions();
28  bool reflect = x_det.reflect();
29  xml_comp_t beam = x_det.beampipe();
30  string det_name = x_det.nameStr();
31  int id = x_det.id();
32  Material air = lcdd.air();
33  DetElement sdet (det_name,id);
34  Layering layering (x_det);
35 
36  Volume motherVol = lcdd.pickMotherVolume(sdet);
37 
38  double rmax = dim.outer_r();
39  double rmin = dim.inner_r();
40  double zinner = dim.inner_z();
41  double outgoingR = beam.outgoing_r();
42  double incomingR = beam.incoming_r();
43  double xangle = beam.crossing_angle();
44  double xangleHalf = xangle/2;
45  double thickness = layering.totalThickness();
46  double zpos = zinner + thickness/2;
47  // Beampipe position in envelope.
48  double beamPosX = std::tan(xangleHalf) * zpos;
49 
50  // Detector envelope solid.
51  Tube envelopeTube(rmin,rmax,thickness/2);
52 
53  // First envelope bool subtracion of outgoing beampipe.
54  // Incoming beampipe solid.
55  Tube beamInTube(0,outgoingR,thickness);
56  // Position of incoming beampipe.
57  Position beamInPos(beamPosX,0,0);
59  Rotation3D beamInRot(RotationY(1.*xangleHalf));
60  Transform3D beamInTrans(beamInRot,beamInPos);
61 
62  // Second envelope bool subtracion of outgoing beampipe.
63  // Outgoing beampipe solid.
64  Tube beamOutTube(0,incomingR,thickness);
65  // Position of outgoing beampipe.
66  Position beamOutPos(-beamPosX,0,0);
67  // Rotation of outgoing beampipe.
68  Rotation3D beamOutRot(RotationY(-xangleHalf));
69  Transform3D beamOutTrans(beamOutRot,beamOutPos);
70 
71  // First envelope bool subtraction of incoming beampipe.
72  SubtractionSolid envelopeSubtraction1(envelopeTube,beamInTube,beamInTrans);
73  SubtractionSolid envelopeSubtraction2(envelopeSubtraction1,beamOutTube,beamOutTrans);
74 
75  // Final envelope bool volume.
76  Volume envelopeVol(det_name+"_envelope", envelopeSubtraction2, air);
77 
78  // Process each layer element.
79  double layerPosZ = -thickness / 2;
80  double layerDisplZ = 0;
81 
82  int layerCount = 1;
83  for(xml_coll_t c(x_det,_U(layer)); c; ++c) {
84  xml_comp_t x_layer = c;
85  double layerThickness = layering.singleLayerThickness(x_layer);
86 
87  // Create tube envelope for this layer, which can be reused in bool definition
88  // in the repeat loop below.
89  Tube layerTube(rmin,rmax,layerThickness);
90 
91  for(int i=0, repeat=x_layer.repeat(); i<repeat; ++i) {
92  string layer_nam = _toString(layerCount,"layer%d");
93  // Increment to new layer position.
94  layerDisplZ += layerThickness / 2;
95  layerPosZ += layerThickness / 2;
96 
97  // First layer subtraction solid.
98  DetElement layer(sdet,layer_nam,sdet.id());
99  double layerGlobalZ = zinner + layerDisplZ;
100  double layerPosX = std::tan(xangleHalf) * layerGlobalZ;
101  Position layer1SubPos( layerPosX,0,0);
102  Position layer2SubPos(-layerPosX,0,0);
103 
104  SubtractionSolid layerSubtraction1(layerTube,beamInTube,Transform3D(beamInRot,layer1SubPos));
105  // Second layer subtraction solid.
106  SubtractionSolid layerSubtraction2(layerSubtraction1,beamOutTube,Transform3D(beamOutRot,layer2SubPos));
107  // Layer LV.
108  Volume layerVol(det_name+"_"+layer_nam,layerSubtraction2,air);
109 
110  // Slice loop.
111  int sliceCount = 1;
112  double slicePosZ = -layerThickness / 2;
113  double sliceDisplZ = 0;
114  for(xml_coll_t l(x_layer,_U(slice)); l; ++l) {
115  xml_comp_t x_slice = l;
116  string slice_nam = _toString(sliceCount,"slice%d");
118  double sliceThickness = x_slice.thickness();
119  Material slice_mat = lcdd.material(x_slice.materialStr());
120 
121  // Go to mid of this slice.
122  sliceDisplZ += sliceThickness / 2;
123  slicePosZ += sliceThickness / 2;
124 
125  // Slice's basic tube.
126  Tube sliceTube(rmin,rmax,sliceThickness);
127  DetElement slice(layer,slice_nam,sdet.id());
128  double sliceGlobalZ = zinner + (layerDisplZ - layerThickness / 2) + sliceDisplZ;
129  double slicePosX = std::tan(xangleHalf) * sliceGlobalZ;
130 
131  // First slice subtraction solid.
132  SubtractionSolid sliceSubtraction1(sliceTube,beamInTube,Transform3D(beamInRot,Position(slicePosX,0,0)));
133  // Second slice subtraction solid.
134  SubtractionSolid sliceSubtraction2(sliceSubtraction1,beamOutTube,Transform3D(beamOutRot,Position(-slicePosX,0,0)));
135  // Slice LV.
136  Volume sliceVol(det_name+"_"+layer_nam+"_"+slice_nam, sliceSubtraction2, slice_mat);
137 
138  if ( x_slice.isSensitive() ) {
139  sens.setType("calorimeter");
140  sliceVol.setSensitiveDetector(sens);
141  }
142  // Set attributes of slice
143  slice.setAttributes(lcdd, sliceVol, x_slice.regionStr(), x_slice.limitsStr(), x_slice.visStr());
144 
145  // Place volume in layer
146  PlacedVolume pv = layerVol.placeVolume(sliceVol,Position(0,0,slicePosZ));
147  pv.addPhysVolID("slice",sliceCount);
148  slice.setPlacement(pv);
149 
150  // Start of next slice.
151  sliceDisplZ += sliceThickness / 2;
152  slicePosZ += sliceThickness / 2;
153  ++sliceCount;
154  }
155  // Set attributes of slice
156  layer.setAttributes(lcdd, layerVol, x_layer.regionStr(), x_layer.limitsStr(), x_layer.visStr());
157 
158  // Layer PV.
159  PlacedVolume layerPV = envelopeVol.placeVolume(layerVol,Position(0,0,layerPosZ));
160  layerPV.addPhysVolID("layer", layerCount);
161  layer.setPlacement(layerPV);
162 
163  // Increment to start of next layer.
164  layerDisplZ += layerThickness / 2;
165  layerPosZ += layerThickness / 2;
166  ++layerCount;
167  }
168  }
169  sdet.setVisAttributes(lcdd, x_det.visStr(), envelopeVol);
170 
171  // Reflect it.
172  if ( reflect ) {
173  Assembly assembly(det_name+"_assembly");
174  PlacedVolume pv = motherVol.placeVolume(assembly);
175  pv.addPhysVolID("system", id);
176  sdet.setPlacement(pv);
177 
178  DetElement sdetA(sdet,det_name+"_A",x_det.id());
179  pv = assembly.placeVolume(envelopeVol,Transform3D(RotationZ(M_PI),Position(0,0,zpos)));
180  pv.addPhysVolID("barrel", 1);
181  sdetA.setPlacement(pv);
182 
183  DetElement sdetB = sdetA.clone(det_name+"_B",x_det.id());
184  sdet.add(sdetB);
185  pv = assembly.placeVolume(envelopeVol,Transform3D(RotationY(M_PI),Position(0,0,-zpos)));
186  pv.addPhysVolID("barrel", 2);
187  sdetB.setPlacement(pv);
188  }
189  else {
190  PlacedVolume pv = motherVol.placeVolume(envelopeVol,Transform3D(RotationZ(M_PI),Position(0,0,zpos)));
191  pv.addPhysVolID("system", id);
192  pv.addPhysVolID("barrel", 1);
193  sdet.setPlacement(pv);
194  }
195  return sdet;
196 }
