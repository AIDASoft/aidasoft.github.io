72  virtual std::vector< std::pair<DDSurfaces::Vector3D, DDSurfaces::Vector3D> > getLines(unsigned){
73 
74  std::vector< std::pair<DDSurfaces::Vector3D, DDSurfaces::Vector3D> > lines ;
75 
76  double alpha = ( _sym ? 2.* M_PI / _sym : 0. ) ;
77 
78  for(unsigned i=0 ; i < _sym ; ++i){
79  double gam0 = i * alpha + _phi0 ;
80  double gam1 = (i+1) * alpha + _phi0 ;
81  lines.push_back( std::make_pair( DDSurfaces::Vector3D( _r*cos(gam0), _r*sin(gam0), origin().z() ),
82  DDSurfaces::Vector3D( _r*cos(gam1), _r*sin(gam1), origin().z() ) ) ) ;
83  }
84  return lines;
85  }
