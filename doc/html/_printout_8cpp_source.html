284 string DD4hep::format(const char* src, const char* fmt, va_list& args) {
285  char str[4096];
286  size_t len1 = ::snprintf(str, sizeof(str), "%s: ", src);
287  size_t len2 = ::vsnprintf(str + len1, sizeof(str) - len1, fmt, args);
288  if ( len2 > sizeof(str) - len1 ) {
289  len2 = sizeof(str) - len1 - 1;
290  str[sizeof(str)-1] = 0;
291  }
292  return string(str);
293 }
