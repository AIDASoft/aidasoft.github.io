105  void MegatileLayerGridXY::calculateDerivedQuantities() const {
106 
107  if (_calculated) return; // already up to date
108 
109  _megaSize = _totalSizeY / _nMegaY ; // size of square megatile in this layer
110  _gridSizeL = (_megaSize - 2.*_deadWidth) / _nStripsX ; // size of cell within megatile (x)
111  _gridSizeT = (_megaSize - 2.*_deadWidth) / _nStripsY ;
112  _gridSizeS = (_megaSize - 2.*_deadWidth) / _nCells;
113 
114  std::cout << "calculateDerivedQuantities: total X width: " << _totalSizeX
115  << " / nMega: " << _nMegaY << " = " << _megaSize
116  << " ; dead width = " << _deadWidth << " ; nstripsX,Y, nCells : " << _nStripsX
117  << " " << _nStripsY << " " << _nCells
118  << " ; stripX,Y/cell size: " << _gridSizeL
119  << " " << _gridSizeT << " " << _gridSizeS << std::endl;
120 
121  _calculated=true;
122 
123  return;
