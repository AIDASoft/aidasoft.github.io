51 Vector3D PolarGridRPhi2::position(const CellID& cID) const {
52  _decoder->setValue(cID);
53  Vector3D cellPosition;
54  const int rBin = (*_decoder)[_rId].value();
55  double R = binToPosition(rBin, _gridRValues, _offsetR);
56  double phi = binToPosition((*_decoder)[_phiId].value(), _gridPhiValues[rBin], _offsetPhi+_gridPhiValues[rBin]*0.5);
57 
58  if ( phi < _offsetPhi) {
59  phi += 2*M_PI;
60  }
61 
62  cellPosition.X = R * cos(phi);
63  cellPosition.Y = R * sin(phi);
64 
65  return cellPosition;
66 }

69  CellID PolarGridRPhi2::cellID(const Vector3D& localPosition, const Vector3D& /* globalPosition */, const VolumeID& vID) const {
70  _decoder->setValue(vID);
71  double phi = atan2(localPosition.Y,localPosition.X);
72  double R = sqrt( localPosition.X * localPosition.X + localPosition.Y * localPosition.Y );
73 
74  const int rBin = positionToBin(R, _gridRValues, _offsetR);
75  (*_decoder)[_rId] = rBin;
76 
77  if ( phi < _offsetPhi) {
78  phi += 2*M_PI;
79  }
80  const int pBin = positionToBin(phi, _gridPhiValues[rBin], _offsetPhi+_gridPhiValues[rBin]*0.5);
81  (*_decoder)[_phiId] = pBin;
82 
83  return _decoder->getValue();
84 }
