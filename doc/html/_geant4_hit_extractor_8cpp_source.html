59 static void* _convertHitCollection(const char* source) {
60  typedef DD4hep::DDEve::SimulationHit SimulationHit;
61  const std::vector<SimpleHit*>* c = (std::vector<SimpleHit*>*)source;
62  std::vector<SimulationHit>* pv = new std::vector<SimulationHit>();
63  if ( source && c->size() > 0 ) {
64  for(std::vector<SimpleHit*>::const_iterator k=c->begin(); k!=c->end(); ++k) {
65  SimpleTracker::Hit* trh = dynamic_cast<SimpleTracker::Hit*>(*k);
66  if ( trh ) {
67  pv->push_back(SimulationHit(trh->position, trh->energyDeposit));
68  continue;
69  }
70  SimpleCalorimeter::Hit* cah = dynamic_cast<SimpleCalorimeter::Hit*>(*k);
71  if ( cah ) {
72  pv->push_back(SimulationHit(cah->position, cah->energyDeposit));
73  continue;
74  }
75  }
76  }
77  return pv;
78 }
