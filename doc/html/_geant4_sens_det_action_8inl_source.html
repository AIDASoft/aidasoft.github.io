84  size_t Geant4SensitiveAction<T>::defineReadoutCollection(const std::string match_name) {
85  typedef Geometry::ReadoutObject _RO;
86  Readout ro = m_sensitive.readout();
87  for(_RO::Collections::const_iterator i=ro->hits.begin(); i!=ro->hits.end(); ++i) {
88  const _RO::Collection& c = *i;
89  if ( c.name == match_name ) {
90  size_t coll_id = defineCollection<HIT>(match_name);
91  Geant4Filter* filter =
92  new Geant4ReadoutVolumeFilter(context(),name()+"_"+c.name,ro,c.name);
93  adopt_front(filter);
94  return coll_id;
95  }
96  }
97  except("+++ Custom collection name '%s' not defined in the Readout object: %s.",
98  m_collectionName.c_str(), ro.name());
