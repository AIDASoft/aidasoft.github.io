25 static Ref_t create_detector(LCDD& lcdd, xml_h e, SensitiveDetector sens) {
26  xml_det_t x_det = e;
27  xml_dim_t dim = x_det.dimensions();
28  int det_id = x_det.id();
29  bool reflect = x_det.reflect(true);
30  string det_name = x_det.nameStr();
31  Material air = lcdd.air();
32  int numsides = dim.numsides();
33  double rmin = dim.rmin();
34  double rmax = dim.rmax()*std::cos(M_PI/numsides);
35  double zmin = dim.zmin();
36  Layering layering(x_det);
37  double totalThickness = layering.totalThickness();
38  Volume endcapVol("endcap",PolyhedraRegular(numsides,rmin,rmax,totalThickness),air);
39  DetElement endcap("endcap",det_id);
40 
41  int l_num = 1;
42  int layerType = 0;
43  double layerZ = -totalThickness/2;
44 
45  endcapVol.setAttributes(lcdd,x_det.regionStr(),x_det.limitsStr(),x_det.visStr());
46 
47  for(xml_coll_t c(x_det,_U(layer)); c; ++c) {
48  xml_comp_t x_layer = c;
49  double l_thick = layering.layer(l_num-1)->thickness();
50  string l_name = _toString(layerType,"layer%d");
51  int l_repeat = x_layer.repeat();
52  Volume l_vol(l_name,PolyhedraRegular(numsides,rmin,rmax,l_thick),air);
53  vector<PlacedVolume> sensitives;
54 
55  int s_num = 1;
56  double sliceZ = -l_thick/2;
57  for(xml_coll_t s(x_layer,_U(slice)); s; ++s) {
58  xml_comp_t x_slice = s;
59  string s_name = _toString(s_num,"slice%d");
60  double s_thick = x_slice.thickness();
61  Material s_mat = lcdd.material(x_slice.materialStr());
62  Volume s_vol(s_name,PolyhedraRegular(numsides,rmin,rmax,s_thick),s_mat);
63 
64  s_vol.setVisAttributes(lcdd.visAttributes(x_slice.visStr()));
65  sliceZ += s_thick/2;
66  PlacedVolume s_phv = l_vol.placeVolume(s_vol,Position(0,0,sliceZ));
67  s_phv.addPhysVolID("slice",s_num);
68  if ( x_slice.isSensitive() ) {
69  sens.setType("calorimeter");
70  s_vol.setSensitiveDetector(sens);
71  sensitives.push_back(s_phv);
72  }
73  sliceZ += s_thick/2;
74  s_num++;
75  }
76  l_vol.setVisAttributes(lcdd.visAttributes(x_layer.visStr()));
77  if ( l_repeat <= 0 ) throw std::runtime_error(x_det.nameStr()+"> Invalid repeat value");
78  for(int j=0; j<l_repeat; ++j) {
79  string phys_lay = _toString(l_num,"layer%d");
80  layerZ += l_thick/2;
81  DetElement layer_elt(endcap, phys_lay, l_num);
82  PlacedVolume pv = endcapVol.placeVolume(l_vol,Position(0,0,layerZ));
83  pv.addPhysVolID("layer", l_num);
84  layer_elt.setPlacement(pv);
85  for(size_t ic=0; ic<sensitives.size(); ++ic) {
86  PlacedVolume sens_pv = sensitives[ic];
87  DetElement comp_elt(layer_elt,sens_pv.volume().name(),l_num);
88  comp_elt.setPlacement(sens_pv);
89  }
90  layerZ += l_thick/2;
91  ++l_num;
92  }
93  ++layerType;
94  }
95 
96  double z_pos = zmin+totalThickness/2;
97  PlacedVolume pv;
98  // Reflect it.
99  if ( reflect ) {
100  Assembly assembly(det_name);
101  DetElement both_endcaps(det_name,det_id);
102  Volume motherVol = lcdd.pickMotherVolume(both_endcaps);
103  DetElement sdetA = endcap;
104  Ref_t(sdetA)->SetName((det_name+"_A").c_str());
105  DetElement sdetB = endcap.clone(det_name+"_B",x_det.id());
106 
107  pv = assembly.placeVolume(endcapVol,Transform3D(RotationZYX(M_PI/numsides,0,0),
108  Position(0,0,z_pos)));
109  pv.addPhysVolID("barrel", 1);
110  sdetA.setPlacement(pv);
111 
112  pv = assembly.placeVolume(endcapVol,Transform3D(RotationZYX(M_PI/numsides,M_PI,0),
113  Position(0,0,-z_pos)));
114  pv.addPhysVolID("barrel", 2);
115  sdetB.setPlacement(pv);
116 
117  pv = motherVol.placeVolume(assembly);
118  pv.addPhysVolID("system", det_id);
119  both_endcaps.setPlacement(pv);
120  both_endcaps.add(sdetA);
121  both_endcaps.add(sdetB);
122  return both_endcaps;
123  }
124  Volume motherVol = lcdd.pickMotherVolume(endcap);
125  pv = motherVol.placeVolume(endcapVol,Transform3D(RotationZYX(M_PI/numsides,0,0),
126  Position(0,0,z_pos)));
127  pv.addPhysVolID("system", det_id);
128  pv.addPhysVolID("barrel", 1);
129  endcap.setPlacement(pv);
130  Ref_t(endcap)->SetName(det_name.c_str());
131  return endcap;
132 }
