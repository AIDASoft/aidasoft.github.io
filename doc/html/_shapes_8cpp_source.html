49 std::string DD4hep::Geometry::toStringSolid(const TGeoShape* shape, int precision) {
50  stringstream log;
51 
52  if ( !shape ) {
53  return "[Invalid shape]";
54  }
55 
56  TClass* cl = shape->IsA();
57  int prec = log.precision();
58  log.setf(ios::fixed,ios::floatfield);
59  log << setprecision(precision);
60  log << cl->GetName();
61  if ( cl == TGeoBBox::Class() ) {
62  TGeoBBox* s = (TGeoBBox*) shape;
63  log << " x:" << s->GetDX()
64  << " y:" << s->GetDY()
65  << " z:" << s->GetDZ();
66  }
67  else if (cl == TGeoHalfSpace::Class()) {
68  TGeoHalfSpace* s = (TGeoHalfSpace*)(const_cast<TGeoShape*>(shape));
69  log << " Point: (" << s->GetPoint()[0] << ", " << s->GetPoint()[1] << ", " << s->GetPoint()[2] << ") "
70  << " Normal: (" << s->GetNorm()[0] << ", " << s->GetNorm()[1] << ", " << s->GetNorm()[2] << ") ";
71  }
72  else if (cl == TGeoTube::Class()) {
73  const TGeoTube* s = (const TGeoTube*) shape;
74  log << " rmin:" << s->GetRmin() << " rmax:" << s->GetRmax() << " dz:" << s->GetDz();
75  }
76  else if (cl == TGeoTubeSeg::Class()) {
77  const TGeoTubeSeg* s = (const TGeoTubeSeg*) shape;
78  log << " rmin:" << s->GetRmin() << " rmax:" << s->GetRmax() << " dz:" << s->GetDz()
79  << " Phi1:" << s->GetPhi1() << " Phi2:" << s->GetPhi2();
80  }
81  else if (cl == TGeoTrd1::Class()) {
82  const TGeoTrd1* s = (const TGeoTrd1*) shape;
83  log << " x1:" << s->GetDx1() << " x2:" << s->GetDx2() << " y:" << s->GetDy() << " z:" << s->GetDz();
84  }
85  else if (cl == TGeoTrd2::Class()) {
86  const TGeoTrd2* s = (const TGeoTrd2*) shape;
87  log << " x1:" << s->GetDx1() << " x2:" << s->GetDx2()
88  << " y1:" << s->GetDy1() << " y2:" << s->GetDy2() << " z:" << s->GetDz();
89  }
90  else if (cl == TGeoHype::Class()) {
91  const TGeoHype* s = (const TGeoHype*) shape;
92  log << " rmin:" << s->GetRmin() << " rmax:" << s->GetRmax() << " dz:" << s->GetDz()
93  << " StIn:" << s->GetStIn() << " StOut:" << s->GetStOut();
94  }
95  else if (cl == TGeoPgon::Class()) {
96  const TGeoPgon* s = (const TGeoPgon*) shape;
97  log << " Phi1:" << s->GetPhi1() << " dPhi:" << s->GetDphi()
98  << " NEdges:" << s->GetNedges() << " Nz:" << s->GetNz();
99  for(int i=0, n=s->GetNz(); i<n; ++i) {
100  log << " i=" << i << " z:" << s->GetZ(i)
101  << " r:[" << s->GetRmin(i) << "," << s->GetRmax(i) << "]";
102  }
103  }
104  else if (cl == TGeoPcon::Class()) {
105  const TGeoPcon* s = (const TGeoPcon*) shape;
106  log << " Phi1:" << s->GetPhi1() << " dPhi:" << s->GetDphi() << " Nz:" << s->GetNz();
107  for(int i=0, n=s->GetNz(); i<n; ++i) {
108  log << " i=" << i << " z:" << s->GetZ(i)
109  << " r:[" << s->GetRmin(i) << "," << s->GetRmax(i) << "]";
110  }
111  }
112  else if (cl == TGeoConeSeg::Class()) {
113  const TGeoConeSeg* s = (const TGeoConeSeg*) shape;
114  log << " rmin1:" << s->GetRmin1() << " rmax1:" << s->GetRmax1()
115  << " rmin2:" << s->GetRmin2() << " rmax2:" << s->GetRmax2()
116  << " dz:" << s->GetDz()
117  << " Phi1:" << s->GetPhi1() << " Phi2:" << s->GetPhi2();
118  }
119  else if (cl == TGeoParaboloid::Class()) {
120  const TGeoParaboloid* s = (const TGeoParaboloid*) shape;
121  log << " dz:" << s->GetDz() << " RLo:" << s->GetRlo() << " Rhi:" << s->GetRhi();
122  }
123  else if (cl == TGeoSphere::Class()) {
124  const TGeoSphere* s = (const TGeoSphere*) shape;
125  log << " rmin:" << s->GetRmin() << " rmax:" << s->GetRmax()
126  << " Phi1:" << s->GetPhi1() << " Phi2:" << s->GetPhi2()
127  << " Theta1:" << s->GetTheta1() << " Theta2:" << s->GetTheta2();
128  }
129  else if (cl == TGeoTorus::Class()) {
130  const TGeoTorus* s = (const TGeoTorus*) shape;
131  log << " rmin:" << s->GetRmin() << " rmax:" << s->GetRmax() << " r:" << s->GetR()
132  << " Phi1:" << s->GetPhi1() << " dPhi:" << s->GetDphi();
133  }
134  else if (cl == TGeoTrap::Class()) {
135  const TGeoTrap* s = (const TGeoTrap*) shape;
136  log << " dz:" << s->GetDz() << " Theta:" << s->GetTheta() << " Phi:" << s->GetPhi()
137  << " H1:" << s->GetH1() << " Bl1:" << s->GetBl1() << " Tl1:" << s->GetTl1() << " Alpha1:" << s->GetAlpha1()
138  << " H2:" << s->GetH2() << " Bl2:" << s->GetBl2() << " Tl2:" << s->GetTl2() << " Alpha2:" << s->GetAlpha2();
139  }
140  else if (shape->IsA() == TGeoCompositeShape::Class()) {
141  const TGeoCompositeShape* s = (const TGeoCompositeShape*) shape;
142  const TGeoBoolNode* boolean = s->GetBoolNode();
143  const TGeoShape* left = boolean->GetLeftShape();
144  const TGeoShape* right = boolean->GetRightShape();
145  TGeoBoolNode::EGeoBoolType oper = boolean->GetBooleanOperator();
146  if (oper == TGeoBoolNode::kGeoSubtraction)
147  log << "Subtraction: ";
148  else if (oper == TGeoBoolNode::kGeoUnion)
149  log << "Union: ";
150  else if (oper == TGeoBoolNode::kGeoIntersection)
151  log << "Intersection: ";
152  log << " Left:" << toStringSolid(left) << " Right:" << toStringSolid(right);
153  }
154  log << setprecision(prec);
155  return log.str();
156 }

213 Polycone::Polycone(double start, double delta, const vector<double>& rmin, const vector<double>& rmax,
214  const vector<double>& z) {
215  vector<double> params;
216  if (rmin.size() < 2) {
217  throw runtime_error("DD4hep: PolyCone::addZPlanes> Not enough Z planes. minimum is 2!");
218  }
219  params.push_back(start/dd4hep::deg);
220  params.push_back(delta/dd4hep::deg);
221  params.push_back(rmin.size());
222  for (size_t i = 0; i < rmin.size(); ++i) {
223  params.push_back(z[i] );
224  params.push_back(rmin[i] );
225  params.push_back(rmax[i] );
226  }
227  _assign(new TGeoPcon(&params[0]), "", "polycone", true);
228 }

231 void Polycone::addZPlanes(const vector<double>& rmin, const vector<double>& rmax, const vector<double>& z) {
232  TGeoPcon* s = *this;
233  vector<double> params;
234  size_t num = s->GetNz();
235  if (rmin.size() < 2) {
236  throw runtime_error("DD4hep: PolyCone::addZPlanes> Not enough Z planes. minimum is 2!");
237  }
238  params.push_back(s->GetPhi1());
239  params.push_back(s->GetDphi());
240  params.push_back(num + rmin.size());
241  for (size_t i = 0; i < num; ++i) {
242  params.push_back(s->GetZ(i));
243  params.push_back(s->GetRmin(i));
244  params.push_back(s->GetRmax(i));
245  }
246  for (size_t i = 0; i < rmin.size(); ++i) {
247  params.push_back(z[i] );
248  params.push_back(rmin[i] );
249  params.push_back(rmax[i] );
250  }
251  _setDimensions(&params[0]);
252 }
