51 Vector3D PolarGridRPhi::position(const CellID& cID) const {
52  _decoder->setValue(cID);
53  Vector3D cellPosition;
54  double R = binToPosition((*_decoder)[_rId].value(), _gridSizeR, _offsetR);
55  double phi = binToPosition((*_decoder)[_phiId].value(), _gridSizePhi, _offsetPhi);

64  CellID PolarGridRPhi::cellID(const Vector3D& localPosition, const Vector3D& /* globalPosition */, const VolumeID& vID) const {
65  _decoder->setValue(vID);
66  double phi = atan2(localPosition.Y,localPosition.X);
67  double R = sqrt( localPosition.X * localPosition.X + localPosition.Y * localPosition.Y );
68 
69  (*_decoder)[_rId] = positionToBin(R, _gridSizeR, _offsetR);
70  (*_decoder)[_phiId] = positionToBin(phi, _gridSizePhi, _offsetPhi);
71  return _decoder->getValue();
72 }
