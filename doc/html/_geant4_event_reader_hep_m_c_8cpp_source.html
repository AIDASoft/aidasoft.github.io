218  Geant4ParticleHandle p(*k);

222  p->vex += pos.x();

224  p->vez += pos.z();

228  Geant4ParticleHandle p(*k);

260  p->daughters.insert(*id);
261  }
262  }
263  EventStream::Vertices::iterator j;
264  for(j=verts.begin(); j != verts.end(); ++j) {
265  Geant4Vertex* v = (*j).second;

267  EventStream::Particles::iterator ipp = parts.find(*ip);
268  Geant4Particle* p = (*ipp).second;
269  for (id=v->in.begin(); id!=v->in.end();++id) {
270  p->parents.insert(*id);
271  }
272  }
273  }
276  vector<Geant4Particle*> beam;
277  for(i=parts.begin(); i != parts.end(); ++i) {
278  Geant4ParticleHandle p((*i).second);
279  if ( p->parents.size() == 0 ) {

281  Geant4Particle *pp = parts[*id];

284  }
285  }

287  //cout << "Clear parents of " << (*ipp)->id << endl;

364 int HepMC::read_particle(EventStream &info, istringstream& input, Geant4Particle * p) {
365  float ene = 0., theta = 0., phi = 0;
366  int size = 0, stat=0;
367  PropertyMask status(p->status);
368 
369  // check that the input stream is still OK after reading item
370  input >> p->id >> p->pdgID >> p->psx >> p->psy >> p->psz >> ene;
371  p->id = info.particles().size();
372  p->psx *= info.mom_unit;
373  p->psy *= info.mom_unit;
374  p->psz *= info.mom_unit;
375  ene *= info.mom_unit;
376  if ( !input )
377  return 0;
378  else if ( info.io_type != ascii ) {
379  input >> p->mass;
380  p->mass *= info.mom_unit;
381  }
382  else {
383  p->mass = std::sqrt(fabs(ene*ene - (p->psx*p->psx + p->psy*p->psy + p->psz*p->psz)));
384  }
385  // Reuse here the secondaries to store the end-vertex ID
386  input >> stat >> theta >> phi >> p->secondaries >> size;
387  if(!input)
388  return 0;
389 
390  //
391  // Generator status
392  // Simulator status 0 until simulator acts on it
393  status.clear();
394  if ( stat == 0 ) status.set(G4PARTICLE_GEN_EMPTY);
395  else if ( stat == 0x1 ) status.set(G4PARTICLE_GEN_STABLE);
396  else if ( stat == 0x2 ) status.set(G4PARTICLE_GEN_DECAYED);
397  else if ( stat == 0x3 ) status.set(G4PARTICLE_GEN_DOCUMENTATION);
398  else if ( stat == 0x4 ) status.set(G4PARTICLE_GEN_DOCUMENTATION);
399  else if ( stat == 0xB ) status.set(G4PARTICLE_GEN_DOCUMENTATION);
400 
401  // read flow patterns if any exist
402  for (int i = 0; i < size; ++i ) {
403  input >> p->colorFlow[0] >> p->colorFlow[1];
404  if(!input) return 0;
405  }
406  return 1;
407 }

615 bool HepMC::EventStream::read() {
616  EventStream& info = *this;
617  bool event_read = false;
618  static int num_evt = 0;
619  int num_line = 0, num_line_accepted = 0;
620 
621  releaseObjects(vertices());
622  releaseObjects(particles());
623 
624  ++num_evt;
625  if ( num_evt == 998 ) {
626  cout << "Hallo";
627  }
628  while( instream.good() ) {
629  char value = instream.peek();
630  istringstream input_line;
631  ++num_line;
632  if ( value == 'E' && event_read )
633  break;
634  else if ( instream.eof() && event_read )
635  goto Done;
636  else if ( instream.eof() )
637  return false;
638  else if ( value=='#' || ::isspace(value) ) {
639  get_input(instream,input_line);
640  continue;
641  }
642  value = get_input(instream,input_line);
643 
644  // On failure switch to end
645  if( !input_line || value < 0 )
646  goto Skip;
647 
648  ++num_line_accepted;
649  switch( value ) {
650  case 'H': {
651  int iotype = 0;
652  string key_value;
653  input_line >> key_value;
654  // search for event listing key before first event only.
655  key_value = key_value.substr(0,key_value.find('\r'));
656  if ( key_value == "H" && (this->io_type == gen || this->io_type == extascii) ) {
657  read_heavy_ion(info, input_line);
658  break;
659  }
660  else if( key_value == "HepMC::IO_GenEvent-START_EVENT_LISTING" )
661  this->set_io(gen,key_value);
662  else if( key_value == "HepMC::IO_Ascii-START_EVENT_LISTING" )
663  this->set_io(ascii,key_value);
664  else if( key_value == "HepMC::IO_ExtendedAscii-START_EVENT_LISTING" )
665  this->set_io(extascii,key_value);
666  else if( key_value == "HepMC::IO_Ascii-START_PARTICLE_DATA" )
667  this->set_io(ascii_pdt,key_value);
668  else if( key_value == "HepMC::IO_ExtendedAscii-START_PARTICLE_DATA" )
669  this->set_io(extascii_pdt,key_value);
670  else if( key_value == "HepMC::IO_GenEvent-END_EVENT_LISTING" )
671  iotype = gen;
672  else if( key_value == "HepMC::IO_Ascii-END_EVENT_LISTING" )
673  iotype = ascii;
674  else if( key_value == "HepMC::IO_ExtendedAscii-END_EVENT_LISTING" )
675  iotype = extascii;
676  else if( key_value == "HepMC::IO_Ascii-END_PARTICLE_DATA" )
677  iotype = ascii_pdt;
678  else if( key_value == "HepMC::IO_ExtendedAscii-END_PARTICLE_DATA" )
679  iotype = extascii_pdt;
680 
681  if( iotype != 0 && this->io_type != iotype ) {
682  cerr << "GenEvent::find_end_key: iotype keys have changed. "
683  << "MALFORMED INPUT" << endl;
684  instream.clear(ios::badbit);
685  return false;
686  }
687  else if ( iotype != 0 ) {
688  break;
689  }
690  continue;
691  }
692  case 'E': // deal with the event line
693  if ( !read_event_header(info, input_line, this->header) )
694  goto Skip;
695  event_read = true;
696  continue;
697 
698  case 'N': // get weight names
699  if ( !read_weight_names(info, input_line) )
700  goto Skip;
701  continue;
702 
703  case 'U': // get unit information if it exists
704  if ( !read_units(info, input_line) )
705  goto Skip;
706  continue;
707 
708  case 'C': // we have a GenCrossSection line
709  if ( !read_cross_section(info, input_line) )
710  goto Skip;
711  continue;
712 
713  case 'V': // Read vertex with particles
714  if ( !read_vertex(info, instream, input_line) )
715  goto Skip;
716  continue;
717 
718  case 'F': // Read PDF
719  if ( !read_pdf(info, input_line) )
720  goto Skip;
721  continue;
722 
723  case 'P': // we should not find this line
724  cerr << "streaming input: found unexpected Particle line." << endl;
725  continue;
726 
727  default: // ignore everything else
728  continue;
729  }
730  continue;
731  Skip:
732  printout(WARNING,"HepMC::EventStream","+++ Skip event with ID: %d",this->header.id);
733  releaseObjects(vertices());
734  releaseObjects(particles());
735  read_until_event_end(instream);
736  event_read = false;
737  if ( instream.eof() ) return false;
738  }
739 
740  if( not instream.good() ) return false;
741  Done:
742  fix_particles(info);
743  releaseObjects(vertices());
744  return true;
745 }
