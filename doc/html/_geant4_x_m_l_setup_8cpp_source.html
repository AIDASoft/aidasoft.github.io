116  enum { SENSITIVE, ACTION, FILTER };
118  Action _createAction(lcdd_t& lcdd, xml_h a, const string& seqType, int what) {
119  string nam = a.attr<string>(_U(name));
120  TypeName typ = TypeName::split(nam);
121  Kernel& kernel = Kernel::instance(lcdd);
122  Action action((what==FILTER) ? (Geant4Action*)kernel.globalFilter(typ.second,false)
123  : (what==ACTION) ? kernel.globalAction(typ.second,false)
125  : 0);
126  // Create the object using the factory method
127  if ( !action ) {
128  action = (what == SENSITIVE) ? Action(_convertSensitive(lcdd, a, seqType))
129  : (what==ACTION) ? _convertAction(lcdd, a)
130  : (what==FILTER) ? _convertAction(lcdd, a)
131  : Action();
132  if ( !action ) {
133  throw runtime_error(format("Geant4ActionSequence","DDG4: The action '%s'"
134  " cannot be created. [Action-Missing]",nam.c_str()));
135  }
136  }
137  return action;
138  }

286  xml_comp_t seq(e);
287  SensitiveSeq sdSeq;
288  string seqNam;
289  TypeName seqType;
290  int what = ACTION;
291  Kernel& kernel = Kernel::instance(lcdd);
292 
293  if ( seq.hasAttr(_U(sd)) ) {
294  string sd_nam = seq.attr<string>(_U(sd));
295  SensitiveDetector sensitive = lcdd.sensitiveDetector(sd_nam);
296  seqNam = seq.attr<string>(_U(type))+"/"+sd_nam;
297  if ( !sensitive.isValid() ) {
298  printout(ALWAYS,"Geant4Setup","+++ ActionSequence %s is defined, "
299  "but no sensitive detector present.",seqNam.c_str());
300  printout(ALWAYS,"Geant4Setup","+++ ---> Sequence for detector %s IGNORED on popular request!",
301  sd_nam.c_str());
302  return;
303  }
304  seqType = TypeName::split(seqNam);
305  sdSeq = SensitiveSeq(kernel,seqNam);
306  what = SENSITIVE;
307  }
308  else {
309  seqNam = seq.attr<string>(_U(name));
310  seqType = TypeName::split(seqNam);
311  }
312  printout(INFO,"Geant4Setup","+++ ActionSequence %s of type %s added.",
313  seqType.second.c_str(),seqType.first.c_str());
314 
315  if ( seqType.second == "PhysicsList" ) {
316  PhysicsActionSeq pl(&kernel.physicsList());
317  PropertyManager& m = kernel.physicsList().properties();
318  m.dump();
319  _setAttributes(pl,e);
320  m.dump();
321  }
322 
323  for(xml_coll_t a(seq,_Unicode(action)); a; ++a) {
324  string nam = a.attr<string>(_U(name));
325  Action action(_createAction(lcdd,a,seqType.second,what));
326  if ( seqType.second == "RunAction" )
327  kernel.runAction().adopt(_action<RunAction::handled_type>(action.get()));
328  else if ( seqType.second == "EventAction" )
329  kernel.eventAction().adopt(_action<EventAction::handled_type>(action.get()));
330  else if ( seqType.second == "GeneratorAction" )
331  kernel.generatorAction().adopt(_action<GenAction::handled_type>(action.get()));
332  else if ( seqType.second == "TrackingAction" )
333  kernel.trackingAction().adopt(_action<TrackAction::handled_type>(action.get()));
334  else if ( seqType.second == "StackingAction" )
335  kernel.stackingAction().adopt(_action<StackAction::handled_type>(action.get()));
336  else if ( seqType.second == "SteppingAction" )
337  kernel.steppingAction().adopt(_action<StepAction::handled_type>(action.get()));
338  else if ( seqType.second == "PhysicsList" )
339  kernel.physicsList().adopt(_action<PhysicsList::handled_type>(action.get()));
340  else if ( sdSeq.get() )
341  sdSeq->adopt(_action<Sensitive::handled_type>(action.get()));
342  else {
343  throw runtime_error(format("ActionSequence","DDG4: The action '%s'"
344  " cannot be attached to any sequence '%s'."
345  " [Sequence-Missing]",nam.c_str(), seqNam.c_str()));
346  }
347  printout(INFO,"Geant4Setup","+++ ActionSequence %s added filter object:%s",
348  seqType.second.c_str(),action->name().c_str());
349  }
350  if ( what == SENSITIVE ) {
351  for(xml_coll_t a(seq,_Unicode(filter)); a; ++a) {

473  template <> void Converter<XMLSetup>::operator()(xml_h seq) const {
474  xml_elt_t compact(seq);
475  // First execute the basic setup from the plugins module
476  long result = PluginService::Create<long>("geant4_XML_reader",&lcdd,&seq);
477  if ( 0 == result ) {
478  throw runtime_error("DD4hep: Failed to locate plugin to interprete files of type"
