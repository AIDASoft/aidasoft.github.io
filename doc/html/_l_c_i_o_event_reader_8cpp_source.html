41  map<EVENT::MCParticle*,int>::const_iterator ip=mcparts.find(part);
42  if ( ip == mcparts.end() ) {
43  throw runtime_error("Unknown particle identifier look-up!");

62 LCIOEventReader::EventReaderStatus
63 LCIOEventReader::readParticles(int event_number,
64  Vertex& /* primary_vertex */,
65  vector<Particle*>& particles)
66 {
67  EVENT::LCCollection* primaries = 0;
68  map<EVENT::MCParticle*,int> mcparts;
69  vector<EVENT::MCParticle*> mcpcoll;
70  EventReaderStatus ret = readParticleCollection(event_number,&primaries);
71 
72  if ( ret != EVENT_READER_OK ) return ret;
73 
74  int NHEP = primaries->getNumberOfElements();
75  // check if there is at least one particle
76  if ( NHEP == 0 ) return EVENT_READER_NO_PRIMARIES;
77 
78  mcpcoll.resize(NHEP,0);
79  for(int i=0; i<NHEP; ++i ) {
80  EVENT::MCParticle* p = dynamic_cast<EVENT::MCParticle*>(primaries->getElementAt(i));
81  mcparts[p] = i;
82  mcpcoll[i] = p;
83  }
84 
85  // build collection of MCParticles
86  for(int i=0; i<NHEP; ++i ) {
87  EVENT::MCParticle* mcp = mcpcoll[i];
88  Geant4ParticleHandle p(new Particle(i));
89  const double *mom = mcp->getMomentum();
90  const double *vsx = mcp->getVertex();
91  const double *vex = mcp->getEndpoint();
92  const float *spin = mcp->getSpin();
93  const int *color = mcp->getColorFlow();
94  const int pdg = mcp->getPDG();
95  PropertyMask status(p->status);
96  p->pdgID = pdg;
97  p->charge = int(mcp->getCharge()*3.0);
98  p->psx = mom[0]*CLHEP::GeV;
99  p->psy = mom[1]*CLHEP::GeV;
100  p->psz = mom[2]*CLHEP::GeV;
101  p->time = mcp->getTime()*CLHEP::ns;
102  p->properTime = mcp->getTime()*CLHEP::ns;
103  p->vsx = vsx[0]*CLHEP::mm;
104  p->vsy = vsx[1]*CLHEP::mm;
105  p->vsz = vsx[2]*CLHEP::mm;
106  p->vex = vex[0]*CLHEP::mm;
107  p->vey = vex[1]*CLHEP::mm;
108  p->vez = vex[2]*CLHEP::mm;
109  p->process = 0;
110  p->spin[0] = spin[0];
111  p->spin[1] = spin[1];
112  p->spin[2] = spin[2];
113  p->colorFlow[0] = color[0];
114  p->colorFlow[0] = color[1];
115  p->mass = mcp->getMass()*CLHEP::GeV;
116  const EVENT::MCParticleVec &par = mcp->getParents(), &dau=mcp->getDaughters();
117  for(int num=dau.size(),k=0; k<num; ++k)
118  p->daughters.insert(GET_ENTRY(mcparts,dau[k]));
119  for(int num=par.size(),k=0; k<num; ++k)
120  p->parents.insert(GET_ENTRY(mcparts,par[k]));
121 
122  int genStatus = mcp->getGeneratorStatus();
123  if ( genStatus == 0 ) status.set(G4PARTICLE_GEN_EMPTY);
124  else if ( genStatus == 1 ) status.set(G4PARTICLE_GEN_STABLE);
125  else if ( genStatus == 2 ) status.set(G4PARTICLE_GEN_DECAYED);
126  else if ( genStatus == 3 ) status.set(G4PARTICLE_GEN_DOCUMENTATION);
127  else {
128  cout << " #### WARNING - LCIOInputAction : unknown generator status : "
129  << genStatus << " -> ignored ! " << endl;
130  }
131  if ( mcp->isCreatedInSimulation() ) status.set(G4PARTICLE_SIM_CREATED);
132  if ( mcp->isBackscatter() ) status.set(G4PARTICLE_SIM_BACKSCATTER);
133  if ( mcp->vertexIsNotEndpointOfParent() ) status.set(G4PARTICLE_SIM_PARENT_RADIATED);
134  if ( mcp->isDecayedInTracker() ) status.set(G4PARTICLE_SIM_DECAY_TRACKER);
135  if ( mcp->isDecayedInCalorimeter() ) status.set(G4PARTICLE_SIM_DECAY_CALO);
136  if ( mcp->hasLeftDetector() ) status.set(G4PARTICLE_SIM_LEFT_DETECTOR);
137  if ( mcp->isStopped() ) status.set(G4PARTICLE_SIM_STOPPED);
138  if ( mcp->isOverlay() ) status.set(G4PARTICLE_SIM_OVERLAY);
139  particles.push_back(p);
140  }
141  return EVENT_READER_OK;
142 }
