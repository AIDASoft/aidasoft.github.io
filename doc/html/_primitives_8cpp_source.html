33 long int DD4hep::makeTime(int year, int month, int day,
34  int hour, int minutes, int seconds)
35 {
36  struct tm tm_init;
37  ::memset(&tm_init,0,sizeof(tm_init));
38  tm_init.tm_year = year > 1900 ? year-1900 : year;
39  tm_init.tm_mon = month;
40  tm_init.tm_mday = day;
41  tm_init.tm_hour = hour;
42  tm_init.tm_min = minutes;
43  tm_init.tm_sec = seconds;
44  tm_init.tm_isdst = -1;
45  long int ti = ::mktime(&tm_init);
46  if ( ti >= 0 ) return ti;
47  except("DD4hep","Invalid time data given for conversion to epoch: %d-%d-%d %02d:%02d:%02d",
48  year, month, day, hour, minutes, seconds);
49  return ti;
50 }

69 static const std::string __typeinfoName(const std::type_info& tinfo) {
70  const char* class_name = tinfo.name();
71  std::string result;
72 #ifdef WIN32
73  std::size_t off = 0;
74  if ( ::strncmp(class_name, "class ", 6) == 0 ) {
75  // The returned name is prefixed with "class "
76  off = 6;
77  }
78  if ( ::strncmp(class_name, "struct ", 7) == 0 ) {
79  // The returned name is prefixed with "struct "
80  off = 7;
81  }
82  if ( off != std::string::npos ) {
83  std::string tmp = class_name + off;
84  size_t loc = 0;
85  while( (loc = tmp.find("class ")) != std::string::npos ) {
86  tmp.erase(loc, 6);
87  }
88  loc = 0;
89  while( (loc = tmp.find("struct ")) != std::string::npos ) {
90  tmp.erase(loc, 7);
91  }
92  result = tmp;
93  }
94  else {
95  result = class_name;
96  }
97  // Change any " *" to "*"
98  while ( (off=result.find(" *")) != std::string::npos ) {
99  result.replace(off, 2, "*");
100  }
101  // Change any " &" to "&"
102  while ( (off=result.find(" &")) != std::string::npos ) {
103  result.replace(off, 2, "&");
104  }
105 #elif defined(sun)
106  result = class_name;
107 #elif !defined(__ICC)
108  if (::strlen(class_name) == 1) {
109  // See http://www.realitydiluted.com/mirrors/reality.sgi.com/dehnert_engr/cxx/abi.pdf
110  // for details
111  switch (class_name[0]) {
112  case 'v':
113  result = "void";
114  break;
115  case 'w':
116  result = "wchar_t";
117  break;
118  case 'b':
119  result = "bool";
120  break;
121  case 'c':
122  result = "char";
123  break;
124  case 'h':
125  result = "unsigned char";
126  break;
127  case 's':
128  result = "short";
129  break;
130  case 't':
131  result = "unsigned short";
132  break;
133  case 'i':
134  result = "int";
135  break;
136  case 'j':
137  result = "unsigned int";
138  break;
139  case 'l':
140  result = "long";
141  break;
142  case 'm':
143  result = "unsigned long";
144  break;
145  case 'x':
146  result = "long long";
147  break;
148  case 'y':
149  result = "unsigned long long";
150  break;
151  case 'n':
152  result = "__int128";
153  break;
154  case 'o':
155  result = "unsigned __int128";
156  break;
157  case 'f':
158  result = "float";
159  break;
160  case 'd':
161  result = "double";
162  break;
163  case 'e':
164  result = "long double";
165  break;
166  case 'g':
167  result = "__float128";
168  break;
169  case 'z':
170  result = "ellipsis";
171  break;
172  }
173  }
174  else {
175  char buff[16 * 1024];
176  std::size_t len = sizeof(buff);
177  int status = 0;
178  result = __cxxabiv1::__cxa_demangle(class_name, buff, &len, &status);
179  }
180 #else
181  result = class_name;
182  throw std::runtime_error("CXXABI is missing for ICC!");
183 #endif
184  return result;
185 }
