77 Utilities::createEveShape(int level, int max_level, TEveElement* p, TGeoNode* n, TGeoHMatrix mat, const std::string& nam) {
78  TGeoVolume* vol = n ? n->GetVolume() : 0;
79  bool created = false;
80 
81  if ( 0 == vol || level > max_level ) return make_pair(created,(TEveElement*)0);
82 
83  VisAttr vis(Volume(vol).visAttributes());
84  TGeoShape* geoShape = vol->GetShape();
85  TEveElement* element = 0;
86 
87  if ( p ) {
88  TGeoNode* pn = (TGeoNode*)p->GetUserData();
89  if ( pn == n ) {
90  element = p;
91  }
92  if ( !element ) {
93  element = (TEveGeoShape*)p->FindChild(n->GetName());
94  }
95  if ( !element && !nam.empty() ) {
96  element = (TEveGeoShape*)p->FindChild(nam.c_str());
97  }
98  if ( element ) goto Daughters;
99  }
100 
101  if ( geoShape->IsA() == TGeoShapeAssembly::Class() ) {
102  //printout(INFO,"createEveShape","+++ Assembly Shape %s Userdata:%p.",n->GetName(),n);
103  ElementList* shape = new ElementList(n->GetName(),n->GetName(),true,true);
104  shape->SetUserData(n);
105  shape->SetMainTransparency(true);
106  shape->SetMainAlpha(0.2);
107  shape->SetPickable(kTRUE);
108  if ( vis.isValid() ) {
109  float r,g,b;
110  vis.rgb(r,g,b);
111  shape->SetMainColorRGB(r,g,b);
112  }
113  element = shape;
114  created = true;
115  goto Daughters;
116  }
117  else if ( 0 == element ) {
118  TEveGeoShape* shape = new TEveGeoShape(n->GetName());
119  //printout(INFO,"createEveShape","+++ Create TEveGeoShape %s [%s] Userdata:%p.",
120  //n->GetName(),geoShape->IsA()->GetName(),n);
121  created = true;
122  if ( vis.isValid() ) {
123  float r,g,b;
124  vis.rgb(r,g,b);
125  shape->SetMainColorRGB(r,g,b);
126  }
127  shape->SetMainTransparency(true);
128  shape->SetMainAlpha(0.2);
129  shape->SetPickable(kTRUE);
130  shape->RefMainTrans().SetFrom(mat);
131  shape->SetShape((TGeoShape*)geoShape->Clone());
132  if ( level < max_level ) {
133  shape->SetRnrSelfChildren(true,true);
134  }
135  else if ( level == max_level ) {
136  shape->SetRnrSelfChildren(true,false);
137  }
138  else if ( level > max_level ) {
139  shape->SetRnrSelfChildren(false,false);
140  }
141  shape->SetUserData(n);
142  element = shape;
143  }
144  else {
145  printout(INFO,"createEveShape","Weird!! %s",n->GetName());
146  }
147 
148  Daughters:
149  for (Int_t idau = 0, ndau = n->GetNdaughters(); idau < ndau; ++idau) {
150  TGeoNode* daughter = n->GetDaughter(idau);
151  TGeoHMatrix dau_mat(mat);
152  TGeoMatrix* matrix = daughter->GetMatrix();
153  dau_mat.Multiply(matrix);
154  pair<bool,TEveElement*> dau_shape =
155  createEveShape(level+1, max_level, element, daughter, dau_mat, daughter->GetName());
156  if ( dau_shape.first ) {
157  element->AddElement(dau_shape.second);
158  }
159  }
160  return make_pair(created,element);
161 }

163 int Utilities::findNodeWithMatrix(TGeoNode* p, TGeoNode* n, TGeoHMatrix* mat, string* sub_path) {
164  if ( p == n ) return 1;
165  TGeoHMatrix dau_mat;
166  for (Int_t idau = 0, ndau = p->GetNdaughters(); idau < ndau; ++idau) {
167  string spath;
168  TGeoNode* daughter = p->GetDaughter(idau);
169  TGeoHMatrix* daughter_matrix = 0;
170  if ( mat ) {
171  TGeoMatrix* matrix = daughter->GetMatrix();
172  dau_mat = *mat;
173  dau_mat.Multiply(matrix);
174  daughter_matrix = &dau_mat;
175  }
176  int level = findNodeWithMatrix(daughter,n,daughter_matrix,sub_path ? &spath : 0);
177  if ( level>0 ) {
178  if ( sub_path ) {
179  *sub_path += "/";
180  *sub_path += spath;
181  }
182  if ( mat ) *mat = *daughter_matrix;
183  return level+1;
184  }
185  }
186  return 0;
187 }
