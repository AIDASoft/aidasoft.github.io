50 static pair<TClass*,void*> load(TBranch* branch, int entry) {
51  TClass* cl = gROOT->GetClass(branch->GetClassName(),kTRUE);
52  if ( !cl ) {
53  return pair<TClass*,void*>(0,0);
54  }
55  void *obj = cl->New();
56  branch->SetAddress(&obj);
57  branch->SetAutoDelete(kFALSE);
58  int nb = branch->GetEntry(entry);
59  if ( nb < 0 ) {
60  cl->Destructor(obj);
61  return pair<TClass*,void*>(0,0);
62  }
63  return pair<TClass*,void*>(cl,obj);
64 }

66 static long dump_root(DD4hep::Geometry::LCDD&, int argc, char** argv) {
67  std::string input = "", tag="Geant4ROOTDump";
68  int entry = -1;
69 
70  for(int j=0; j<argc; ++j) {
71  std::string a = argv[j];
72  if ( a[0]=='-' ) a = argv[j]+1;
73  if ( a[0]=='-' ) a = argv[j]+2;
74  size_t idx = a.find('=');
75  if ( strncmp(a.c_str(),"usage",5)==0 ) {
76  usage();
77  return 1;
78  }
79  if ( idx > 0 ) {
80  string p1 = a.substr(0,idx);
81  string p2 = a.substr(idx+1);
82  if ( strncmp(p1.c_str(),"input",3)==0 ) {
83  input = p2;
84  }
85  if ( strncmp(p1.c_str(),"entry",5)==0 ) {
86  if ( 1 != ::sscanf(p2.c_str(),"%d",&entry) ) {
87  printout(FATAL,tag,"+++ Argument %s is not properly formatted. must be --entry=<number>.",argv[j]);
88  return usage();
89  }
90  }
91  continue;
92  }
93  printout(FATAL,tag,"+++ Argument %s is IGNORED! No value is found (string has no '=')",argv[j]);
94  return usage();
95  }
96  const char* line = "+----------------------------------------------------------------------------------+";
97  printout(INFO,tag,line);
98  printout(INFO,tag,"| Input:%-74s|",input.c_str());
99  printout(INFO,tag,line);
100 
101  if ( input.empty() ) {
102  return usage();
103  }
104 
105 
106  Geant4DataDump dump("Geant4Data");
107  TFile* f = TFile::Open(input.c_str());
108 
109  if ( f && !f->IsZombie() ) {
110  TTree* tree = (TTree*)f->Get("EVENT");
111  TClass* cl_calo = gROOT->GetClass(typeid(CalorimeterHits));
112  TClass* cl_tracker = gROOT->GetClass(typeid(TrackerHits));
113  TClass* cl_particles = gROOT->GetClass(typeid(Particles));
114  TObjArray* branches = tree->GetListOfBranches();
115  Int_t nbranches = branches->GetEntriesFast();
116  typedef pair<TClass*,void*> ENTRY;
117  typedef map<string,ENTRY> ENTRIES;
118 
119  for(Int_t ievt=entry<0 ? 0 : entry, nevt=entry<0 ? tree->GetEntries() : entry+1; ievt<nevt; ++ievt) {
120  ENTRIES event;
121  printout(INFO,tag,line);
122  printout(INFO,tag,"| Event: %6d |",ievt);
123  printout(INFO,tag,line);
124 
125  // First suck in all data
126  for (Int_t i=0;i<nbranches;i++) {
127  TBranch* branch = (TBranch*)branches->UncheckedAt(i);
128  pair<TClass*,void*> data = load(branch,ievt);
129  if ( data.first ) event[branch->GetName()] = data;
130  }
131  // Now dump the stuff
132  for(ENTRIES::const_iterator i=event.begin(); i!=event.end(); ++i) {
133  pair<TClass*,void*> data = (*i).second;
134  if ( data.first == cl_particles ) {
135  Particles* parts = (Particles*)data.second;
136  dump.print(INFO, (*i).first, parts);
137  for_each(parts->begin(), parts->end(), DestroyObject<Geant4Particle*>());
138  }
139  }
140  for(ENTRIES::const_iterator i=event.begin(); i!=event.end(); ++i) {
141  pair<TClass*,void*> data = (*i).second;
142  if ( data.first == cl_particles ) {
143  }
144  else if ( data.first == cl_tracker ) {
145  TrackerHits* hits = (TrackerHits*)data.second;
146  dump.print(INFO, (*i).first, hits);
147  for_each(hits->begin(), hits->end(), DestroyObject<Geant4Tracker::Hit*>());
148  }
149  else if ( data.first == cl_calo ) {
150  CalorimeterHits* hits = (CalorimeterHits*)data.second;
151  dump.print(INFO, (*i).first, hits);
152  for_each(hits->begin(), hits->end(), DestroyObject<Geant4Calorimeter::Hit*>());
153  }
154  if ( data.first ) data.first->Destructor(data.second);
155  }
156  }
157  delete f;
158  }
159  else {
160  printout(FATAL,tag,"+++ FAILED to open input file:%s",input.c_str());
161  return usage();
162  }
163  return 1;
164 }
