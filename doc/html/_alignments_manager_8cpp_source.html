106 void AlignmentsManagerObject::to_world(AlignContext& new_alignments, UserPool& pool, DetElement det, TGeoHMatrix& mat) const {
107  using Conditions::Condition;
108  DetElement par = det.parent();
109  while( par.isValid() ) {
110  // If we find that the parent also got updated, directly take this transformation.
111  // Since we compute top-down, it is already valid!
112  AlignContext::Keys::const_iterator i = new_alignments.keys.find(par.key());
113  if ( i != new_alignments.keys.end() ) {
114  const AlignContext::Entry& e = new_alignments.entries[(*i).second];
115  AlignmentCondition cond(e.cond);
116  AlignmentData& align = cond.data();
117  mat.MultiplyLeft(&align.worldTransformation());
118  return;
119  }
120  // The parent did not get updated: We have to search the the conditions pool if
121  // there is a still valid condition, which we can use to build the world transformation
122  // The parent's alignment condition by defintiion must be present in the pool,
123  // since it got updated in the past!
124  i = all_alignments->keys.find(par.key());
125  if ( i != all_alignments->keys.end() ) {
126  const AlignContext::Entry& e = all_alignments->entries[(*i).second];
127  Condition::key_type key = e.dep->target.hash;
128  AlignmentCondition cond = pool.get(key);
129  AlignmentData& align = cond.data();
130  mat.MultiplyLeft(&align.worldTransformation());
131  return;
132  }
133  // There is no special alignment for this detector element.
134  // Hence to nominal (relative) transformation to the parent is valid
135  mat.MultiplyLeft(&par.nominal().detectorTransformation());
136  par = par.parent();
137  }
138 }

146 void AlignmentsManagerObject::compute(UserPool& pool, const Dependencies& deps) const {
147  AlignContext::DetectorMap::const_iterator i;
148  AlignContext new_alignments;
149  new_alignments.entries.reserve(deps.size());
150  pool.compute(deps, &new_alignments);
151  std::string prev = "-----";
152  for(i=new_alignments.detectors.begin(); i!=new_alignments.detectors.end(); ++i) {
153  AlignContext::Entry& e = new_alignments.entries[(*i).second];
154  DetElement det = e.det;
155  const std::string& p = det.path();
156  size_t idx = p.find(prev);
157  if ( idx == 0 ) {
158  continue;
159  }
160  prev = p;
161  printout(DEBUG,"Alignment","Update top Node: Lvl:%d Key:%08X: %s", det.level(), det.key(), p.c_str());
162  e.top = 1;
163  }
164  // We got now the top nodes of the new_alignments. From the top nodes we have to
165  // recursively calculate all changes downwards the lower levels!
166  // Note: The upper levels are already correct and do not need to be updated!
167  printout(INFO,"Alignment","Working down the tree....");
168  for(i=new_alignments.detectors.begin(); i != new_alignments.detectors.end(); ++i) {
169  AlignContext::Entry& e = new_alignments.entries[(*i).second];
170  if ( e.top ) {
171  compute(new_alignments, pool, e.det, 0);
172  }
173  }
174 }
176 void computeDelta(AlignmentCondition cond, TGeoHMatrix& tr_delta) {
177  const AlignmentData& align = cond.data();
178  const Delta& delta = align.delta;
179  const TGeoHMatrix& nom = align.detectorTransformation();
180  const Position& pos = delta.translation;
181  const Translation3D& piv = delta.pivot;
182  const RotationZYX& rot = delta.rotation;
183 
184  switch(delta.flags) {
185  case Delta::HAVE_TRANSLATION+Delta::HAVE_ROTATION+Delta::HAVE_PIVOT:
186  Geometry::_transform(tr_delta, Transform3D(Translation3D(pos)*piv*rot*(piv.Inverse())));
187  break;
188  case Delta::HAVE_TRANSLATION+Delta::HAVE_ROTATION:
189  Geometry::_transform(tr_delta, Transform3D(rot,pos));
190  break;
191  case Delta::HAVE_ROTATION+Delta::HAVE_PIVOT:
192  Geometry::_transform(tr_delta, Transform3D(piv*rot*(piv.Inverse())));
193  break;
194  case Delta::HAVE_ROTATION:
195  Geometry::_transform(tr_delta, rot);
196  break;
197  case Delta::HAVE_TRANSLATION:
198  Geometry::_transform(tr_delta, pos);
199  break;
200  default:
201  break;
202  }
203  tr_delta.MultiplyLeft(&nom);
204  align.detectorTrafo = tr_delta;
205  align.worldTrafo = tr_delta;
206 }
208 void AlignmentsManagerObject::compute(AlignContext& new_alignments, UserPool& pool, DetElement det, int level) const {
209  AlignContext::Keys::const_iterator k=new_alignments.keys.find(det.key());
210  bool has_cond = k != new_alignments.keys.end();
211  const AlignContext::Entry* ent = has_cond ? &new_alignments.entries[(*k).second] : 0;
212 
213  if ( ent ) {
214  char fmt[128];
215  TGeoHMatrix tr_delta;
216  AlignmentCondition cond(ent->cond);
217  AlignmentData& align = cond.data();
218  computeDelta(cond, tr_delta);
219  to_world(new_alignments, pool, det, align.worldTrafo);
220  align.trToWorld = Geometry::_transform(&align.worldTrafo);
221  ::snprintf(fmt,sizeof(fmt),"%%d %%%ds %%s %%08X: %%s IOV:%%s",2*level);
222  printout(DEBUG,"ComputeAlignment",fmt,
223  det.level(), "", has_cond ? "NO " : "YES",
224  det.key(), det.path().c_str(), cond.iov().str().c_str());
225  }
226  else {
227  // DetElement 'det' has no specific alignment. If any of the children has one
228  // and any of the parents, 'to_world' should pick up all the proper components
229  // to build the proper world-transformation of the child.
230  // Hence, we do not have to recompute it's value.
231  // We just continue to trickle down to the children.
232  //
233  // Alternatively we could inject 'special' alignment conditions, which would depend
234  // on the parent...
235  // Under circumstances, this might be cheaper to re-compute.
236  }
237  const DetElement::Children& children = det.children();
238  for(auto c=children.begin(); c!=children.end(); ++c) {
239  DetElement child = (*c).second;
240  compute(new_alignments, pool, child, level+1);
241  }
242 }
