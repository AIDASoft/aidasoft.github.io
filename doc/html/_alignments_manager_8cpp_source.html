00106 void AlignmentsManagerObject::to_world(AlignContext& new_alignments, UserPool& pool, DetElement det, TGeoHMatrix& mat)  const  {
00107   using Conditions::Condition;
00108   DetElement par = det.parent();
00109   while( par.isValid() )   {
00110     // If we find that the parent also got updated, directly take this transformation.
00111     // Since we compute top-down, it is already valid!
00112     AlignContext::Keys::const_iterator i = new_alignments.keys.find(par.key());
00113     if ( i != new_alignments.keys.end() )  {
00114       const AlignContext::Entry& e = new_alignments.entries[(*i).second];
00115       AlignmentCondition cond(e.cond);
00116       AlignmentData&     align = cond.data();
00117       mat.MultiplyLeft(&align.worldTransformation());
00118       return;
00119     }
00120     // The parent did not get updated: We have to search the the conditions pool if
00121     // there is a still valid condition, which we can use to build the world transformation
00122     // The parent's alignment condition by defintiion must be present in the pool,
00123     // since it got updated in the past!
00124     i = all_alignments->keys.find(par.key());
00125     if ( i != all_alignments->keys.end() )  {
00126       const AlignContext::Entry& e = all_alignments->entries[(*i).second];
00127       Condition::key_type key = e.dep->target.hash;
00128       AlignmentCondition cond = pool.get(key);
00129       AlignmentData&    align = cond.data();
00130       mat.MultiplyLeft(&align.worldTransformation());
00131       return;
00132     }
00133     // There is no special alignment for this detector element.
00134     // Hence to nominal (relative) transformation to the parent is valid
00135     mat.MultiplyLeft(&par.nominal().detectorTransformation());
00136     par = par.parent();
00137   }

00164   // We got now the top nodes of the new_alignments. From the top nodes we have to
00165   // recursively calculate all changes downwards the lower levels!
00166   // Note: The upper levels are already correct and do not need to be updated!

00169     AlignContext::Entry& e = new_alignments.entries[(*i).second];
00170     if ( e.top )     {
00171       compute(new_alignments, pool, e.det, 0);

00176 void computeDelta(AlignmentCondition cond, TGeoHMatrix& tr_delta)  {
00177   const AlignmentData& align = cond.data();
00178   const Delta&         delta = align.delta;
00179   const TGeoHMatrix&     nom = align.detectorTransformation();
00180   const Position&        pos = delta.translation;
00181   const Translation3D&   piv = delta.pivot;
00182   const RotationZYX&     rot = delta.rotation;
00183 
00184   switch(delta.flags)   {
00185   case Delta::HAVE_TRANSLATION+Delta::HAVE_ROTATION+Delta::HAVE_PIVOT:
00186     Geometry::_transform(tr_delta, Transform3D(Translation3D(pos)*piv*rot*(piv.Inverse())));
00187     break;
00188   case Delta::HAVE_TRANSLATION+Delta::HAVE_ROTATION:
00189     Geometry::_transform(tr_delta, Transform3D(rot,pos));
00190     break;
00191   case Delta::HAVE_ROTATION+Delta::HAVE_PIVOT:
00192     Geometry::_transform(tr_delta, Transform3D(piv*rot*(piv.Inverse())));
00193     break;
00194   case Delta::HAVE_ROTATION:
00195     Geometry::_transform(tr_delta, rot);
00196     break;
00197   case Delta::HAVE_TRANSLATION:
00198     Geometry::_transform(tr_delta, pos);
00199     break;
00200   default:
00201     break;
00202   }
00203   tr_delta.MultiplyLeft(&nom);
00204   align.detectorTrafo = tr_delta;
00205   align.worldTrafo    = tr_delta;
00206 }
00208 void AlignmentsManagerObject::compute(AlignContext& new_alignments, UserPool& pool, DetElement det, int level) const  {
00209   AlignContext::Keys::const_iterator k=new_alignments.keys.find(det.key());
00210   bool has_cond    = k != new_alignments.keys.end();
00211   const AlignContext::Entry* ent = has_cond ? &new_alignments.entries[(*k).second] : 0;
00212 
00213   if ( ent )  {
00214     char fmt[128];
00215     TGeoHMatrix        tr_delta;
00216     AlignmentCondition cond(ent->cond);
00217     AlignmentData&     align = cond.data();
00218     computeDelta(cond, tr_delta);
00219     to_world(new_alignments, pool, det, align.worldTrafo);
00220     align.trToWorld = Geometry::_transform(&align.worldTrafo);
00221     ::snprintf(fmt,sizeof(fmt),"%%d %%%ds %%s %%08X: %%s IOV:%%s",2*level);
00222     printout(DEBUG,"ComputeAlignment",fmt,
00223              det.level(), "", has_cond ? "NO " : "YES",
00224              det.key(), det.path().c_str(), cond.iov().str().c_str());
00225   }
00226   else  {
00227     // DetElement 'det' has no specific alignment. If any of the children has one
00228     // and any of the parents, 'to_world' should pick up all the proper components
00229     // to build the proper world-transformation of the child.
00230     // Hence, we do not have to recompute it's value.
00231     // We just continue to trickle down to the children.
00232     //
00233     // Alternatively we could inject 'special' alignment conditions, which would depend
00234     // on the parent...
00235     // Under circumstances, this might be cheaper to re-compute.
00236   }
00237   const DetElement::Children& children = det.children();
00238   for(auto c=children.begin(); c!=children.end(); ++c)    {
00239     DetElement child = (*c).second;
00240     compute(new_alignments, pool, child, level+1);
00241   }
