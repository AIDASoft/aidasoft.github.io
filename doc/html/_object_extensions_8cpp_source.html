83 void* ObjectExtensions::addExtension(void* ptr, const std::type_info& info, copy_t ctor, destruct_t dtor) {
84  Extensions::iterator j = extensions.find(&info);
85  if (j == extensions.end()) {
86  ExtensionMap::iterator i = extensionMap->find(&info);
87  if (i == extensionMap->end()) {
88  Entry entry;
89  entry.destruct = dtor;
90  entry.copy = ctor;
91  entry.id = ++s_extensionID;
92  extensionMap->insert(make_pair(&info, entry));
93  i = extensionMap->find(&info);
94  }
95  return extensions[&info] = ptr;
96  }
97  throw std::runtime_error("DD4hep: addExtension: Object already has an extension of type:" + typeName(info) + ".");
98 }

101 void* ObjectExtensions::removeExtension(const std::type_info& info, bool destroy) {
102  Extensions::iterator j = extensions.find(&info);
103  if (j != extensions.end()) {
104  void *ptr = (*j).second;
105  if ( destroy ) {
106  ExtensionMap::iterator i = extensionMap->find(&info);
107  if (i != extensionMap->end()) {
108  Entry& e = (*i).second;

112  }
113  extensions.erase(j);
114  return ptr;

116  throw std::runtime_error("DD4hep: removeExtension: The object of type " + typeName(info) + " is not present.");
