183  template <> bool Geant4SensitiveAction<Geant4OpticalCalorimeter>::process(G4Step* step,G4TouchableHistory*) {
184  G4Track * track = step->GetTrack();
185  // check that particle is optical photon:
186  if( track->GetDefinition() != G4OpticalPhoton::OpticalPhotonDefinition() ) {
187  return false;
188  }
189  else if ( track->GetCreatorProcess()->G4VProcess::GetProcessName() != "Cerenkov") {
190  track->SetTrackStatus(fStopAndKill);
191  return false;
192  }
193  else {
194  typedef Geant4Calorimeter::Hit Hit;
195  StepHandler h(step);
196  HitCollection* coll = collection(m_collectionID);
197  HitContribution contrib = Hit::extractContribution(step);
198  Position pos = h.prePos();
199  Hit* hit = coll->find<Hit>(PositionCompare<Hit,Position>(pos));
200  if ( !hit ) {
201  hit = new Hit(pos);
202  hit->cellID = volumeID(step);
203  coll->add(hit);
204  if ( 0 == hit->cellID ) {
205  hit->cellID = volumeID(step);
206  except("+++ Invalid CELL ID for hit!");
207  }
208  }
209  hit->energyDeposit += contrib.deposit;
210  hit->truth.push_back(contrib);
211  track->SetTrackStatus(fStopAndKill); // don't step photon any further
212  mark(h.track);
213  return true;
214  }

335  void start(G4Step* step, G4StepPoint* point) {
336  pre.storePoint(step,point);
337  pre.truth.deposit = 0.0;
338  current = pre.truth.trackID;
339  sensitive->mark(step->GetTrack());
340  mean_pos.SetXYZ(0,0,0);
341  mean_time = 0;
342  post = pre;
343  combined = 0;
344  cell = 0;
345  }
346 
348  void update(G4Step* step) {
349  post.storePoint(step,step->GetPostStepPoint());
350  pre.truth.deposit += post.truth.deposit;
351  mean_pos.SetX(mean_pos.x()+post.position.x()*post.truth.deposit);
352  mean_pos.SetY(mean_pos.y()+post.position.y()*post.truth.deposit);
353  mean_pos.SetZ(mean_pos.z()+post.position.z()*post.truth.deposit);
354  mean_time += post.truth.time*post.truth.deposit;
355  if ( 0 == cell ) {
356  cell = sensitive->cellID(step);
357  if ( 0 == cell ) {
358  cell = sensitive->volumeID(step) ;
359  sensitive->except("+++ Invalid CELL ID for hit!");
360  }
361  }
362  ++combined;
363  }
364 
366  void clear() {
367  mean_pos.SetXYZ(0,0,0);
368  mean_time = 0;
369  post.clear();
370  pre.clear();
371  current = -1;
372  combined = 0;
373  cell = 0;
374  }

377  bool mustSaveTrack(const G4Track* tr) const {
378  return current > 0 && current != tr->GetTrackID();
379  }

382  void extractHit(Geant4HitCollection* collection) {
383  if ( current == -1 ) {
384  return;
385  }
386  double deposit = pre.truth.deposit;
387  double time = deposit != 0 ? mean_time / deposit : mean_time;
388  Position pos = deposit != 0 ? mean_pos / deposit : mean_pos;
389  Momentum mom = 0.5 * (pre.momentum + post.momentum);
390  double path_len = (post.position - pre.position).R();
391  Geant4Tracker::Hit* hit = new Geant4Tracker::Hit(pre.truth.trackID,
392  pre.truth.pdgID,
393  deposit,time);
394  hit->position = pos;
395  hit->momentum = mom;
396  hit->length = path_len;
397  hit->cellID = cell;
398  collection->add(hit);
399  sensitive->printM2("+++ TrackID:%6d [%s] CREATE hit combination with %2d deposit(s):"
400  " %e MeV Pos:%8.2f %8.2f %8.2f",
401  pre.truth.trackID,sensitive->c_name(),combined,pre.truth.deposit/CLHEP::MeV,
402  pos.X()/CLHEP::mm,pos.Y()/CLHEP::mm,pos.Z()/CLHEP::mm);
403  clear();
404  }

408  G4bool process(G4Step* step, G4TouchableHistory* ) {
409  Geant4StepHandler h(step);
410 
411  // std::cout << " process called - pre pos: " << h.prePos() << " post pos " << h.postPos()
412  // << " edep: " << h.deposit() << std::endl ;
413 
414  void *prePV = h.volume(h.pre), *postPV = h.volume(h.post);
415 
416  Geant4HitCollection* coll = sensitive->collection(0);
418  if ( mustSaveTrack(h.track) ) {
419  extractHit(coll);
420  }
422  if ( current < 0 ) {
423  start(step, h.pre);
424  }
426  update(step);
427 
428  if ( prePV != postPV ) {
429  void* postSD = h.sd(h.post);
430  extractHit(coll);
431  if ( 0 != postSD ) {
432  void* preSD = h.sd(h.pre);
433  if ( preSD == postSD ) {
434  start(step,h.post);
435  }
436  }
437  }
438  else if ( h.track->GetTrackStatus() == fStopAndKill ) {
439  extractHit(coll);
440  }
441  return true;
442  }

445  void endEvent(const G4Event* /* event */) {
446  // We need to add the possibly last added hit to the collection here.
447  // otherwise the last hit would be assigned to the next event and the
448  // MC truth would be screwed.
449  //
450  // Alternatively the 'update' method would become rather CPU consuming,
451  // beacuse the extract action would have to be recalculated over and over.
452  if ( current > 0 ) {
453  Geant4HitCollection* coll = sensitive->collection(0);
454  extractHit(coll);
455  }
456  }
