26 static Ref_t create_element(LCDD& lcdd, xml_h e, Ref_t) {
27  xml_det_t x_det (e);
28  string det_name = x_det.nameStr();
29  DetElement sdet(det_name, x_det.id());
30  Volume vol;
31 
32  bool useRot = x_det.hasChild(_U(rotation));
33  bool usePos = x_det.hasChild(_U(position));
34  Position pos;
35  RotationZYX rot;
36 
37  sdet.setType("compound");
38  XML::setDetectorTypeFlag( e, sdet ) ;
39 
40  if( usePos ) {
41  pos = Position(x_det.position().x(), x_det.position().y(), x_det.position().z());
42  }
43  if( useRot ) {
44  rot = RotationZYX(x_det.rotation().x(), x_det.rotation().y(), x_det.rotation().z());
45  }
46 
47  if ( x_det.hasChild(_U(shape)) ) {
48  xml_comp_t x_shape = x_det.child(_U(shape));
49  string type = x_shape.typeStr();
50  Solid solid = XML::createShape(lcdd, type, x_shape);
51  Material mat = lcdd.material(x_shape.materialStr());
52  printout(DEBUG,det_name,"+++ Creating detector assembly with shape of type:%s",type.c_str());
53  vol = Volume(det_name,solid,mat);
54  }
55  else {
56  printout(DEBUG,det_name,"+++ Creating detector assembly without shape");
57  vol = Assembly(det_name);
58  }
59 
60  for(xml_coll_t c(x_det,_U(composite)); c; ++c) {
61  xml_dim_t component = c;
62  string nam = component.nameStr();
63  lcdd.declareMotherVolume(nam, vol);
64  }
65 
66  vol.setAttributes(lcdd,x_det.regionStr(),x_det.limitsStr(),x_det.visStr());
67 
68  Volume mother = lcdd.pickMotherVolume(sdet);
69  PlacedVolume pv;
70  if( useRot && usePos ){
71  pv = mother.placeVolume(vol, Transform3D(rot, pos));
72  } else if( useRot ){
73  pv = mother.placeVolume(vol, rot);
74  } else if( usePos ){
75  pv = mother.placeVolume(vol, pos);
76  } else {
77  pv = mother.placeVolume(vol);
78  }
79 
80  sdet.setPlacement(pv);
81  return sdet;
82 }
