69 void Geant4GeneratorWrapper::operator()(G4Event* event) {
70  Geant4PrimaryEvent* prim = context()->event().extension<Geant4PrimaryEvent>();
71  Geant4PrimaryMap* primaryMap = context()->event().extension<Geant4PrimaryMap>();
72  set<G4PrimaryVertex*> primaries;
73 
75  for(G4PrimaryVertex* v=event->GetPrimaryVertex(); v; v=v->GetNext())
76  primaries.insert(v);
77 
78  // Now generate the new interaction
79  generator()->GeneratePrimaryVertex(event);
80 
82  for(G4PrimaryVertex* gv=event->GetPrimaryVertex(); gv; gv=gv->GetNext()) {
83  if ( primaries.find(gv) == primaries.end() ) {
84  Geant4PrimaryInteraction* inter = createPrimary(m_mask, primaryMap, gv);
85  prim->add(m_mask, inter);
86  }
87  }
