103  template <> lcio::LCCollectionVec*
104  Geant4DataConversion<lcio::LCCollectionVec,
105  pair<const Geant4Context*,Geant4HitCollection*>,
106  Geant4Tracker::Hit>::operator()(const arg_t& args) const {
107 
108  Geant4HitCollection* coll = args.second;
109  Geant4Sensitive* sd = coll->sensitive();
110  size_t nhits = coll->GetSize();
111  string dsc = encoding(sd->sensitiveDetector());
112  Geant4ParticleMap* pm = args.first->event().extension<Geant4ParticleMap>();
113  lcio::LCEventImpl* lc_evt = args.first->event().extension<lcio::LCEventImpl>();
114  EVENT::LCCollection* lc_part = lc_evt->getCollection(lcio::LCIO::MCPARTICLE);
115  lcio::LCCollectionVec* lc_coll = new lcio::LCCollectionVec(lcio::LCIO::SIMTRACKERHIT);
116  UTIL::CellIDEncoder<SimTrackerHit> decoder(dsc,lc_coll);
117  int hit_creation_mode = sd->hitCreationMode();
118 
119  if ( hit_creation_mode == Geant4Sensitive::DETAILED_MODE )
120  lc_coll->setFlag(UTIL::make_bitset32(LCIO::THBIT_MOMENTUM,LCIO::THBIT_ID1));
121  else
122  lc_coll->setFlag(LCIO::THBIT_ID1);
123 
124  lc_coll->reserve(nhits);
125  for(size_t i=0; i<nhits; ++i) {
126  const Geant4Tracker::Hit* hit = coll->hit(i);
127  const Geant4Tracker::Hit::Contribution& t = hit->truth;
128  int trackID = pm->particleID(t.trackID);
129  EVENT::MCParticle* lc_mcp = (EVENT::MCParticle*)lc_part->getElementAt(trackID);
130  double pos[3] = {hit->position.x()/CLHEP::mm, hit->position.y()/CLHEP::mm, hit->position.z()/CLHEP::mm};
131  lcio::SimTrackerHitImpl* lc_hit = new lcio::SimTrackerHitImpl;
132  lc_hit->setCellID0((hit->cellID >> 0 ) & 0xFFFFFFFF);
133  lc_hit->setCellID1((hit->cellID >> sizeof(int)*8) & 0xFFFFFFFF);
134  lc_hit->setEDep(hit->energyDeposit/CLHEP::GeV);
135  lc_hit->setPathLength(hit->length/CLHEP::mm);
136  lc_hit->setTime(hit->truth.time/CLHEP::ns);
137  lc_hit->setMCParticle(lc_mcp);
138  lc_hit->setPosition(pos);
139  lc_hit->setMomentum(hit->momentum.x()/CLHEP::GeV,hit->momentum.y()/CLHEP::GeV,hit->momentum.z()/CLHEP::GeV);
140  lc_coll->addElement(lc_hit);
141  }
142  return lc_coll;
143  }

156  Geant4DataConversion<lcio::LCCollectionVec,
157  pair<const Geant4Context*,Geant4HitCollection*>,
158  Geant4Calorimeter::Hit>::operator()(const arg_t& args) const {
159  typedef Geant4HitData::Contributions Contributions;
160  Geant4HitCollection* coll = args.second;
161  Geant4Sensitive* sd = coll->sensitive();
162  size_t nhits = coll->GetSize();
163  string dsc = encoding(sd->sensitiveDetector());
164  Geant4ParticleMap* pm = args.first->event().extension<Geant4ParticleMap>();
165  lcio::LCEventImpl* lc_evt = args.first->event().extension<lcio::LCEventImpl>();
166  EVENT::LCCollection* lc_parts = lc_evt->getCollection(lcio::LCIO::MCPARTICLE);
167  lcio::LCCollectionVec* lc_coll = new lcio::LCCollectionVec(lcio::LCIO::SIMCALORIMETERHIT);
168  UTIL::CellIDEncoder<SimCalorimeterHit> decoder(dsc,lc_coll);
169  int hit_creation_mode = sd->hitCreationMode();
170 
171  if ( hit_creation_mode == Geant4Sensitive::DETAILED_MODE )
172  lc_coll->setFlag(UTIL::make_bitset32(LCIO::CHBIT_LONG,LCIO::CHBIT_STEP,LCIO::CHBIT_ID1));
173  else
174  lc_coll->setFlag(UTIL::make_bitset32(LCIO::CHBIT_LONG,LCIO::CHBIT_ID1));
175 
176  lc_coll->reserve(nhits);
177  if ( sd->hasProperty("HitCreationMode") ) {
178  hit_creation_mode = sd->property("HitCreationMode").value<int>();
179  }
180  for(size_t i=0; i<nhits; ++i) {
181  const Geant4Calorimeter::Hit* hit = coll->hit(i);
182  float pos[3] = {float(hit->position.x()/mm), float(hit->position.y()/mm), float(hit->position.z()/mm)};
183  lcio::SimCalorimeterHitImpl* lc_hit = new lcio::SimCalorimeterHitImpl;
184  lc_hit->setCellID0((hit->cellID >> 0 ) & 0xFFFFFFFF);
185  lc_hit->setCellID1((hit->cellID >> sizeof(int)*8) & 0xFFFFFFFF); // ????
186  lc_hit->setPosition(pos);
188  lc_coll->addElement(lc_hit);
190  for(Contributions::const_iterator j=hit->truth.begin(); j!=hit->truth.end(); ++j) {
191  const Geant4HitData::Contribution& c = *j;
192  int trackID = pm->particleID(c.trackID);
193  EVENT::MCParticle* lc_mcp = (EVENT::MCParticle*)lc_parts->getElementAt(trackID);
194  if ( hit_creation_mode == Geant4Sensitive::DETAILED_MODE ) {
195  float contrib_pos[] = {float(c.x/mm), float(c.y/mm), float(c.z/mm)};
196  lc_hit->addMCParticleContribution(lc_mcp, c.deposit/GeV, c.time/ns, c.pdgID, contrib_pos);
197  }
198  else {
199  lc_hit->addMCParticleContribution(lc_mcp, c.deposit/GeV, c.time/ns);
200  }
201  }
202  }
203  return lc_coll;
204  }

206  template <typename T>
207  lcio::LCCollectionVec* moveEntries(Geant4HitCollection* coll,
208  lcio::LCCollectionVec* lc_coll)
209  {
210  size_t nhits = coll->GetSize();
211  lc_coll->reserve(nhits);
212  for(size_t i=0; i<nhits; ++i) {
213  Geant4HitWrapper& wrap = coll->hit(i);
214  T* lc_hit = wrap;
215  wrap.release(); // Now we have ownership!
216  lc_coll->addElement(lc_hit);
217  }
218  coll->clear(); // Since the collection now only contains NULL pointers, better clear it!
219  return lc_coll;
220  }
