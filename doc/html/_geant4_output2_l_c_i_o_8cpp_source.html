204 lcio::LCCollectionVec* Geant4Output2LCIO::saveParticles(Geant4ParticleMap* particles) {
205  typedef ReferenceBitMask<const int> PropertyMask;
206  typedef Geant4ParticleMap::ParticleMap ParticleMap;
207  const ParticleMap& pm = particles->particleMap;
208  size_t nparts = pm.size();
209  lcio::LCCollectionVec* lc_coll = new lcio::LCCollectionVec(lcio::LCIO::MCPARTICLE);
210  lc_coll->reserve(nparts);
211  if ( nparts > 0 ) {
212  size_t cnt = 0;
213  map<int,int> p_ids;
214  vector<const Geant4Particle*> p_part(pm.size(),0);
215  vector<MCParticleImpl*> p_lcio(pm.size(),0);
216  // First create the particles
217  for(ParticleMap::const_iterator i=pm.begin(); i!=pm.end();++i, ++cnt) {
218  int id = (*i).first;
219  const Geant4ParticleHandle p = (*i).second;
220  PropertyMask mask(p->status);
221  // std::cout << " ********** mcp status : 0x" << std::hex << p->status << ", mask.isSet(G4PARTICLE_GEN_STABLE) x" << std::dec << mask.isSet(G4PARTICLE_GEN_STABLE) <<std::endl ;
222  const G4ParticleDefinition* def = p.definition();
223  MCParticleImpl* q = new lcio::MCParticleImpl();
224  q->setPDG(p->pdgID);
225 
226  float ps_fa[3] = {float(p->psx/CLHEP::GeV),float(p->psy/CLHEP::GeV),float(p->psz/CLHEP::GeV)};
227  q->setMomentum( ps_fa );
228 
229 #if LCIO_VERSION_GE(2,7)
230  float pe_fa[3] = {float(p->pex/CLHEP::GeV),float(p->pey/CLHEP::GeV),float(p->pez/CLHEP::GeV)};
231  q->setMomentumAtEndpoint( pe_fa );
232 #endif
233  double vs_fa[3] = { p->vsx/CLHEP::mm, p->vsy/CLHEP::mm, p->vsz/CLHEP::mm } ;
234  q->setVertex( vs_fa );
235 
236  double ve_fa[3] = { p->vex/CLHEP::mm, p->vey/CLHEP::mm, p->vez/CLHEP::mm } ;
237  q->setEndpoint( ve_fa );
238 
239  q->setTime(p->time/CLHEP::ns);
240  q->setMass(p->mass/CLHEP::GeV);
241  q->setCharge(def ? def->GetPDGCharge() : 0); // Charge(e+) = 1 !
242 
243  // Set generator status
244  q->setGeneratorStatus(0);
245  if ( mask.isSet(G4PARTICLE_GEN_STABLE) ) q->setGeneratorStatus(1);
246  else if ( mask.isSet(G4PARTICLE_GEN_DECAYED) ) q->setGeneratorStatus(2);
247  else if ( mask.isSet(G4PARTICLE_GEN_DOCUMENTATION) ) q->setGeneratorStatus(3);
248  // std::cout << " ********** mcp genstatus : " << q->getGeneratorStatus() << std::endl ;
249 
250  // Set simulation status
251  q->setCreatedInSimulation( mask.isSet(G4PARTICLE_SIM_CREATED) );
252  q->setBackscatter( mask.isSet(G4PARTICLE_SIM_BACKSCATTER) );
253  q->setVertexIsNotEndpointOfParent( mask.isSet(G4PARTICLE_SIM_PARENT_RADIATED) );
254  q->setDecayedInTracker( mask.isSet(G4PARTICLE_SIM_DECAY_TRACKER) );
255  q->setDecayedInCalorimeter( mask.isSet(G4PARTICLE_SIM_DECAY_CALO) );
256  q->setHasLeftDetector( mask.isSet(G4PARTICLE_SIM_LEFT_DETECTOR) );
257  q->setStopped( mask.isSet(G4PARTICLE_SIM_STOPPED) );
258  q->setOverlay( false );
259 
260  //fg: if simstatus !=0 we have to set the generator status to 0:
261  if( q->isCreatedInSimulation() )
262  q->setGeneratorStatus( 0 ) ;
263 
264  q->setSpin(p->spin);
265  q->setColorFlow(p->colorFlow);
266 
267  lc_coll->addElement(q);
268  p_ids[id] = cnt;
269  p_part[cnt] = p;
270  p_lcio[cnt] = q;
271  }
272 
273  // Now establish parent-daughter relationships
274  for(size_t i=0, n=p_ids.size(); i<n; ++i) {
275  map<int,int>::iterator k;
276  const Geant4Particle* p = p_part[i];
277  MCParticleImpl* q = p_lcio[i];
278  const Geant4Particle::Particles& dau = p->daughters;
279  for(Geant4Particle::Particles::const_iterator j=dau.begin(); j!=dau.end(); ++j) {
280  int idau = *j;
281  if ( (k=p_ids.find(idau)) == p_ids.end() ) { // Error!!!
282  printout(FATAL,"Geant4Conversion","+++ Particle %d: FAILED to find daughter with ID:%d",p->id,idau);
283  continue;
284  }
285  int iqdau = (*k).second;
286  MCParticleImpl* qdau = p_lcio[iqdau];
287  qdau->addParent(q);
288  }
289  const Geant4Particle::Particles& par = p->parents;
290  for(Geant4Particle::Particles::const_iterator j=par.begin(); j!=par.end(); ++j) {
291  int ipar = *j; // A parent ID iof -1 means NO parent, because a base of 0 is perfectly leagal!
292  if ( ipar>=0 && (k=p_ids.find(ipar)) == p_ids.end() ) { // Error!!!
293  printout(FATAL,"Geant4Conversion","+++ Particle %d: FAILED to find parent with ID:%d",p->id,ipar);
294  continue;
295  }
296  int iqpar = (*k).second;
297  MCParticleImpl* qpar = p_lcio[iqpar];
298  q->addParent(qpar);
299  }
300  }
301  }
302  return lc_coll;
303 }

306 void Geant4Output2LCIO::saveEvent(OutputContext<G4Event>& ctxt) {
307  lcio::LCEventImpl* e = context()->event().extension<lcio::LCEventImpl>();
308  e->setRunNumber(m_runNo);
309  e->setEventNumber(ctxt.context->GetEventID());
310  e->setDetectorName(context()->lcdd().header().name());
311  e->setRunNumber(m_runNo);
312  lcio::LCEventImpl* evt = context()->event().extension<lcio::LCEventImpl>();
313  Geant4ParticleMap* part_map = context()->event().extension<Geant4ParticleMap>(false);
314  if ( part_map ) {
315  print("+++ Saving %d LCIO particles....",int(part_map->particleMap.size()));
316  if ( part_map->particleMap.size() > 0 ) {
317  lcio::LCCollectionVec* col = saveParticles(part_map);
318  evt->addCollection(col,lcio::LCIO::MCPARTICLE);
319  }
320  }
