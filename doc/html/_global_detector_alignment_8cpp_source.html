97  GlobalAlignment _alignment(const GlobalDetectorAlignment& det) {
98  DetElement::Object& e = det._data();
99  if ( !e.global_alignment.isValid() ) {
100  string path = DetectorTools::placementPath(det);
101  e.global_alignment = Ref_t(new GlobalAlignmentData(path));
102  }
103  Handle<GlobalAlignmentData> h(e.global_alignment);
104  if ( h.isValid() && h->global.isValid() ) {
105  return h->global;
106  }
107  throw runtime_error("DD4hep: Cannot access global alignment data. [Invalid Handle]");

223 GlobalAlignment GlobalDetectorAlignment::align(const string& elt_path, TGeoHMatrix* matrix, bool chk, double overlap) {
224  if ( elt_path.empty() )
225  return _align(_alignment(*this),matrix,chk,overlap);
226  else if ( elt_path == placementPath() )
227  return _align(_alignment(*this),matrix,chk,overlap);
228  else if ( elt_path[0] == '/' ) {
229  GlobalAlignment a(elt_path);
230  volumeAlignments().push_back(a);
231  return _align(a,matrix,chk,overlap);
232  }
233  GlobalAlignment a(placementPath()+'/'+elt_path);
234  volumeAlignments().push_back(a);
235  return _align(a,matrix,chk,overlap);
236 }
