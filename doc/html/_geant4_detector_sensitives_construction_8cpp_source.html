84  typedef Geometry::GeoHandlerTypes::SensitiveVolumes _SV;
85  typedef Geometry::GeoHandlerTypes::ConstVolumeSet VolSet;
86  Geant4GeometryInfo* p = Geant4Mapping::instance().ptr();

88 
89  for(_SV::const_iterator iv=vols.begin(); iv != vols.end(); ++iv) {
90  Geometry::SensitiveDetector sd = (*iv).first;
91  string typ = sd.type(), nam = sd.name();
92  G4VSensitiveDetector* g4sd =
93  PluginService::Create<G4VSensitiveDetector*>(typ, nam, &ctxt->lcdd);
94  if (!g4sd) {
95  string tmp = typ;
96  tmp[0] = ::toupper(tmp[0]);
97  typ = "Geant4" + tmp;
98  g4sd = PluginService::Create<G4VSensitiveDetector*>(typ, nam, &ctxt->lcdd);
99  if ( !g4sd ) {
100  PluginDebug dbg;
101  g4sd = PluginService::Create<G4VSensitiveDetector*>(typ, nam, &ctxt->lcdd);
102  if ( !g4sd ) {
103  throw runtime_error("ConstructSDandField: FATAL Failed to "
104  "create Geant4 sensitive detector " + nam +
105  " of type " + typ + ".");
106  }
107  }
108  }
109  g4sd->Activate(true);
110  G4SDManager::GetSDMpointer()->AddNewDetector(g4sd);
111  const VolSet& sens_vols = (*iv).second;
112  for(VolSet::const_iterator i=sens_vols.begin(); i!= sens_vols.end(); ++i) {
113  const TGeoVolume* vol = *i;
114  G4LogicalVolume* g4v = p->g4Volumes[vol];
115  if ( !g4v ) {
116  throw runtime_error("ConstructSDandField: Failed to access G4LogicalVolume for SD "+
117  nam + " of type " + typ + ".");
118  }
119  ctxt->setSensitiveDetector(g4v,g4sd);
120  }
121  }
122  print("Geant4Converter", "++ Handled %ld sensitive detectors.",vols.size());
