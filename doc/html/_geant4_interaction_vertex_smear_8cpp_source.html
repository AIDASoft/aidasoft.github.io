46 void Geant4InteractionVertexSmear::smear(Interaction* inter) const {
47  Geant4Random& rndm = context()->event().random();
48  if ( inter ) {
49  double dx = rndm.gauss(m_offset.x(),m_sigma.x());
50  double dy = rndm.gauss(m_offset.y(),m_sigma.y());
51  double dz = rndm.gauss(m_offset.z(),m_sigma.z());
52  double dt = rndm.gauss(m_offset.t(),m_sigma.t());
53  print("+++ Smearing primary vertex for interaction type %d (%d Vertices, %d particles) "
54  "by (%+.2e mm, %+.2e mm, %+.2e mm, %+.2e ns)",
55  m_mask,int(inter->vertices.size()),int(inter->particles.size()),dx,dy,dz,dt);
56  smearInteraction(this,inter,dx,dy,dz,dt);
57  return;
58  }
59  print("+++ No interaction of type %d present.",m_mask);
60 }

63 void Geant4InteractionVertexSmear::operator()(G4Event*) {
64  typedef std::vector<Geant4PrimaryInteraction*> _I;
65  Geant4PrimaryEvent* evt = context()->event().extension<Geant4PrimaryEvent>();
66 
67  if ( m_mask >= 0 ) {
68  Interaction* inter = evt->get(m_mask);
69  smear(inter);
70  return;
71  }
72  _I interactions = evt->interactions();
