32 AlignmentUpdateCall::handle(const ConditionKey& key, const UpdateContext& ctxt, const Delta& delta)
33 {
34  AlignmentCondition target(key.name);
35  AlignmentData& data = target.data();
36  data.delta = delta;
37  data.flag = AlignmentData::HAVE_NONE;
38  data.detector = ctxt.dependency.detector;
39  // We MUST register the undigested condition to the alignment manager in order to
40  // later be able to compute the derived transformations.
41  // The derived transfomrations can only be computed 'later' once all delta stubs
42  // are present in a second pass. This is necessary, because the parent information
43  // may actually be supplied also 'later'.
44  AlignmentsManager::newEntry(ctxt, data.detector, &ctxt.dependency, target);
45  return target;
46 }
