90 void Geant4UIManager::start() {
91  typedef std::vector<std::string> _V;
92  // Get the pointer to the User Interface manager
93  G4UImanager* mgr = G4UImanager::GetUIpointer();
94  bool executed_statements = false;
95 
96  // Start visualization
97  if ( m_haveVis || !m_visSetup.empty() ) {
98  m_vis = startVis();
99  m_haveVis = true; // If graphics setup, vis is always true
100  m_haveUI = true; // No graphics without UI!
101  }
102  // Start UI instance
103  if ( m_haveUI ) {
104  m_ui = startUI();
105  }
106  // Configure visualization instance
107  if ( !m_visSetup.empty() ) {
108  printout(INFO,"Geant4UIManager","++ Executing visualization setup:%s",m_visSetup.c_str());
109  mgr->ApplyCommand(make_cmd(m_visSetup).c_str());
110  }
111  // Configure UI instance
112  if ( !m_uiSetup.empty() ) {
113  printout(INFO,"Geant4UIManager","++ Executing UI setup:%s",m_uiSetup.c_str());
114  mgr->ApplyCommand(make_cmd(m_uiSetup).c_str());
115  executed_statements = true;
116  }
117  // Execute the chained macro files
118  for(_V::const_iterator i=m_macros.begin(); i!=m_macros.end(); ++i) {
119  printout(INFO,"Geant4UIManager","++ Executing Macro file:%s",(*i).c_str());
120  mgr->ApplyCommand(make_cmd(*i).c_str());
121  executed_statements = true;
122  }
123  // Execute the chained command statements
124  for(_V::const_iterator i=m_commands.begin(); i!=m_commands.end(); ++i) {
125  printout(INFO,"Geant4UIManager","++ Executing Command statement:%s",(*i).c_str());
126  mgr->ApplyCommand((*i).c_str());
127  executed_statements = true;
128  }
129  // Start UI session if present
130  if ( m_haveUI && m_ui ) {
131  m_ui->SessionStart();
132  return;
133  }
134  else if ( m_haveUI ) {
135  printout(WARNING,"Geant4UIManager","++ No UI manager found. Exit.");
136  return;
137  }
138  else if ( executed_statements ) {
139  return;
140  }
141 
142  // No UI. Pure batch mode: Simply execute requested number of events
143  long numEvent = context()->kernel().property("NumEvents").value<long>();
144  printout(INFO,"Geant4UIManager","++ Start run with %d events.",numEvent);
145  context()->kernel().runManager().BeamOn(numEvent);
146 }
