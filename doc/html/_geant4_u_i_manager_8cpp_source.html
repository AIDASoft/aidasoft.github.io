00092   // Get the pointer to the User Interface manager

00103   if ( m_haveUI ) {
00104     m_ui = startUI();
00105   }
00106   // Configure visualization instance
00107   if ( !m_visSetup.empty() ) {
00108     printout(INFO,"Geant4UIManager","++ Executing visualization setup:%s",m_visSetup.c_str());
00109     mgr->ApplyCommand(make_cmd(m_visSetup).c_str());
00110   }
00111   // Configure UI instance
00112   if ( !m_uiSetup.empty() )   {
00113     printout(INFO,"Geant4UIManager","++ Executing UI setup:%s",m_uiSetup.c_str());
00114     mgr->ApplyCommand(make_cmd(m_uiSetup).c_str());
00115     executed_statements = true;
00116   }
00117   // Execute the chained macro files
00118   for(_V::const_iterator i=m_macros.begin(); i!=m_macros.end(); ++i)   {
00119     printout(INFO,"Geant4UIManager","++ Executing Macro file:%s",(*i).c_str());
00120     mgr->ApplyCommand(make_cmd(*i).c_str());
00121     executed_statements = true;
00122   }
00123   // Execute the chained command statements
00124   for(_V::const_iterator i=m_commands.begin(); i!=m_commands.end(); ++i)   {
00125     printout(INFO,"Geant4UIManager","++ Executing Command statement:%s",(*i).c_str());
00126     mgr->ApplyCommand((*i).c_str());
00127     executed_statements = true;
00128   }
00129   // Start UI session if present
00130   if ( m_haveUI && m_ui )   {
00131     m_ui->SessionStart();
00132     return;
00133   }
00134   else if ( m_haveUI )   {
00135     printout(WARNING,"Geant4UIManager","++ No UI manager found. Exit.");
00136     return;
00137   }
00138   else if ( executed_statements )  {
00139     return;
00140   }
00141 
00142   // No UI. Pure batch mode: Simply execute requested number of events
00143   long numEvent = context()->kernel().property("NumEvents").value<long>();

00145   context()->kernel().runManager().BeamOn(numEvent);
