25 static Ref_t create_detector(LCDD& lcdd, xml_h e, SensitiveDetector sens) {
26  typedef vector<PlacedVolume> Placements;
27  xml_det_t x_det = e;
28  Material vacuum = lcdd.vacuum();
29  int det_id = x_det.id();
30  string det_name = x_det.nameStr();
31  bool reflect = x_det.reflect(false);
32  DetElement sdet (det_name,det_id);
33  Assembly assembly (det_name);
34  //Volume assembly (det_name,Box(10000,10000,10000),vacuum);
35  Volume motherVol = lcdd.pickMotherVolume(sdet);
36  int m_id=0, c_id=0, n_sensor=0;
37  map<string,Volume> modules;
38  map<string, Placements> sensitives;
39  PlacedVolume pv;
40 
41  assembly.setVisAttributes(lcdd.invisible());
42  sens.setType("tracker");
43 
44  for(xml_coll_t mi(x_det,_U(module)); mi; ++mi, ++m_id) {
45  xml_comp_t x_mod = mi;
46  string m_nam = x_mod.nameStr();
47  xml_comp_t trd = x_mod.trd();
48  double posY;
49  double x1 = trd.x1();
50  double x2 = trd.x2();
51  double z = trd.z();
52  double y1, y2, total_thickness=0.;
53  xml_coll_t ci(x_mod,_U(module_component));
54  for(ci.reset(), total_thickness=0.0; ci; ++ci)
55  total_thickness += xml_comp_t(ci).thickness();
56 
57  y1 = y2 = total_thickness / 2;
58  Volume m_volume(m_nam, Trapezoid(x1, x2, y1, y2, z), vacuum);
59  m_volume.setVisAttributes(lcdd.visAttributes(x_mod.visStr()));
60 
61  for(ci.reset(), n_sensor=1, c_id=0, posY=-y1; ci; ++ci, ++c_id) {
62  xml_comp_t c = ci;
63  double c_thick = c.thickness();
64  Material c_mat = lcdd.material(c.materialStr());
65  string c_name = _toString(c_id,"component%d");
66  Volume c_vol(c_name, Trapezoid(x1,x2,c_thick/2e0,c_thick/2e0,z), c_mat);
67 
68  c_vol.setVisAttributes(lcdd.visAttributes(c.visStr()));
69  pv = m_volume.placeVolume(c_vol,Position(0,posY+c_thick/2,0));
70  if ( c.isSensitive() ) {
71  sdet.check(n_sensor > 2,"SiTrackerEndcap2::fromCompact: "+c_name+" Max of 2 modules allowed!");
72  pv.addPhysVolID("sensor",n_sensor);
73  c_vol.setSensitiveDetector(sens);
74  sensitives[m_nam].push_back(pv);
75  ++n_sensor;
76  }
77  posY += c_thick;
78  }
79  modules[m_nam] = m_volume;
80  }
81 
82  for(xml_coll_t li(x_det,_U(layer)); li; ++li) {
83  xml_comp_t x_layer(li);
84  int l_id = x_layer.id();
85  int mod_num = 1;
86  for(xml_coll_t ri(x_layer,_U(ring)); ri; ++ri) {
87  xml_comp_t x_ring = ri;
88  double r = x_ring.r();
89  double phi0 = x_ring.phi0(0);
90  double zstart = x_ring.zstart();
91  double dz = x_ring.dz(0);
92  int nmodules = x_ring.nmodules();
93  string m_nam = x_ring.moduleStr();
94  Volume m_vol = modules[m_nam];
95  double iphi = 2*M_PI/nmodules;
96  double phi = phi0;
97  Placements& sensVols = sensitives[m_nam];
98 
99  for(int k=0; k<nmodules; ++k) {
100  string m_base = _toString(l_id,"layer%d") + _toString(mod_num,"_module%d");
101  double x = -r*std::cos(phi);
102  double y = -r*std::sin(phi);
103  DetElement module(sdet,m_base+"_pos",det_id);
104  pv = assembly.placeVolume(m_vol,Transform3D(RotationZYX(0,-M_PI/2-phi,-M_PI/2),Position(x,y,zstart+dz)));
105  pv.addPhysVolID("barrel",1).addPhysVolID("layer", l_id).addPhysVolID("module",mod_num);
106  module.setPlacement(pv);
107  for(size_t ic=0; ic<sensVols.size(); ++ic) {
108  PlacedVolume sens_pv = sensVols[ic];
109  DetElement comp_elt(module,sens_pv.volume().name(),mod_num);
110  comp_elt.setPlacement(sens_pv);
111  }
112 
113  if ( reflect ) {
114  pv = assembly.placeVolume(m_vol,Transform3D(RotationZYX(M_PI,-M_PI/2-phi,-M_PI/2),Position(x,y,-zstart-dz)));
115  pv.addPhysVolID("barrel",2).addPhysVolID("layer",l_id).addPhysVolID("module",mod_num);
116  DetElement r_module(sdet,m_base+"_neg",det_id);
117  r_module.setPlacement(pv);
118  for(size_t ic=0; ic<sensVols.size(); ++ic) {
119  PlacedVolume sens_pv = sensVols[ic];
120  DetElement comp_elt(r_module,sens_pv.volume().name(),mod_num);
121  comp_elt.setPlacement(sens_pv);
122  }
123  }
124  dz = -dz;
125  phi += iphi;
126  ++mod_num;
127  }
128  }
129  }
130  pv = motherVol.placeVolume(assembly);
131  pv.addPhysVolID("system",det_id);
132  sdet.setPlacement(pv);
133  return sdet;
134 }
