90 void Geant4TCUserParticleHandler::end(const G4Track* /* track */, Particle& p) {
91 
92  double r_prod = std::sqrt(p.vsx*p.vsx + p.vsy*p.vsy);
93  double z_prod = std::fabs(p.vsz);
94  bool starts_in_trk_vol = ( r_prod <= m_rTracker && z_prod <= m_zTracker ) ;
95 
96  // created in tracking volume but below energy cut
97  if( starts_in_trk_vol && ! (p.reason&G4PARTICLE_ABOVE_ENERGY_THRESHOLD) ) {
98  p.reason = 0;
99  return;
100  }
101 
102  double r_end = std::sqrt(p.vex*p.vex + p.vey*p.vey);
103  double z_end = std::fabs(p.vez);
104  bool ends_in_trk_vol = ( r_end <= m_rTracker && z_end <= m_zTracker ) ;
105 
106  // created and ended in calo
107  if( !starts_in_trk_vol ) {
108 
109  if( !ends_in_trk_vol ){
110  p.reason = 0;
111  }
112  //fg: dont keep backscatter that did not create a tracker hit
113  else if( !( p.reason & G4PARTICLE_CREATED_TRACKER_HIT ) )
114  p.reason = 0;
115  }
116 
117  // Set the simulator status bits
118  DD4hep::ReferenceBitMask<int> simStatus(p.status);
119 
120  if( ends_in_trk_vol ) {
121  simStatus.set(G4PARTICLE_SIM_DECAY_TRACKER);
122  }
123 
124  // if the particle doesn't end in the tracker volume it must have ended in the calorimeter
125  if( not ends_in_trk_vol && not simStatus.isSet(G4PARTICLE_SIM_LEFT_DETECTOR) ) {
126  simStatus.set(G4PARTICLE_SIM_DECAY_CALO);
127  }
128 
129  if( not starts_in_trk_vol && ends_in_trk_vol ) {
130  simStatus.set(G4PARTICLE_SIM_BACKSCATTER);
131  }
132 
133  return ;
134 
135 }
