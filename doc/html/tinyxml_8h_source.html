00344   // Get a character, while interpreting entities.
00345   // The length can be from 0 to 4 bytes.
00346   inline static const char* GetChar(const char* p, char* _value, int* length, TiXmlEncoding encoding) {
00347     assert( p);
00348     if (encoding == TIXML_ENCODING_UTF8) {
00349       *length = utf8ByteTable[*((const unsigned char*) p)];
00350       assert( *length >= 0 && *length < 5);
00351     }
00352     else {
00353       *length = 1;
00354     }
00355 
00356     if (*length == 1) {
00357       if (*p == '&')
00358         return GetEntity(p, _value, length, encoding);
00359       *_value = *p;
00360       return p + 1;
00361     }
00362     else if (*length) {
00363       //strncpy( _value, p, *length );  // lots of compilers don't like this function (unsafe),
00364       // and the null terminator isn't needed
00365       for (int i = 0; p[i] && i < *length; ++i) {
00366         _value[i] = p[i];
00367       }
00368       return p + (*length);
00369     }
00370     else {
00371       // Not valid text.
00372       return 0;
00373     }
00374   }
