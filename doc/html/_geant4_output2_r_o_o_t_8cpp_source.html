82 int Geant4Output2ROOT::fill(const string& nam, const ComponentCast& type, void* ptr) {
83  if (m_file) {
84  TBranch* b = 0;
85  Branches::const_iterator i = m_branches.find(nam);
86  if (i == m_branches.end()) {
87  TClass* cl = TBuffer::GetClass(type.type);
88  if (cl) {
89  b = m_tree->Branch(nam.c_str(), cl->GetName(), (void*) 0);
90  b->SetAutoDelete(false);
91  m_branches.insert(make_pair(nam, b));
92  }
93  else {
94  throw runtime_error("No ROOT TClass object availible for object type:" + typeName(type.type));
95  }
96  }
97  else {
98  b = (*i).second;
99  }
100  Long64_t evt = b->GetEntries(), nevt = b->GetTree()->GetEntries(), num = nevt - evt;
101  if (nevt > evt) {
102  b->SetAddress(0);
103  while (num > 0) {
104  b->Fill();
105  --num;
106  }
107  }
108  b->SetAddress(&ptr);
109  int nbytes = b->Fill();
110  if (nbytes < 0) {
111  throw runtime_error("Failed to write ROOT collection:" + nam + "!");
112  }
113  return nbytes;
114  }
115  return 0;
116 }

119 void Geant4Output2ROOT::commit(OutputContext<G4Event>& ctxt) {
120  if (m_file) {
121  TObjArray* a = m_tree->GetListOfBranches();
122  Long64_t evt = m_tree->GetEntries() + 1;
123  Int_t nb = a->GetEntriesFast();
125  for (Int_t i = 0; i < nb; ++i) {
126  TBranch* br_ptr = (TBranch*) a->UncheckedAt(i);
127  Long64_t br_evt = br_ptr->GetEntries();
128  if (br_evt < evt) {
129  Long64_t num = evt - br_evt;
130  br_ptr->SetAddress(0);
131  while (num > 0) {
132  br_ptr->Fill();
133  --num;
134  }
135  }
136  }
137  m_tree->SetEntries(evt);
138  }
139  Geant4OutputAction::commit(ctxt);
140 }

159 void Geant4Output2ROOT::saveCollection(OutputContext<G4Event>& /* ctxt */, G4VHitsCollection* collection) {
160  Geant4HitCollection* coll = dynamic_cast<Geant4HitCollection*>(collection);
161  string hc_nam = collection->GetName();
162  vector<void*> hits;
163  if (coll) {
164  coll->getHitsUnchecked(hits);
165  size_t nhits = coll->GetSize();
166  if ( m_handleMCTruth && m_truth && nhits > 0 ) {
167  try {
168  for(size_t i=0; i<nhits; ++i) {
169  Geant4HitData* h = coll->hit(i);
170  Geant4Tracker::Hit* trk_hit = dynamic_cast<Geant4Tracker::Hit*>(h);
171  if ( 0 != trk_hit ) {
172  Geant4HitData::Contribution& t = trk_hit->truth;
173  int trackID = t.trackID;
174  t.trackID = m_truth->particleID(trackID);
175  }
176  Geant4Calorimeter::Hit* cal_hit = dynamic_cast<Geant4Calorimeter::Hit*>(h);
177  if ( 0 != cal_hit ) {
178  Geant4HitData::Contributions& c = cal_hit->truth;
179  for(Geant4HitData::Contributions::iterator j=c.begin(); j!=c.end(); ++j) {
180  Geant4HitData::Contribution& t = *j;
181  int trackID = t.trackID;
182  t.trackID = m_truth->particleID(trackID);
183  }
184  }
185  }
186  }
187  catch(...) {
188  printout(ERROR,name(),"+++ Exception while saving collection %s.",hc_nam.c_str());
189  }
190  }
191  fill(hc_nam, coll->vector_type(), &hits);
192  }
