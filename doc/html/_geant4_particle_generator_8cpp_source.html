78 void Geant4ParticleGenerator::printInteraction(Geant4PrimaryInteraction* inter) const {
79  typedef Geant4PrimaryInteraction::VertexMap _V;
80  int count = 0;
81  if ( !inter ) {
82  warning("printInteraction: Invalid interaction pointer [NULL-Pointer].");
83  return;
84  }
85  for(_V::const_iterator iv=inter->vertices.begin(); iv!=inter->vertices.end(); ++iv) {
86  Geant4Vertex* v = (*iv).second;
87  print("+-> Interaction [%d] %.3f GeV %s pos:(%.3f %.3f %.3f)[mm]",
88  count, m_energy/CLHEP::GeV, m_particleName.c_str(),
89  v->x/CLHEP::mm, v->y/CLHEP::mm, v->z/CLHEP::mm);
90  ++count;
91  for(set<int>::const_iterator i=v->out.begin(); i!=v->out.end(); ++i) {
92  Geant4ParticleHandle p = inter->particles[*i];
93  p.dumpWithVertex(outputLevel(),name()," +->");
94  }
95  }
96 }

99 void Geant4ParticleGenerator::operator()(G4Event*) {
100  typedef Geant4Particle Particle;
101 
102  if (0 == m_particle || m_particle->GetParticleName() != m_particleName.c_str()) {
103  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
104  m_particle = particleTable->FindParticle(m_particleName);
105  if (0 == m_particle) {
106  throw runtime_error("Geant4ParticleGenerator: Bad particle type:"+m_particleName+"!");
107  }
108  }
109  Geant4Event& evt = context()->event();
110  Geant4PrimaryEvent* prim = evt.extension<Geant4PrimaryEvent>();
111  Geant4PrimaryInteraction* inter = new Geant4PrimaryInteraction();
112  prim->add(m_mask, inter);
113 
114  Geant4Vertex* vtx = new Geant4Vertex();
115  int multiplicity = m_multiplicity;
116  ROOT::Math::XYZVector unit_direction, direction, position = m_position;
117  getVertexPosition(position);
118  getParticleMultiplicity(multiplicity);
119  vtx->mask = m_mask;
120  vtx->x = position.X();
121  vtx->y = position.Y();
122  vtx->z = position.Z();
123  inter->vertices.insert(make_pair(inter->vertices.size(),vtx));
124  for(int i=0; i<m_multiplicity; ++i) {
125  double momentum = m_energy;
126  Particle* p = new Particle();
127  direction = m_direction;
128  getParticleDirection(i, direction, momentum);
129  unit_direction = direction.unit();
130  p->id = inter->nextPID();
131  p->status |= G4PARTICLE_GEN_STABLE;
132  p->mask = m_mask;
133  p->pdgID = m_particle->GetPDGEncoding();
134 
135  p->psx = unit_direction.X()*momentum;
136  p->psy = unit_direction.Y()*momentum;
137  p->psz = unit_direction.Z()*momentum;
138  p->mass = m_particle->GetPDGMass();
139  p->charge = m_particle->GetPDGCharge();
140  p->spin[0] = 0;
141  p->spin[1] = 0;
142  p->spin[2] = 0;
143  p->colorFlow[0] = 0;
144  p->colorFlow[1] = 0;
145  p->vsx = vtx->x;
146  p->vsy = vtx->y;
147  p->vsz = vtx->z;
148  //fg: do not set the endpoint to the start point of the particle
149  // p->vex = vtx->x;
150  // p->vey = vtx->y;
151  // p->vez = vtx->z;
152  inter->particles.insert(make_pair(p->id,p));
153  vtx->out.insert(p->id);
154  printout(INFO,name(),"Particle [%d] %s %.3f GeV direction:(%6.3f %6.3f %6.3f)",
155  p->id, m_particleName.c_str(), momentum/CLHEP::GeV,
156  direction.X(), direction.Y(), direction.Z());
157 
158  }
159 }
