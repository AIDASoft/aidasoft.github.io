45  template <typename UserData>
46  void Installer<UserData>::install(DetElement component, PlacedVolume pv) {
47  Volume comp_vol = pv.volume();
48  if ( comp_vol.isSensitive() ) {
49  Volume mod_vol = parentVolume(component);
50  DD4hep::Geometry::Box mod_shape(mod_vol.solid()), comp_shape(comp_vol.solid());
51 
52  if ( !comp_shape.isValid() || !mod_shape.isValid() ) {
53  invalidInstaller("Components and/or modules are not boxes -- invalid shapes");
54  }
55  else if ( !handleUsingCache(component,comp_vol) ) {
56  const double* trans = placementTranslation(component);
57  double half_module_thickness = mod_shape->GetDZ();
58  double sensitive_z_position = trans[2];
59  double outer_thickness = half_module_thickness + sensitive_z_position;
60  double inner_thickness = half_module_thickness - sensitive_z_position;
61  //Surface is placed at the center of the volume, no need to shift origin
62  //Make sure u,v,n form a right-handed coordinate system, v along the final z
63  Vector3D u(-1.,0.,0.), v(0.,-1.,0.), n(0.,0.,1.), o(0.,0.,0.);
64 
65  Type type( Type::Sensitive ) ;
66 
67  if( data.dimension == 1 ) {
68  type.setProperty( Type::Measurement1D , true ) ;
69  } else if( data.dimension != 2 ) {
70  throw std::runtime_error("**** DD4hep_SiTrackerBarrelSurfacePlugin: no or wrong "
71  "'dimension' argument given - has to be 1 or 2") ;
72  }
73  VolPlane surf(comp_vol, type, inner_thickness, outer_thickness, u, v, n, o);
74  addSurface(component,surf);
75  }
76  }
77  }
