106 size_t LCIOEventHandler::collectionLoop(const std::string& collection, DDEveHitActor& actor) {
107  Branches::const_iterator ibr = m_branches.find(collection);
108  if ( ibr != m_branches.end() ) {
109  LCCollection* c = (*ibr).second;
110  if ( c ) {
111  DDEveHit hit;
112  int n = c->getNumberOfElements();
113  actor.setSize(n);
114  for(int i=0; i<n; ++i) {
115  LCObject* ptr = c->getElementAt(i);
116  if ( _convertHitFunc(ptr,&hit) ) {
117  actor(hit);
118  }
119  }
120  return n;
121  }
122  }
123  return 0;
124 }
