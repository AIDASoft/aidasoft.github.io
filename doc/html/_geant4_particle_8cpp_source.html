362  tag,
363  p->id,
364  p.particleName().c_str(),
365  parent_id,equiv,
366  yes_no(mask.isSet(G4PARTICLE_PRIMARY)),
367  yes_no(mask.isSet(G4PARTICLE_HAS_SECONDARIES)),
