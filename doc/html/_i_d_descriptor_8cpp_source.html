28  void _construct(IDDescriptor::Object* o, const string& dsc) {
29  BitField64& bf = *o;
30  o->fieldIDs.clear();
31  o->fieldMap.clear();
32  o->description = dsc;
33  for (size_t i = 0; i < bf.size(); ++i) {
34  IDDescriptor::Field f = &bf[i];
35  o->fieldIDs.push_back(make_pair(i, f->name()));

37  }
38  }
39 }

107 VolumeID IDDescriptor::encode(const std::vector<VolID>& id_vector) const {
108  typedef std::vector<VolID> VolIds;
109  VolumeID id = 0;
110  for (VolIds::const_iterator i = id_vector.begin(); i != id_vector.end(); ++i) {
111  Field f = field((*i).first);
112  VolumeID vid = (*i).second;
113  int offset = f->offset();
114  id |= ((f->value(vid<<offset) << offset)&f->mask());
115  }
116  return id;
117 }
