00074     /*
00075       --------------------------------------------------------------------
00076       mix -- mix 3 32-bit values reversibly.
00077       For every delta with one or two bits set, and the deltas of all three
00078       high bits or all three low bits, whether the original value of a,b,c
00079       is almost all zero or is uniformly distributed,
00080       * If mix() is run forward or backward, at least 32 bits in a,b,c
00081       have at least 1/4 probability of changing.
00082       * If mix() is run forward, every bit of c will change between 1/3 and
00083       2/3 of the time.  (Well, 22/100 and 78/100 for some 2-bit deltas.)
00084       mix() was built out of 36 single-cycle latency instructions in a
00085       structure that could supported 2x parallelism, like so:
00086       a -= b;
00087       a -= c; x = (c>>13);
00088       b -= c; a ^= x;
00089       b -= a; x = (a<<8);
00090       c -= a; b ^= x;
00091       c -= b; x = (b>>13);
00092       ...
00093       Unfortunately, superscalar Pentiums and Sparcs can't take advantage
00094       of that parallelism.  They've also turned some of those single-cycle
00095       latency instructions into multi-cycle latency instructions.  Still,
00096       this is the fastest good hash I could find.  There were about 2^^68
00097       to choose from.  I only looked at a billion or so.
00098       --------------------------------------------------------------------
00099     */
00100 #define mix(a,b,c)                              \
00101     {                                           \
00102       a -= b; a -= c; a ^= (c>>13);             \
00103       b -= c; b -= a; b ^= (a<<8);              \
00104       c -= a; c -= b; c ^= (b>>13);             \
00105       a -= b; a -= c; a ^= (c>>12);             \
00106       b -= c; b -= a; b ^= (a<<16);             \
00107       c -= a; c -= b; c ^= (b>>5);              \
00108       a -= b; a -= c; a ^= (c>>3);              \
00109       b -= c; b -= a; b ^= (a<<10);             \

00117       len     : the length of the key, counting by bytes

00124       mod a prime (mod is sooo slow!).  If you need less than 32 bits,

00126       h = (h & hashmask(10));

00129       If you are hashing n strings (ub1 **)k, do it like this:
00130       for (i=0, h=0; i<n; ++i) h = hash( k[i], len[i], h);

00132       By Bob Jenkins, 1996.  bob_jenkins@burtleburtle.net.  You may use this
