33 SurfaceInstaller::SurfaceInstaller(LCDD& lcdd, int argc, char** argv)
34  : m_lcdd(lcdd), m_det(), m_stopScanning(false)
35 {
36  if ( argc > 0 ) {
37  string det_name = argv[0];
38  string n = det_name[0] == '-' ? det_name.substr(1) : det_name;
39  m_det = lcdd.detector(n);
40  if ( !m_det.isValid() ) {
41  stringstream err;
42  err << "The subdetector " << det_name << " is not known to the geometry.";
43  printout(INFO,"SurfaceInstaller",err.str().c_str());
44  throw runtime_error(err.str());
45  }
46  printout(INFO,m_det.name(),"+++ Processing SurfaceInstaller for subdetector: '%s'",m_det.name());
47  return;
48  }
49  throw runtime_error("The plugin takes at least one argument. No argument supplied");
50 }

80 void SurfaceInstaller::install(DetElement component, PlacedVolume pv) {
81  if ( pv.volume().isSensitive() ) {
82  stringstream log;
83  PlacementPath all_nodes;
84  ElementPath det_elts;
85  DetectorTools::elementPath(component,det_elts);
86  DetectorTools::placementPath(component,all_nodes);
87  string elt_path = DetectorTools::elementPath(det_elts);
88  string node_path = DetectorTools::placementPath(all_nodes);
89 
90  log << "Lookup " << " Detector[" << det_elts.size() << "]: " << elt_path;
91  printout(INFO,m_det.name(),log.str());
92  log.str("");
93  log << " " << " Places[" << all_nodes.size() << "]: " << node_path;
94  printout(INFO,m_det.name(),log.str());
95  log.str("");
96  log << " " << " Matrices[" << all_nodes.size() << "]: ";
97  for(PlacementPath::const_reverse_iterator i=all_nodes.rbegin(); i!=all_nodes.rend(); ++i) {
98  PlacedVolume placed = *i;
99  log << (void*)(placed->GetMatrix()) << " ";
100  if ( placed->GetUserExtension() ) {
101  const PlacedVolume::VolIDs& vid = placed.volIDs();
102  for(PlacedVolume::VolIDs::const_iterator j=vid.begin(); j!=vid.end(); ++j) {
103  log << (*j).first << ":" << (*j).second << " ";
104  }
105  }
106  log << " ";
107  if ( i+1 == all_nodes.rend() ) log << "( -> " << placed->GetName() << ")";
108  }
109  // Get the module element:
110  printout(INFO,m_det.name(),log.str());
111  log.str("");
112  Volume vol = pv.volume();
113  log << " "
114  << " Sensitive: " << (vol.isSensitive() ? "YES" : "NO ")
115  << " Volume: " << (void*)vol.ptr() << " "
116  << " Shape: " << vol.solid().toString();
117  printout(INFO,m_det.name(),log.str());
118  return;
119  }
120  cout << component.name() << ": " << pv.name() << endl;
121 }
