75 double SubdetectorExtensionImpl::getRMin() const {
76  if (_setRMin) {
77  return _rMin;
78  }
79  if (det.isValid() and det.volume().isValid() and det.volume().solid().isValid()) {
80  Geometry::Solid solid = det.volume().solid();
81  try {
82  Geometry::Tube tube(solid);
83  if (tube.isValid()) {
84  return tube->GetRmin();
85  }
86  } catch (std::runtime_error& e) {
87  }
88  try {
89  Geometry::PolyhedraRegular polyhedra(solid);
90  if (polyhedra.isValid()) {
91  return polyhedra->GetRmin()[0];
92  }
93  } catch (std::runtime_error& e) {
94  }
95  }
96  return 0.;
97 }
