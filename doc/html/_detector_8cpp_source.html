113  static void make_path(DetElement::Object* o) {
114  DetElement par = o->parent;
115  if ( par.isValid() ) {
116  o->path = par.path() + "/" + o->name;
117  if ( o->level < 0 ) o->level = par.level() + 1;
118  }
119  else {
120  o->path = "/" + o->name;
121  o->level = 0;
122  }
123  o->key = DD4hep::hash32(o->path);
124  }
125 }

248  if (isValid()) {
249  pair<Children::iterator, bool> r = object<Object>().children.insert(make_pair(sdet.name(), sdet));
250  if (r.second) {
251  sdet.access()->parent = *this;
