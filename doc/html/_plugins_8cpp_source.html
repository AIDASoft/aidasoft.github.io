94  template <typename T>
95  static inline T get_func(void* handle, const char* plugin, const char* entry) {
96 #if !defined(DD4HEP_PARSERS_NO_ROOT)
97  PluginService::FuncPointer<Func_t> fun(gSystem->DynFindSymbol(plugin,entry));
98  PluginService::FuncPointer<T> fp(fun.fptr.ptr);
99  if ( handle ) {}
100 #else
101  PluginService::FuncPointer<T> fp(::dlsym(handle, entry));
102  if ( !fp.fptr.ptr ) fp.fptr.ptr = ::dlsym(0, entry);
103 #endif
104  if ( 0 == fp.fptr.ptr ) {
105  string err = "DD4hep:PluginService: Failed to access symbol "
106  "\""+string(entry)+"\" in plugin library "+string(plugin)+
107  " ["+string(::strerror(errno))+"]";
108  throw runtime_error(err);
109  }
110  return fp.fptr.fcn;
111  }
