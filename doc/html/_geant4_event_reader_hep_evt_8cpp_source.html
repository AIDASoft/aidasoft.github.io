146 Geant4EventReader::EventReaderStatus
147 Geant4EventReaderHepEvt::readParticles(int /* event_number */,
148  Vertex& /* primary_vertex */,
149  vector<Particle*>& particles) {
150 
151  // First check the input file status
152  if ( !m_input.good() || m_input.eof() ) {
153  return EVENT_READER_IO_ERROR;
154  }
155 
156  //static const double c_light = 299.792;// mm/ns
157  //
158  // Read the event, check for errors
159  //
160  int NHEP; // number of entries
161  m_input >> NHEP;
162  if( m_input.eof() ) {
163  //
164  // End of File :: ??? Exception ???
165  // -> FG: EOF is not an exception as it happens for every file at the end !
166  return EVENT_READER_IO_ERROR;
167  }
168  //
169  // Loop over particles
170  int ISTHEP; // status code
171  int IDHEP; // PDG code
172  int JMOHEP1; // first mother
173  int JMOHEP2; // last mother
174  int JDAHEP1; // first daughter
175  int JDAHEP2; // last daughter
176  double PHEP1; // px in GeV/c
177  double PHEP2; // py in GeV/c
178  double PHEP3; // pz in GeV/c
179  double PHEP4; // energy in GeV
180  double PHEP5; // mass in GeV/c**2
181  double VHEP1; // x vertex position in mm
182  double VHEP2; // y vertex position in mm
183  double VHEP3; // z vertex position in mm
184  double VHEP4; // production time in mm/c
185 
186  vector<int> daughter1;
187  vector<int> daughter2;
188 
189  for( int IHEP=0; IHEP<NHEP; IHEP++ ) {
190  if ( m_format == HEPEvtShort )
191  m_input >> ISTHEP >> IDHEP >> JDAHEP1 >> JDAHEP2
192  >> PHEP1 >> PHEP2 >> PHEP3 >> PHEP5;
193  else
194  m_input >> ISTHEP >> IDHEP
195  >> JMOHEP1 >> JMOHEP2
196  >> JDAHEP1 >> JDAHEP2
197  >> PHEP1 >> PHEP2 >> PHEP3
198  >> PHEP4 >> PHEP5
199  >> VHEP1 >> VHEP2 >> VHEP3
200  >> VHEP4;
201 
202  if(m_input.eof())
203  return EVENT_READER_IO_ERROR;
204  //
205  // Create a MCParticle and fill it from stdhep info
206  Particle* p = new Particle(IHEP);
207  PropertyMask status(p->status);
208  //
209  // PDGID
210  p->pdgID = IDHEP;
211  //
212  // Momentum vector
213  p->pex = p->psx = PHEP1*GeV;
214  p->pey = p->psy = PHEP2*GeV;
215  p->pez = p->psz = PHEP3*GeV;
216  //
217  // Mass
218  p->mass = PHEP5*GeV;
219  //
220  // Vertex
221  // (missing information in HEPEvt files)
222  p->vsx = 0.0;
223  p->vsy = 0.0;
224  p->vsz = 0.0;
225  p->vex = 0.0;
226  p->vey = 0.0;
227  p->vez = 0.0;
228  //
229  // Generator status
230  // Simulator status 0 until simulator acts on it
231  p->status = 0;
232  if ( ISTHEP == 0 ) status.set(G4PARTICLE_GEN_EMPTY);
233  else if ( ISTHEP == 1 ) status.set(G4PARTICLE_GEN_STABLE);
234  else if ( ISTHEP == 2 ) status.set(G4PARTICLE_GEN_DECAYED);
235  else if ( ISTHEP == 3 ) status.set(G4PARTICLE_GEN_DOCUMENTATION);
236  else status.set(G4PARTICLE_GEN_DOCUMENTATION);
237  //
238  // Creation time (note the units [1/c_light])
239  // (No information in HEPEvt files)
240  p->time = 0.0;
241  p->properTime = 0.0;
242  //
243  // Keep daughters information for later
244  daughter1.push_back(JDAHEP1);
245  daughter2.push_back(JDAHEP2);
246  //
247  // Add the particle to the collection vector
248  particles.push_back(p);
249  //
250  }// End loop over particles
251 
252  //
253  // Now make a second loop over the particles, checking the daughter
254  // information. This is not always consistent with parent
255  // information, and this utility assumes all parents listed are
256  // parents and all daughters listed are daughters
257  //
258  for( int IHEP=0; IHEP<NHEP; IHEP++ ) {
259  struct ParticleHandler {
260  Particle* m_part;
261  ParticleHandler(Particle* p) : m_part(p) {}
262  void addParent(const Particle* p) {
263  m_part->parents.insert(p->id);
264  }
265  Particle* findParent(const Particle* p) {
266  return m_part->parents.find(p->id)==m_part->parents.end() ? 0 : m_part;
267  }
268  };
269 
270  //
271  // Get the MCParticle
272  //
273  Particle* mcp = particles[IHEP];
274  //
275  // Get the daughter information, discarding extra information
276  // sometimes stored in daughter variables.
277  //
278  int fd = daughter1[IHEP] - 1;
279  int ld = daughter2[IHEP] - 1;
280 
281  //
282  // As with the parents, look for range, 2 discreet or 1 discreet
283  // daughter.
284  if( (fd > -1) && (ld > -1) ) {
285  if(ld >= fd) {
286  for(int id=fd;id<ld+1;id++) {
287  //
288  // Get the daughter, and see if it already lists this particle as
289  // a parent. If not, add this particle as a parent
290  //
291  ParticleHandler part(particles[id]);
292  if ( !part.findParent(mcp) ) part.addParent(mcp);
293  }
294  }
295  else { // Same logic, discrete cases
296  ParticleHandler part_fd(particles[fd]);
297  if ( !part_fd.findParent(mcp) ) part_fd.addParent(mcp);
298 
299  ParticleHandler part_ld(particles[ld]);
300  if ( !part_ld.findParent(mcp) ) part_ld.addParent(mcp);
301  }
302  }
303  else if(fd > -1) {
304  ParticleHandler part(particles[fd]);
305  if ( !part.findParent(mcp) ) part.addParent(mcp);
306  }
307  else if(ld > -1) {
308  ParticleHandler part(particles[ld]);
309  if ( !part.findParent(mcp) ) part.addParent(mcp);
310  }
311  } // End second loop over particles
312  ++m_currEvent;
313  return EVENT_READER_OK;
314 }
