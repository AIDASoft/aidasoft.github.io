00090 void LayeringCnv::fromCompact(Layering& layering) const {
00091   vector<Layer*>& layers = layering.layers();
00092   int count = 0;
00093   for_each(layers.begin(), layers.end(), deletePtr<Layer>);
00094   for (Collection_t c(m_element, _U (layer)); c; ++c) {
00095     Layer lay;
00096     Component layer = c;
00097     int repeat = layer.hasAttr(_U(repeat)) ? layer.repeat() : 1;
00098     ++count;
00099     for (Collection_t s(c, _U(slice)); s; ++s) {
00100       Component slice = s;
00101       string mat = slice.materialStr();
00102       LayerSlice lslice(slice.isSensitive(), slice.thickness(), mat);
00103       lay.add(lslice);
00104     }
00105     lay.compute();
00106     // Add layer to stack once for each repetition
00107     for (int k = 0; k < repeat; ++k)
00108       layers.push_back(new Layer(lay));
00109   }
00110   if (0 == count) {
00111     throw runtime_error("LayeringCnv::fromCompact: No layer children to be build!");
00112   }
