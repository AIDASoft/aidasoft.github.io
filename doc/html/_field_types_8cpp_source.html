64 void DipoleField::fieldComponents(const double* pos, double* field) {
65  double z = pos[2], r = std::sqrt(pos[0] * pos[0] + pos[1] * pos[1]);
66  // Check if z coordinate is within dipole z bounds.
67  if (z > zmin && z < zmax && r < rmax) {
68  // Apply all coefficients to this z coordinate.
69  double pp = 1.0;
70  double abs_z = fabs(z);
71  double bx = coefficents[0];
72  for (size_t i = 1; i < coefficents.size(); ++i) {
73  pp *= abs_z;
74  bx += coefficents[i] * pp;
75  }
76  // Flip sign for negative z.
77  if (z < 0)
78  bx = -bx;
79  // Add Bx to the input field.
80  field[0] += bx;
81  }
82 }

90 void MultipoleField::fieldComponents(const double* pos, double* field) {
91  Transform3D::Point p = transform * Transform3D::Point(pos[0],pos[1],pos[2]);
92  //const Transform3D::Point::CoordinateType& c = p.Coordinates();
93  double x=p.X(), y=p.Y(), z=p.Z();
94  double coord[3] = {x,y,z};
95  if ( 0 == volume.ptr() || volume->Contains(coord) ) {
96  double bx = 0.0;
97  double by = 0.0;
98  double xy = x*y;
99  double x2 = x*x;
100  double y2 = y*y;
101  switch(coefficents.size()) {
102  case 4: // Ocupole momentum
103  by += coefficents[3] * (x2*x - 3.0*x*y2) + skews[3]*(y2*y - 3.0*x*y2);
104  bx += coefficents[3] * (3.0*x2*y - y2*y) + skews[3]*(x2*x - 3.0*x*y2);
105  case 3: // Sextupole momentum:
106  by += -coefficents[2] * (x2 - y2) + skews[2] * 2.0 * xy;
107  bx += coefficents[2] * 2.0 * xy + skews[2] * (x2 - y2);
108  case 2: // Quadrupole momentum:
109  bx += coefficents[1] * x - skews[1]*y;
110  by += coefficents[1] * y + skews[1]*x;
111  case 1: // Dipole momentum:
112  bx += skews[0];
113  by += coefficents[0];
114  case 0: // Nothing, but still valid
115  break;
116  default: // Error condition
117  throw runtime_error("Invalid multipole field definition!");
118  }
119  field[0] += bx;
120  field[1] += by;
121  field[2] += B_z;
122  }
123 }
