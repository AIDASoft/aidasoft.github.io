00064 void DipoleField::fieldComponents(const double* pos, double* field) {
00065   double z = pos[2], r = std::sqrt(pos[0] * pos[0] + pos[1] * pos[1]);
00066   // Check if z coordinate is within dipole z bounds.
00067   if (z > zmin && z < zmax && r < rmax) {
00068     // Apply all coefficients to this z coordinate.
00069     double pp    = 1.0;
00070     double abs_z = fabs(z);
00071     double bx    = coefficents[0];
00072     for (size_t i = 1; i < coefficents.size(); ++i) {
00073       pp *= abs_z;
00074       bx += coefficents[i] * pp;
00075     }
00076     // Flip sign for negative z.
00077     if (z < 0)
00078       bx = -bx;
00079     // Add Bx to the input field.
00080     field[0] += bx;
00081   }
00082 }
