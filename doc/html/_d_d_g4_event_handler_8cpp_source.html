160 Int_t DDG4EventHandler::ReadEvent(Long64_t event_number) {
161  m_data.clear();
162  m_hasEvent = false;
163  if ( hasFile() ) {
164  if ( event_number >= m_file.second->GetEntries() ) {
165  event_number = m_file.second->GetEntries()-1;
166  printout(ERROR,"DDG4EventHandler","+++ ReadEvent: Cannot read across End-of-file! Reading last event:%d.",event_number);
167  }
168  else if ( event_number < 0 ) {
169  event_number = 0;
170  printout(ERROR,"DDG4EventHandler","+++ nextEvent: Cannot read across Start-of-file! Reading first event:%d.",event_number);
171  }
172 
173  Int_t nbytes = m_file.second->GetEntry(event_number);
174  if ( nbytes >= 0 ) {
175  printout(ERROR,"DDG4EventHandler","+++ ReadEvent: Read %d bytes of event data for entry:%d",nbytes,event_number);
176  for(Branches::const_iterator i=m_branches.begin(); i != m_branches.end(); ++i) {
177  TBranch* b = (*i).second.first;
178  std::vector<void*>* ptr_data = *(std::vector<void*>**)b->GetAddress();
179  m_data[b->GetClassName()].push_back(make_pair(b->GetName(),ptr_data->size()));
180  }
181  m_hasEvent = true;
182  return nbytes;
183  }
184  printout(ERROR,"DDG4EventHandler","+++ ReadEvent: Cannot read event data for entry:%d",event_number);
185  throw runtime_error("+++ EventHandler::readEvent: Failed to read event");
186  }
187  throw runtime_error("+++ EventHandler::readEvent: No file open!");
188 }

191 bool DDG4EventHandler::Open(const std::string&, const std::string& name) {
192  if ( m_file.first ) m_file.first->Close();
193  m_hasFile = false;
194  m_hasEvent = false;
195  TFile* f = TFile::Open(name.c_str());
196  if ( f && !f->IsZombie() ) {
197  m_file.first = f;
198  TTree* t = (TTree*)f->Get("EVENT");
199  if ( t ) {
200  TObjArray* br = t->GetListOfBranches();
201  m_file.second = t;
202  m_entry = -1;
203  m_branches.clear();
204  for(Int_t i=0; i<br->GetSize(); ++i) {
205  TBranch* b = (TBranch*)br->At(i);
206  if ( !b ) continue;
207  m_branches[b->GetName()] = make_pair(b,(void*)0);
208  printout(INFO,"DDG4EventHandler::open","+++ Branch %s has %ld entries.",b->GetName(),b->GetEntries());
209  }
210  for(Int_t i=0; i<br->GetSize(); ++i) {
211  TBranch* b = (TBranch*)br->At(i);
212  if ( !b ) continue;
213  b->SetAddress(&m_branches[b->GetName()].second);
214  }
215  m_hasFile = true;
216  return true;
217  }
218  throw runtime_error("+++ Failed to access tree EVENT in ROOT file:"+name);
219  }
220  throw runtime_error("+++ Failed to open ROOT file:"+name);
221 }
