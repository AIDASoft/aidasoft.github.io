91 void Geant4UIMessenger::SetNewValue(G4UIcommand *c, G4String v) {
92  Commands::iterator i = m_propertyCmd.find(c);
93  if (m_properties && i != m_propertyCmd.end()) {
94  const string& n = (*i).second;
95  try {
96  if (!v.empty()) {
97  Property& p = (*m_properties)[n];
98  p.str(v);
99  printout(INFO, "Geant4UIMessenger",
100  "+++ %s> Setting property value %s = %s native:%s.",
101  m_name.c_str(), n.c_str(), v.c_str(), p.str().c_str());
102  }
103  else {
104  string value = (*m_properties)[n].str();
105  printout(INFO, "Geant4UIMessenger", "+++ %s> Unchanged property value %s = %s.",
106  m_name.c_str(), n.c_str(), value.c_str());
107  }
108  }
109  catch(const exception& e) {
110  printout(INFO, "Geant4UIMessenger", "+++ %s> Exception: Failed to change property %s = '%s'.",
111  m_name.c_str(), n.c_str(), v.c_str());
112  printout(INFO, "Geant4UIMessenger", "+++ %s> Exception: %s", m_name.c_str(), e.what());
113  }
114  catch(...) {
115  printout(INFO, "Geant4UIMessenger", "+++ %s> UNKNOWN Exception: Failed to change property %s = '%s'.",
116  m_name.c_str(), n.c_str(), v.c_str());
117  }
118  return;
119  }
120  else {
121  Actions::iterator j = m_actionCmd.find(c);
122  if (j != m_actionCmd.end()) {
123  try {
124  (*j).second.execute(0);
125  }
126  catch(const exception& e) {
127  printout(INFO, "Geant4UIMessenger", "+++ %s> Exception: Failed to exec action '%s' [%s].",
128  m_name.c_str(), c->GetCommandName().c_str(), c->GetCommandPath().c_str());
129  printout(INFO, "Geant4UIMessenger", "+++ %s> Exception: %s",e.what());
130  }
131  catch(...) {
132  printout(INFO, "Geant4UIMessenger", "+++ %s> UNKNOWN Exception: Failed to exec action '%s' [%s].",
133  m_name.c_str(), c->GetCommandName().c_str(), c->GetCommandPath().c_str());
134  }
135  return;
136  }
137  }
138  printout(INFO, "Geant4UIMessenger", "+++ %s> Unknown command callback!", m_name.c_str());
139 }
