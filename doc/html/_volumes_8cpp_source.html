117  TGeoMatrix *matrix = mat;
118  if (matrix == 0)
119  matrix = gGeoIdentity;
120  else
121  matrix->RegisterYourself();
122  if (!vol) {

124  return;
125  }

132  this->T::fNodes = new TObjArray();
133 
134  if (this->T::fFinder) {
135  // volume already divided.
136  this->T::Error("AddNode", "Cannot add node %s_%i into divided volume %s", vol->GetName(), copy_no,
137  this->T::GetName());

141  TGeoNodeMatrix *node = new DD_TGeoNodeMatrix(vol, matrix);

146  if (this->T::fNodes->FindObject(name))
147  this->T::Warning("AddNode", "Volume %s : added node %s with same name", this->T::GetName(), name.Data());

186  TGeoVolume *vol = _copyVol(fShape);
187  Int_t i;
188  // copy volume attributes
189  vol->SetLineColor(GetLineColor());
190  vol->SetLineStyle(GetLineStyle());
191  vol->SetLineWidth(GetLineWidth());
192  vol->SetFillColor(GetFillColor());
193  vol->SetFillStyle(GetFillStyle());
194  // copy other attributes
195  Int_t nbits = 8 * sizeof(UInt_t);
196  for (i = 0; i < nbits; i++)
197  vol->SetAttBit(1 << i, TGeoAtt::TestAttBit(1 << i));
198  for (i = 14; i < 24; i++)
199  vol->SetBit(1 << i, this->TGeoVolume::TestBit(1 << i));
200 
201  // copy field
202  vol->SetField(fField);
203  // Set bits
204  for (i = 0; i < nbits; i++)
205  vol->SetBit(1 << i, this->TGeoVolume::TestBit(1 << i));
206  vol->SetBit(kVolumeClone);
207  // copy nodes
208  // CloneNodesAndConnect(vol);
209  vol->MakeCopyNodes(this);
210  // if volume is divided, copy finder
211  vol->SetFinder(fFinder);
212  // copy voxels
213  if (fVoxels) {
214  TGeoVoxelFinder *voxels = new TGeoVoxelFinder(vol);
215  vol->SetVoxelFinder(voxels);
216  }
217  // copy option, uid
218  vol->SetOption(fOption);
219  vol->SetNumber(fNumber);
220  vol->SetNtotal(fNtotal);

491 static PlacedVolume _addNode(TGeoVolume* par, TGeoVolume* daughter, TGeoMatrix* transform) {
492  if ( !daughter ) {
493  throw runtime_error("DD4hep: Volume: Attempt to assign an invalid physical daughter volume.");
494  }
495  TGeoVolume* parent = par;
496  TObjArray* a = parent->GetNodes();
497  Int_t id = a ? a->GetEntries() : 0;
498  if (transform && transform != identityTransform()) {
499  string nam = string(daughter->GetName()) + "_placement";
500  transform->SetName(nam.c_str());
501  }
502  TGeoShape* shape = daughter->GetShape();
503  // Need to fix the daughter's BBox of assemblies, if the BBox was not calculated....
504  if ( shape->IsA() == TGeoShapeAssembly::Class() ) {
505  TGeoShapeAssembly* as = (TGeoShapeAssembly*)shape;
506  if ( std::fabs(as->GetDX()) < numeric_limits<double>::epsilon() &&
507  std::fabs(as->GetDY()) < numeric_limits<double>::epsilon() &&
508  std::fabs(as->GetDZ()) < numeric_limits<double>::epsilon() ) {
509  as->NeedsBBoxRecompute();
510  as->ComputeBBox();
511  }
512  }
513  parent->AddNode(daughter, id, transform);
514  geo_node_t* n = static_cast<geo_node_t*>(parent->GetNode(id));
515  n->geo_node_t::SetUserExtension(new PlacedVolume::Object());
516  return PlacedVolume(n);
517 }
