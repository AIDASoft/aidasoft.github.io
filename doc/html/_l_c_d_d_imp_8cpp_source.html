190 LCDD& LCDDImp::addDetector(const Ref_t& ref_det) {
191  DetElement det_element(ref_det);
192  LCDDHelper helper(this);
193  DetElement existing_det = helper.detectorByID(det_element.id());
194 
195  if ( existing_det.isValid() ) {
196  SensitiveDetector sd = helper.sensitiveDetector(existing_det);
197  if ( sd.isValid() ) {
198  stringstream str;
199  str << "LCDD: The sensitive sub-detectors " << det_element.name() << " and "
200  << existing_det.name() << " have the identical ID:" << det_element.id() << ".";
201  printout(ERROR,"LCDD",str.str());
202  throw runtime_error(str.str());
203  }
204  }
205  m_detectors.append(ref_det);
206  det_element->flag |= DetElement::Object::IS_TOP_LEVEL_DETECTOR;
207  Volume volume = det_element.placement()->GetMotherVolume();
208  if ( volume == m_worldVol ) {
209  printout(DEBUG,"LCDD","Added detector %s to the world instance.",det_element.name());
210  m_world.add(det_element);
211  return *this;
212  }
213  // The detector's placement must be one of the existing detectors
214  for(HandleMap::iterator i = m_detectors.begin(); i!=m_detectors.end(); ++i) {
215  DetElement parent((*i).second);
216  Volume vol = parent.placement().volume();
217  if ( vol == volume ) {
218  printout(INFO,"LCDD","Added detector %s to the parent:%s.",det_element.name(),parent.name());
219  parent.add(det_element);
220  return *this;
221  }
222  }
223  throw runtime_error("LCDD: The detector " + string(det_element.name()) + " has no known parent.");
224 }

402  void patchShapes() {
403  GeoHandler::Data& data = *m_data;
404  char text[32];
405  string nam;
406  printout(INFO,"LCDD","+++ Patching names of anonymous shapes....");
407  for (GeoHandler::Data::const_reverse_iterator i = data.rbegin(); i != data.rend(); ++i) {
408  const GeoHandler::Data::mapped_type& v = (*i).second;
409  for (GeoHandler::Data::mapped_type::const_iterator j = v.begin(); j != v.end(); ++j) {
410  const TGeoNode* n = *j;
411  TGeoVolume* vol = n->GetVolume();
412  TGeoShape* s = vol->GetShape();
413  const char* sn = s->GetName();
414  ::snprintf(text,sizeof(text),"_shape_%p",(void*)s);
415  if (0 == sn || 0 == ::strlen(sn)) {
416  nam = vol->GetName();
417  nam += text;
418  s->SetName(nam.c_str());
419  }
420  else if (0 == ::strcmp(sn, s->IsA()->GetName())) {
421  nam = vol->GetName();
422  nam += text;
423  s->SetName(nam.c_str());
424  }
425  else {
426  nam = sn;
427  if (nam.find("_shape") == string::npos)
428  nam += text;
429  s->SetName(nam.c_str());
430  }
431  if (s->IsA() == TGeoCompositeShape::Class()) {
432  TGeoCompositeShape* c = (TGeoCompositeShape*) s;
433  const TGeoBoolNode* boolean = c->GetBoolNode();
434  s = boolean->GetLeftShape();
435  sn = s->GetName();
436  if (0 == sn || 0 == ::strlen(sn)) {
437  s->SetName((nam + "_left").c_str());
438  }
439  else if (0 == ::strcmp(sn, s->IsA()->GetName())) {
440  s->SetName((nam + "_left").c_str());
441  }
442  s = boolean->GetRightShape();
443  sn = s->GetName();
444  if (0 == sn || 0 == ::strlen(sn)) {
445  s->SetName((nam + "_right").c_str());
446  }
447  else if (0 == ::strcmp(s->GetName(), s->IsA()->GetName())) {
448  s->SetName((nam + "_right").c_str());
449  }
450  }
451  }
452  }
