385 string DD4hep::XML::getEnviron(const string& env) {
386  size_t id1 = env.find("${");
387  size_t id2 = env.rfind("}");
388  if ( id1 == string::npos || id2 == string::npos ) {
389  return "";
390  }
391  else {
392  string v = env.substr(0,id2+1);
393  const char* ret = eval.getEnviron(v.c_str());
394  if (eval.status() != XmlTools::Evaluator::OK) {
395  cerr << env << ": ";
396  eval.print_error();
397  throw runtime_error("DD4hep: Severe error during environment lookup of " + env);
398  }
399  v = env.substr(0,id1);
400  v += ret;
401  v += env.substr(id2+1);
402  return v;
403  }

943 static unsigned int adler32(unsigned int adler, const XmlChar* xml_buff, size_t len) {
944 #define DO1(buf,i) {s1 +=(unsigned char)buf[i]; s2 += s1;}
945 #define DO2(buf,i) DO1(buf,i); DO1(buf,i+1);
946 #define DO4(buf,i) DO2(buf,i); DO2(buf,i+2);
947 #define DO8(buf,i) DO4(buf,i); DO4(buf,i+4);
948 #define DO16(buf) DO8(buf,0); DO8(buf,8);
949 
950  static const unsigned int BASE = 65521; /* largest prime smaller than 65536 */
951  /* NMAX is the largest n such that 255n(n+1)/2 + (n+1)(BASE-1) <= 2^32-1 */
952  static const unsigned int NMAX = 5550;
953  unsigned int s1 = adler & 0xffff;
954  unsigned int s2 = (adler >> 16) & 0xffff;
955  const char* buf = (const char*)xml_buff;
956 
957  if (buf == NULL)
958  return 1;
959 
960  while (len > 0) {
961  int k = len < NMAX ? (int) len : NMAX;
962  len -= k;
963  while (k >= 16) {
964  DO16(buf);
965  buf += 16;
966  k -= 16;
967  }
968  if (k != 0)
969  do {
970  s1 += (unsigned char) *buf++;
971  s2 += s1;
972  } while (--k);
973  s1 %= BASE;
974  s2 %= BASE;
975  }
976  unsigned int result = (s2 << 16) | s1;
977  return result;
978 }
