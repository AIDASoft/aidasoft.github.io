00943 static unsigned int adler32(unsigned int adler, const XmlChar* xml_buff, size_t len) {
00944 #define DO1(buf,i)  {s1 +=(unsigned char)buf[i]; s2 += s1;}
00945 #define DO2(buf,i)  DO1(buf,i); DO1(buf,i+1);
00946 #define DO4(buf,i)  DO2(buf,i); DO2(buf,i+2);
00947 #define DO8(buf,i)  DO4(buf,i); DO4(buf,i+4);
00948 #define DO16(buf)   DO8(buf,0); DO8(buf,8);
00949 
00950   static const unsigned int BASE = 65521; /* largest prime smaller than 65536 */
00951   /* NMAX is the largest n such that 255n(n+1)/2 + (n+1)(BASE-1) <= 2^32-1 */
00952   static const unsigned int NMAX = 5550;
00953   unsigned int s1 = adler & 0xffff;
00954   unsigned int s2 = (adler >> 16) & 0xffff;
00955   const char* buf = (const char*)xml_buff;
00956 
00957   if (buf == NULL)
00958     return 1;
00959 
00960   while (len > 0) {
00961     int k = len < NMAX ? (int) len : NMAX;
00962     len -= k;
00963     while (k >= 16) {
00964       DO16(buf);
00965       buf += 16;
00966       k -= 16;
00967     }
00968     if (k != 0)
00969       do {
00970         s1 += (unsigned char) *buf++;
00971         s2 += s1;
00972       } while (--k);
00973     s1 %= BASE;
00974     s2 %= BASE;
00975   }
00976   unsigned int result = (s2 << 16) | s1;
00977   return result;
00978 }
