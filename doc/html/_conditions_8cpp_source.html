62 string Condition::str(int flags) const {
63  stringstream output;
64  Object* o = access();
65  const IOV* ptr_iov = o->iovData();
66  if ( 0 == (flags&NO_NAME) )
67  output << setw(16) << left << o->name;
68  if ( flags&WITH_IOV )
69  output << " " << (ptr_iov ? ptr_iov->str().c_str() : "IOV:[UNKNOWN]");
70  if ( flags&WITH_TYPE )
71  output << " (" << o->type << ")";
72  if ( flags&WITH_ADDRESS )
73  output << " " << o->address;
74  if ( flags&WITH_DATATYPE )
75  output << " -> " << o->data.dataType();
76  if ( flags&WITH_DATA )
77  output << " Data:" << o->data.str();
78  if ( flags&WITH_COMMENT )
79  output << " \"" << o->comment << "\"";
80  return output.str();
81 }
