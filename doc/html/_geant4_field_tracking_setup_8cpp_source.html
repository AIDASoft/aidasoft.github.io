200 int Geant4FieldTrackingSetup::execute(Geometry::LCDD& lcdd) {
201  Geometry::OverlayedField fld = lcdd.field();
202  G4MagneticField* mag_field = new Simulation::Geant4Field(fld);
203  G4Mag_EqRhs* mag_equation = PluginService::Create<G4Mag_EqRhs*>(eq_typ,mag_field);
204  G4MagIntegratorStepper* fld_stepper = PluginService::Create<G4MagIntegratorStepper*>(stepper_typ,mag_equation);
205  G4ChordFinder* chordFinder = new G4ChordFinder(mag_field,min_chord_step,fld_stepper);
206  G4TransportationManager* transportMgr = G4TransportationManager::GetTransportationManager();
207  G4PropagatorInField* propagator = transportMgr->GetPropagatorInField();
208  G4FieldManager* fieldManager = transportMgr->GetFieldManager();
209 
210  fieldManager->SetFieldChangesEnergy(fld.changesEnergy());
211  fieldManager->SetDetectorField(mag_field);
212  fieldManager->SetChordFinder(chordFinder);
213 
214  if ( delta_chord >= 0e0 )
215  chordFinder->SetDeltaChord(delta_chord);
216  if ( delta_one_step >= 0e0 )
217  fieldManager->SetAccuraciesWithDeltaOneStep(delta_one_step);
218  if ( delta_intersection >= 0e0 )
219  fieldManager->SetDeltaIntersection(delta_intersection);
220  if ( eps_min >= 0e0 )
221  propagator->SetMinimumEpsilonStep(eps_min);
222  if ( eps_max >= 0e0 )
223  propagator->SetMaximumEpsilonStep(eps_max);
224  if ( largest_step >= 0e0 ) {
225  propagator->SetLargestAcceptableStep(largest_step);
226  } else {
227  largest_step = propagator->GetLargestAcceptableStep();
228  }
229  return 1;
230 }

232 static long setup_fields(lcdd_t& lcdd, const DD4hep::Geometry::GeoHandler& /* cnv */, const map<string,string>& vals) {
233  struct XMLFieldTrackingSetup : public Geant4FieldTrackingSetup {
234  XMLFieldTrackingSetup(const map<string,string>& values) : Geant4FieldTrackingSetup() {
235  Geant4SetupPropertyMap pm(values);
236  lcdd_t::PropertyValues::const_iterator iV = values.find("min_chord_step");
237  eq_typ = pm.value("equation");
238  stepper_typ = pm.value("stepper");
239  min_chord_step = _toDouble((iV==values.end()) ? string("1e-2 * mm") : (*iV).second);
240  if ( pm["eps_min"] ) eps_min = pm.toDouble("eps_min");
241  if ( pm["eps_max"] ) eps_max = pm.toDouble("eps_max");
242  if ( pm["delta_chord"] ) delta_chord = pm.toDouble("delta_chord");
243  if ( pm["delta_one_step"] ) delta_one_step = pm.toDouble("delta_one_step");
244  if ( pm["delta_intersection"] ) delta_intersection = pm.toDouble("delta_intersection");
245  if ( pm["largest_step"] ) largest_step = pm.toDouble("largest_step");
246  }
247  virtual ~XMLFieldTrackingSetup() {}
248  } setup(vals);
249  return setup.execute(lcdd);
250 }
