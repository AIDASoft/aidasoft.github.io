00071 void Geant4IsotropeGenerator::getParticleDirectionEta(int, ROOT::Math::XYZVector& direction, double& momentum) const   {
00072   struct Distribution {
00073     static double eta(double x)  { return -1.0*std::log(std::tan(x/2.0)); }
00074   };
00075   Geant4Event&  evt = context()->event();
00076   Geant4Random& rnd = evt.random();
00077   // See https://en.wikipedia.org/wiki/Pseudorapidity
00078   const double dmin = std::numeric_limits<double>::epsilon();
00079   double phi        = m_phiMin+(m_phiMax-m_phiMin)*rnd.rndm();
00080   double eta_min    = Distribution::eta(m_thetaMin>dmin ? m_thetaMin : dmin);
00081   double eta_max    = Distribution::eta(m_thetaMax>(M_PI-dmin) ? m_thetaMax : M_PI-dmin);
00082   double eta        = eta_min + (eta_max-eta_min)*rnd.rndm();
00083   double x1         = std::cos(phi);
00084   double x2         = std::sin(phi);
00085   double x3         = std::sinh(eta);
00086   double r          = std::sqrt(1.0+x3*x3);
00087   direction.SetXYZ(x1/r,x2/r,x3/r);
00088   momentum = rnd.rndm()*momentum;
00089 }

00092 void Geant4IsotropeGenerator::getParticleDirectionFFbar(int, ROOT::Math::XYZVector& direction, double& momentum) const   {
00093   struct Distribution {
00094     static double ffbar(double x)  { double c = std::cos(x); return 1 + c*c; }
00095     //static double integral(double x)  { return 1.5*x + sin(2.*x)/4.0; }
00096   };
00097   Geant4Event&  evt = context()->event();
00098   Geant4Random& rnd = evt.random();
00099   double phi = m_phiMin+(m_phiMax-m_phiMin)*rnd.rndm();
00100 
00101   // 1 + cos^2(theta) cannot be integrated and then inverted.
00102   // We have to throw the dice until it fits.....
00103   double v1 = Distribution::ffbar(m_thetaMin), v2 = Distribution::ffbar(m_thetaMax);
00104   double vmax = std::max(v1,v2); // ffbar symmetric and monotonic in |x|.
00105   while(1)  {
00106     double dice  = rnd.rndm()*vmax;
00107     double theta = m_thetaMin+(m_thetaMax-m_thetaMin)*rnd.rndm();
00108     double dist  = Distribution::ffbar(theta);
00109     if ( dice <= dist )   {
00110       double x1 = std::sin(theta)*std::cos(phi);
00111       double x2 = std::sin(theta)*std::sin(phi);
00112       double x3 = std::cos(theta);
00113       direction.SetXYZ(x1,x2,x3);
00114       momentum = rnd.rndm()*momentum;
00115       return;
00116     }
00117   }
00118 }
