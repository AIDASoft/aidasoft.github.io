56 void Geant4IsotropeGenerator::getParticleDirectionCosTheta(int, ROOT::Math::XYZVector& direction, double& momentum) const {
57  Geant4Event& evt = context()->event();
58  Geant4Random& rnd = evt.random();
59  double phi = m_phiMin+(m_phiMax-m_phiMin)*rnd.rndm();
60  double cos_theta = std::cos(m_thetaMin)+(std::cos(m_thetaMax)-std::cos(m_thetaMin))*rnd.rndm();
61  double sin_theta = std::sqrt(1.0-cos_theta*cos_theta);
62  double x1 = sin_theta*std::cos(phi);
63  double x2 = sin_theta*std::sin(phi);
64  double x3 = cos_theta;
65 
66  direction.SetXYZ(x1,x2,x3);
67  momentum = rnd.rndm()*momentum;
68 }

71 void Geant4IsotropeGenerator::getParticleDirectionEta(int, ROOT::Math::XYZVector& direction, double& momentum) const {
72  struct Distribution {
73  static double eta(double x) { return -1.0*std::log(std::tan(x/2.0)); }
74  };
75  Geant4Event& evt = context()->event();
76  Geant4Random& rnd = evt.random();
77  // See https://en.wikipedia.org/wiki/Pseudorapidity
78  const double dmin = std::numeric_limits<double>::epsilon();
79  double phi = m_phiMin+(m_phiMax-m_phiMin)*rnd.rndm();
80  double eta_min = Distribution::eta(m_thetaMin>dmin ? m_thetaMin : dmin);
81  double eta_max = Distribution::eta(m_thetaMax>(M_PI-dmin) ? m_thetaMax : M_PI-dmin);
82  double eta = eta_min + (eta_max-eta_min)*rnd.rndm();
83  double x1 = std::cos(phi);
84  double x2 = std::sin(phi);
85  double x3 = std::sinh(eta);
86  double r = std::sqrt(1.0+x3*x3);
87  direction.SetXYZ(x1/r,x2/r,x3/r);
88  momentum = rnd.rndm()*momentum;
89 }

92 void Geant4IsotropeGenerator::getParticleDirectionFFbar(int, ROOT::Math::XYZVector& direction, double& momentum) const {
93  struct Distribution {
94  static double ffbar(double x) { double c = std::cos(x); return 1 + c*c; }
95  //static double integral(double x) { return 1.5*x + sin(2.*x)/4.0; }
96  };
97  Geant4Event& evt = context()->event();
98  Geant4Random& rnd = evt.random();
99  double phi = m_phiMin+(m_phiMax-m_phiMin)*rnd.rndm();
100 
101  // 1 + cos^2(theta) cannot be integrated and then inverted.
102  // We have to throw the dice until it fits.....
103  double v1 = Distribution::ffbar(m_thetaMin), v2 = Distribution::ffbar(m_thetaMax);
104  double vmax = std::max(v1,v2); // ffbar symmetric and monotonic in |x|.
105  while(1) {
106  double dice = rnd.rndm()*vmax;
107  double theta = m_thetaMin+(m_thetaMax-m_thetaMin)*rnd.rndm();
108  double dist = Distribution::ffbar(theta);
109  if ( dice <= dist ) {
110  double x1 = std::sin(theta)*std::cos(phi);
111  double x2 = std::sin(theta)*std::sin(phi);
112  double x3 = std::cos(theta);
113  direction.SetXYZ(x1,x2,x3);
114  momentum = rnd.rndm()*momentum;
115  return;
116  }
117  }
118 }

121 void Geant4IsotropeGenerator::getParticleDirection(int num, ROOT::Math::XYZVector& direction, double& momentum) const {
122  switch(::toupper(m_distribution[0])) {
123  case 'C': // cos(theta)
124  return getParticleDirectionCosTheta(num, direction, momentum);
125  case 'F': // ffbar: ~ 1 + cos^2(theta)
126  return getParticleDirectionFFbar(num, direction, momentum);
127  case 'E': // eta, rapidity, pseudorapidity
128  case 'P':
129  case 'R':
130  return getParticleDirectionEta(num, direction, momentum);
131  case 'U': // uniform
132  return getParticleDirectionUniform(num, direction, momentum);
133  default:
134  break;
135  }
136  except("Unknown distribution densitiy: %s. Cannot generate primaries.",
137  m_distribution.c_str());
138 }
