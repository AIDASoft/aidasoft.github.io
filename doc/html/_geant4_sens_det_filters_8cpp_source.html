152 const G4ParticleDefinition* ParticleFilter::definition() const {
153  if ( m_definition ) return m_definition;
154  m_definition = G4ParticleTable::GetParticleTable()->FindParticle(m_particle);
155  if ( 0 == m_definition ) {
156  throw runtime_error("Invalid particle name:'"+m_particle+"' [Not-in-particle-table]");
157  }
158  return m_definition;
159 }

168 bool ParticleFilter::isGeantino(const G4Track* track) const {
