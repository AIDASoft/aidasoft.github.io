110 G4bool Geant4SensitiveDetector::process(G4Step* step, G4TouchableHistory* hist) {
111  double ene_cut = m_sensitive.energyCutoff();
112  if (step->GetTotalEnergyDeposit() > ene_cut) {
113  if (!Geant4Hit::isGeantino(step->GetTrack())) {
114 #if DEBUG
115  dumpStep(step, hist);
116 #endif
117  return buildHits(step, hist);
118  }
119  }
120 #if DEBUG
121  std::cout << " *** too small energy deposit : " << step->GetTotalEnergyDeposit()
122  << " < " << ene_cut << " at " << step->GetPreStepPoint()->GetPosition() << std::endl;
123 #endif
124  return false;
125 }

152 void Geant4SensitiveDetector::dumpStep(G4Step* st, G4TouchableHistory* /* history */) {
153  Geant4StepHandler step(st);
154  Geant4Mapping& cnv = Geant4Mapping::instance();
155  //Geant4Converter::G4GeometryInfo& data = cnv.data();
156 
157  Position pos1 = step.prePos();
158  Position pos2 = step.postPos();
159  Momentum mom = step.postMom();
160 
161  printout(INFO, "G4Step", " Track:%08ld Pos:(%8f %8f %8f) -> (%f %f %f) Mom:%7.0f %7.0f %7.0f",
162  long(step.track), pos1.X(),
163  pos1.Y(), pos1.Z(), pos2.X(), pos2.Y(), pos2.Z(), mom.X(), mom.Y(), mom.Z());
164  printout(INFO, "G4Step", " pre-Vol: %s Status:%s",
165  step.preVolume()->GetName().c_str(), step.preStepStatus());
166  printout(INFO, "G4Step", " post-Vol:%s Status:%s",
167  step.postVolume()->GetName().c_str(),
168  step.postStepStatus());
169 
170  const G4VPhysicalVolume* pv = step.volume(step.post);
171 
172  typedef Geant4GeometryMaps::PlacementMap Places;
173  const Places& places = cnv.data().g4Placements;
174 
175  for (Places::const_iterator i = places.begin(); i != places.end(); ++i) {
176  const G4VPhysicalVolume* pl = (*i).second;
177  const G4VPhysicalVolume* qv = pl;
178 
179  if (qv == pv) {
180  const TGeoNode* tpv = (*i).first.ptr();
181  printf(" Found TGeoNode:%s!\n", tpv->GetName());
182  }
183  }

220 long long Geant4SensitiveDetector::getCellID(G4Step* s) {
221  StepHandler h(s);
222  Geant4VolumeManager volMgr = Geant4Mapping::instance().volumeManager();
223  VolumeID volID = volMgr.volumeID(h.preTouchable());
224  Geometry::Segmentation seg = m_readout.segmentation();
225  if ( seg.isValid() ) {
226  G4ThreeVector global = 0.5 * ( h.prePosG4()+h.postPosG4());
227  G4ThreeVector local = h.preTouchable()->GetHistory()->GetTopTransform().TransformPoint(global);
228  Position loc(local.x()*MM_2_CM, local.y()*MM_2_CM, local.z()*MM_2_CM);
229  Position glob(global.x()*MM_2_CM, global.y()*MM_2_CM, global.z()*MM_2_CM);
230  VolumeID cID = seg.cellID(loc,glob,volID);
231  return cID;
232  }
233  return volID;
