58 void Geant4OutputAction::end(const G4Event* evt) {
59  OutputContext < G4Event > ctxt(evt);
60  G4HCofThisEvent* hce = evt->GetHCofThisEvent();
61  if ( hce ) {
62  int nCol = hce->GetNumberOfCollections();
63  try {
64  m_truth = context()->event().extension<Geant4ParticleMap>(false);
65  if ( m_truth && !m_truth->isValid() ) {
66  m_truth = 0;
67  printout(WARNING,name(),"+++ [Event:%d] No valid MC truth info present. "
68  "Is a Particle handler installed ?",evt->GetEventID());
69  }
70  try {
71  saveEvent(ctxt);
72  for (int i = 0; i < nCol; ++i) {
73  G4VHitsCollection* hc = hce->GetHC(i);
74  saveCollection(ctxt, hc);
75  }
76  }
77  catch(const exception& e) {
78  printout(ERROR,name(),"+++ [Event:%d] Exception while saving event:%s",
79  evt->GetEventID(),e.what());
80  if ( m_errorFatal ) throw;
81  }
82  catch(...) {
83  printout(ERROR,name(),"+++ [Event:%d] UNKNWON Exception while saving event",
84  evt->GetEventID());
85  if ( m_errorFatal ) throw;
86  }
87  commit(ctxt);
88  }
89  catch(const exception& e) {
90  printout(ERROR,name(),"+++ [Event:%d] Exception while saving event:%s",
91  evt->GetEventID(),e.what());
92  if ( m_errorFatal ) throw;
93  }
94  catch(...) {
95  printout(ERROR,name(),"+++ [Event:%d] UNKNWON Exception while saving event",
96  evt->GetEventID());
97  if ( m_errorFatal ) throw;
98  }
99  m_truth = 0;
100  return;
101  }
102  printout(WARNING,"Geant4OutputAction",
103  "+++ The value of G4HCofThisEvent is NULL. No collections saved!");
104 }
