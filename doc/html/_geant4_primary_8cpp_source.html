91 bool Geant4PrimaryInteraction::applyMask() {
92  if ( vertices.size() <= 1 ) {
93  Geant4PrimaryInteraction::ParticleMap::iterator ip, ipend;
94  for( ip=particles.begin(), ipend=particles.end(); ip != ipend; ++ip )
95  (*ip).second->mask = mask;
96 
97  Geant4PrimaryInteraction::VertexMap::iterator iv, ivend;
98  for( iv=vertices.begin(), ivend=vertices.end(); iv != ivend; ++iv )
99  (*iv).second->mask = mask;
100  return true;
101  }
102  return false;
103 }

127 void Geant4PrimaryEvent::add(int id, Geant4PrimaryInteraction* interaction) {
128  if ( interaction ) {
129  Interactions::iterator i = m_interactions.find(id);
130  if ( i == m_interactions.end() ) {
131  interaction->mask = id;
132  m_interactions.insert(std::make_pair(id,interaction));
133  return;
134  }
135  char text[132];
136  ::snprintf(text,sizeof(text),"Geant4PrimaryEvent: Interaction with ID '%d' "
137  "exists and cannot be added twice!",id);
138  throw std::runtime_error(text);
139  }
140  throw std::runtime_error("Geant4PrimaryEvent: CANNOT add invalid Interaction!");
141 }
