27 Geant4ReadoutVolumeFilter::Geant4ReadoutVolumeFilter(Geant4Context* ctxt,

35  const Collection& c = ro->hits[i];

37  m_collection = &c;
38  m_key = ro.idSpec().field(c.key);

40  }
41  }
42  except("+++ Custom collection name '%s' not defined in the Readout object: %s.",
