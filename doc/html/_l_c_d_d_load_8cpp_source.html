86  string tag = xml_root.tag();
87  string type = tag + "_XML_reader";
88  XML::Handle_t handle = xml_root;
89  long result = PluginService::Create<long>(type, m_lcdd, &handle);
90  if (0 == result) {
91  PluginDebug dbg;
92  result = PluginService::Create<long>(type, m_lcdd, &handle);
93  if ( 0 == result ) {
94  throw runtime_error("DD4hep: Failed to locate plugin to interprete files of type"
95  " \"" + tag + "\" - no factory:" + type + ". " + dbg.missingFactory(type));
96  }
97  }
98  result = *(long*) result;
99  if (result != 1) {
100  throw runtime_error("DD4hep: Failed to parse the XML file " + xmlfile + " with the plugin " + type);
