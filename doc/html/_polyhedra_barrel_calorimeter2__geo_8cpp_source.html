25 static void placeStaves(DetElement& parent, DetElement& stave, double rmin, int numsides, double total_thickness,
26  Volume envelopeVolume, double innerAngle, Volume sectVolume) {
27  double innerRotation = innerAngle;
28  double offsetRotation = -innerRotation / 2;
29  double sectCenterRadius = rmin + total_thickness / 2;
30  double rotX = M_PI / 2;
31  double rotY = -offsetRotation;
32  double posX = -sectCenterRadius * std::sin(rotY);
33  double posY = sectCenterRadius * std::cos(rotY);
34 
35  for (int module = 1; module <= numsides; ++module) {
36  DetElement det = module > 1 ? stave.clone(_toString(module,"stave%d")) : stave;
37  Transform3D trafo(RotationZYX(0, rotY, rotX), Translation3D(-posX, -posY, 0));
38  PlacedVolume pv = envelopeVolume.placeVolume(sectVolume,trafo);
39  // Not a valid volID: pv.addPhysVolID("stave", 0);
40  pv.addPhysVolID("module", module);
41  det.setPlacement(pv);
42  parent.add(det);
43  rotY -= innerRotation;
44  posX = -sectCenterRadius * std::sin(rotY);
45  posY = sectCenterRadius * std::cos(rotY);
46  }
47 }

49 static Ref_t create_detector(LCDD& lcdd, xml_h e, SensitiveDetector sens) {
50  xml_det_t x_det = e;
51  Layering layering(x_det);
52  xml_comp_t staves = x_det.staves();
53  xml_dim_t dim = x_det.dimensions();
54  string det_name = x_det.nameStr();
55  Material air = lcdd.air();
56  double totalThickness = layering.totalThickness();
57  double gap = xml_dim_t(x_det).gap();
58  int numSides = dim.numsides();
59  double detZ = dim.z();
60  double rmin = dim.rmin();
61  DetElement sdet(det_name, x_det.id());
62  DetElement stave("stave1", x_det.id());
63  Volume motherVol = lcdd.pickMotherVolume(sdet);
64 
65 #if 0
66  int totalRepeat = 0;
67  int totalSlices = 0;
68  for (xml_coll_t c(x_det, _U(layer)); c; ++c) {
69  xml_comp_t x_layer = c;
70  int repeat = x_layer.repeat();
71  totalRepeat += repeat;
72  totalSlices += x_layer.numChildren(_U(slice));
73  }
74 #endif
75  PolyhedraRegular polyhedra(numSides, rmin, rmin + totalThickness, detZ);
76  Volume envelopeVol(det_name, polyhedra, air);
77 
78  // Add the subdetector envelope to the structure.
79  double innerAngle = 2 * M_PI / numSides;
80  double halfInnerAngle = innerAngle / 2;
81  double tan_inner = std::tan(halfInnerAngle) * 2;
82  double innerFaceLen = rmin * tan_inner;
83  double outerFaceLen = (rmin + totalThickness) * tan_inner;
84  double staveThickness = totalThickness;
85 
86  Trapezoid staveTrdOuter(innerFaceLen / 2, outerFaceLen / 2, detZ / 2, detZ / 2, staveThickness / 2);
87  Volume staveOuterVol("stave_outer", staveTrdOuter, air);
88 
89  Trapezoid staveTrdInner(innerFaceLen / 2 - gap, outerFaceLen / 2 - gap, detZ / 2, detZ / 2, staveThickness / 2);
90  Volume staveInnerVol("stave_inner", staveTrdInner, air);
91 
92  double layerOuterAngle = (M_PI - innerAngle) / 2;
93  double layerInnerAngle = (M_PI / 2 - layerOuterAngle);
94  double layer_pos_z = -(staveThickness / 2);
95  double layer_dim_x = innerFaceLen / 2 - gap * 2;
96  int layer_num = 1;
97 
98  //#### LayeringExtensionImpl* layeringExtension = new LayeringExtensionImpl();
99  //#### Position layerNormal(0,0,1);
100 
101  for (xml_coll_t c(x_det, _U(layer)); c; ++c) {
102  xml_comp_t x_layer = c;
103  int repeat = x_layer.repeat(); // Get number of times to repeat this layer.
104  const Layer* lay = layering.layer(layer_num - 1); // Get the layer from the layering engine.
105  // Loop over repeats for this layer.
106  for (int j = 0; j < repeat; j++) {
107  string layer_name = _toString(layer_num, "layer%d");
108  double layer_thickness = lay->thickness();
109  DetElement layer(stave, layer_name, layer_num);
110  //### layeringExtension->setLayer(layer_num, layer, layerNormal);
111 
112  // Layer position in Z within the stave.
113  layer_pos_z += layer_thickness / 2;
114  // Layer box & volume
115  Volume layer_vol(layer_name, Box(layer_dim_x, detZ / 2, layer_thickness / 2), air);
116 
117  // Create the slices (sublayers) within the layer.
118  double slice_pos_z = -(layer_thickness / 2);
119  int slice_number = 1;
120  for (xml_coll_t k(x_layer, _U(slice)); k; ++k) {
121  xml_comp_t x_slice = k;
122  string slice_name = _toString(slice_number, "slice%d");
123  double slice_thickness = x_slice.thickness();
124  Material slice_material = lcdd.material(x_slice.materialStr());
125  DetElement slice(layer, slice_name, slice_number);
126 
127  slice_pos_z += slice_thickness / 2;
128  // Slice volume & box
129  Volume slice_vol(slice_name, Box(layer_dim_x, detZ / 2, slice_thickness / 2), slice_material);
130 
131  if (x_slice.isSensitive()) {
132  sens.setType("calorimeter");
133  slice_vol.setSensitiveDetector(sens);
134  }
135  // Set region, limitset, and vis.
136  slice_vol.setAttributes(lcdd, x_slice.regionStr(), x_slice.limitsStr(), x_slice.visStr());
137  // slice PlacedVolume
138  PlacedVolume slice_phv = layer_vol.placeVolume(slice_vol, Position(0, 0, slice_pos_z));
139  slice_phv.addPhysVolID("slice", slice_number);
140 
141  slice.setPlacement(slice_phv);
142  // Increment Z position for next slice.
143  slice_pos_z += slice_thickness / 2;
144  // Increment slice number.
145  ++slice_number;
146  }
147  // Set region, limitset, and vis.
148  layer_vol.setAttributes(lcdd, x_layer.regionStr(), x_layer.limitsStr(), x_layer.visStr());
149 
150  // Layer physical volume.
151  PlacedVolume layer_phv = staveInnerVol.placeVolume(layer_vol, Position(0, 0, layer_pos_z));
152  layer_phv.addPhysVolID("layer", layer_num);
153  layer.setPlacement(layer_phv);
154 
155  // Increment the layer X dimension.
156  layer_dim_x += layer_thickness * std::tan(layerInnerAngle); // * 2;
157  // Increment the layer Z position.
158  layer_pos_z += layer_thickness / 2;
159  // Increment the layer number.
160  ++layer_num;
161  }
162  }
163 
164  // Add stave inner physical volume to outer stave volume.
165  staveOuterVol.placeVolume(staveInnerVol);
166  if ( staves ) {
167  // Set the vis attributes of the outer stave section.
168  stave.setVisAttributes(lcdd, staves.visStr(), staveInnerVol);
169  stave.setVisAttributes(lcdd, staves.visStr(), staveOuterVol);
170  }
171  // Place the staves.
172  placeStaves(sdet, stave, rmin, numSides, totalThickness, envelopeVol, innerAngle, staveOuterVol);
173  // Set envelope volume attributes.
174  envelopeVol.setAttributes(lcdd, x_det.regionStr(), x_det.limitsStr(), x_det.visStr());
175 
176  double z_offset = dim.hasAttr(_U(z_offset)) ? dim.z_offset() : 0.0;
177  Transform3D transform(RotationZ(M_PI / numSides), Translation3D(0, 0, z_offset));
178  PlacedVolume env_phv = motherVol.placeVolume(envelopeVol, transform);
179  env_phv.addPhysVolID("system", sdet.id());
180  env_phv.addPhysVolID("barrel", 0);
181  sdet.setPlacement(env_phv);
182 
183  //#### sdet.addExtension<SubdetectorExtension>(new SubdetectorExtensionImpl(sdet));
184  //#### sdet.addExtension<LayeringExtension>(layeringExtension);
185  return sdet;
186 }
