00026                         Volume envelopeVolume, double innerAngle, Volume sectVolume) {
00027   double innerRotation = innerAngle;
00028   double offsetRotation = -innerRotation / 2;
00029   double sectCenterRadius = rmin + total_thickness / 2;
00030   double rotX = M_PI / 2;
00031   double rotY = -offsetRotation;
00032   double posX = -sectCenterRadius * std::sin(rotY);
00033   double posY = sectCenterRadius * std::cos(rotY);

00035   for (int module = 1; module <= numsides; ++module) {
00036     DetElement det = module > 1 ? stave.clone(_toString(module,"stave%d")) : stave;
00037     Transform3D trafo(RotationZYX(0, rotY, rotX), Translation3D(-posX, -posY, 0));
00038     PlacedVolume pv = envelopeVolume.placeVolume(sectVolume,trafo);

00040     pv.addPhysVolID("module", module);

00042     parent.add(det);
00043     rotY -= innerRotation;
00044     posX = -sectCenterRadius * std::sin(rotY);
