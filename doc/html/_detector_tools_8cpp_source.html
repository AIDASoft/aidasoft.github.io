77 bool DetectorTools::findChild(PlacedVolume parent, PlacedVolume child, PlacementPath& path) {
78  if ( parent.isValid() && child.isValid() ) {
79  // Check self
80  if ( parent.ptr() == child.ptr() ) {
81  path.push_back(child);
82  return true;
83  }
84  TIter next(parent->GetVolume()->GetNodes());
85  // Now check next layer children
86  for (TGeoNode *daughter = (TGeoNode*) next(); daughter; daughter = (TGeoNode*) next()) {
87  if ( daughter == child.ptr() ) {
88  path.push_back(daughter);
89  return true;
90  }
91  }
92  next.Reset();
93  // Finally crawl down the tree
94  for (TGeoNode *daughter = (TGeoNode*) next(); daughter; daughter = (TGeoNode*) next()) {
95  PlacementPath sub_path;
96  bool res = findChild(daughter, child, sub_path);
97  if (res) {
98  path.insert(path.end(), sub_path.begin(), sub_path.end());
99  path.push_back(daughter);
100  return res;
101  }
102  }
103  }
104  return false;
105 }

145 void DetectorTools::elementPath(DetElement parent, DetElement child, ElementPath& detectors) {
146  detectors.clear();

148  if ( parent.ptr() == child.ptr() ) {
149  detectors.push_back(child);
150  return;
151  }
152  ElementPath elements;

154  elements.push_back(par);

156  detectors = elements;
157  return;
158  }
159  }
160  throw runtime_error(string("The detector element ")+parent.name()+string(" is no parent of ")+child.name());
161  }
162  throw runtime_error("Search for parent detector element with invalid handles not allowed.");

228 DetElement DetectorTools::findDaughterElement(DetElement parent, const std::string& subpath) {
229  if ( parent.isValid() ) {
230  size_t idx = subpath.find('/',1);
231  if ( subpath[0] == '/' ) {
232  DetElement top = topElement(parent);
233  if ( idx == string::npos ) return top;
234  return findDaughterElement(top,subpath.substr(idx+1));
235  }
236  if ( idx == string::npos )
237  return parent.child(subpath);
238  string name = subpath.substr(0,idx);
239  DetElement node = parent.child(name);
240  if ( node.isValid() ) {
241  return findDaughterElement(node,subpath.substr(idx+1));
242  }
243  throw runtime_error("DD4hep: DetElement "+parent.path()+" has no child named:"+name+" [No such child]");
244  }
245  throw runtime_error("DD4hep: Cannot determine child with path "+subpath+" from invalid parent [invalid handle]");
246 }

326 void DetectorTools::placementTrafo(const PlacementPath& nodes, bool inverse, TGeoHMatrix& mat) {
327  mat = *gGeoIdentity;

338 PlacedVolume DetectorTools::findNode(PlacedVolume top_place, const std::string& place) {
339  TGeoNode* top = top_place.ptr();
340  const char* path = place.c_str();
341  // Check if a geometry path is valid without changing the state of the navigator.
342  Int_t length = strlen(path);
343  if (!length) return 0;
344  TString spath = path;
345  TGeoVolume *vol;
346  // Check first occurance of a '/'
347  Int_t ind1 = spath.Index("/");
348  if (ind1<0) {
349  // No '/' so we check directly the path against the name of the top
350  if ( strcmp(path,top->GetName()) ) return 0;
351  return top;
352  }
353  Int_t ind2 = ind1;
354  Bool_t end = kFALSE;
355  if (ind1>0) ind1 = -1; // no trailing '/'
356  else ind2 = spath.Index("/", ind1+1);
357 
358  if (ind2<0) ind2 = length;
359  TString name(spath(ind1+1, ind2-ind1-1));
360  if ( name == top->GetName() ) {
361  if (ind2>=length-1) return top;
362  ind1 = ind2;
363  }
364  else {
365  return 0;
366  }
367  TGeoNode *node = top;
368  // Deeper than just top level
369  while (!end) {
370  ind2 = spath.Index("/", ind1+1);
371  if (ind2<0) {
372  ind2 = length;
373  end = kTRUE;
374  }
375  vol = node->GetVolume();
376  name = spath(ind1+1, ind2-ind1-1);
377  node = vol->GetNode(name.Data());
378  if (!node)
379  return 0;
380  else if (ind2>=length-1)
381  return node;
382  ind1 = ind2;
383  }
384  return node;
385 }
