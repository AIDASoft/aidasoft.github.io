00352   }
00353   Int_t ind2 = ind1;
00354   Bool_t end = kFALSE;
00355   if (ind1>0) ind1 = -1;   // no trailing '/'
00356   else ind2 = spath.Index("/", ind1+1);
00357 
00358   if (ind2<0) ind2 = length;

00360   if ( name == top->GetName() ) {
00361     if (ind2>=length-1) return top;
00362     ind1 = ind2;
00363   }
00364   else  {
00365     return 0;
00366   }
00367   TGeoNode *node = top;
00368   // Deeper than just top level
00369   while (!end) {
00370     ind2 = spath.Index("/", ind1+1);
00371     if (ind2<0) {
00372       ind2 = length;
