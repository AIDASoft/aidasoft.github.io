25 static Ref_t create_detector(LCDD& lcdd, xml_h e, SensitiveDetector sens) {
26  typedef vector<PlacedVolume> Placements;
27  xml_det_t x_det = e;
28  Material air = lcdd.air();
29  int det_id = x_det.id();
30  string det_name = x_det.nameStr();
31  DetElement sdet (det_name,det_id);
32  Assembly assembly (det_name);
33  map<string, Volume> volumes;
34  map<string, Placements> sensitives;
35  PlacedVolume pv;
36 
37  sens.setType("tracker");
38  for(xml_coll_t mi(x_det,_U(module)); mi; ++mi) {
39  xml_comp_t x_mod = mi;
40  xml_comp_t m_env = x_mod.child(_U(module_envelope));
41  string m_nam = x_mod.nameStr();
42  Volume m_vol(m_nam,Box(m_env.width()/2,m_env.length()/2,m_env.thickness()/2),air);
43  int ncomponents = 0, sensor_number = 1;
44 
45  if ( volumes.find(m_nam) != volumes.end() ) {
46  printout(ERROR,"SiTrackerBarrel","Logics error in building modules.");
47  throw runtime_error("Logics error in building modules.");
48  }
49  volumes[m_nam] = m_vol;
50  m_vol.setVisAttributes(lcdd.visAttributes(x_mod.visStr()));
51  for(xml_coll_t ci(x_mod,_U(module_component)); ci; ++ci, ++ncomponents) {
52  xml_comp_t x_comp = ci;
53  xml_comp_t x_pos = x_comp.position(false);
54  xml_comp_t x_rot = x_comp.rotation(false);
55  string c_nam = _toString(ncomponents,"component%d");
56  Box c_box(x_comp.width()/2,x_comp.length()/2,x_comp.thickness()/2);
57  Volume c_vol(c_nam,c_box,lcdd.material(x_comp.materialStr()));
58 
59  if ( x_pos && x_rot ) {
60  Position c_pos(x_pos.x(0),x_pos.y(0),x_pos.z(0));
61  RotationZYX c_rot(x_rot.z(0),x_rot.y(0),x_rot.x(0));
62  pv = m_vol.placeVolume(c_vol, Transform3D(c_rot,c_pos));
63  }
64  else if ( x_rot ) {
65  pv = m_vol.placeVolume(c_vol,RotationZYX(x_rot.z(0),x_rot.y(0),x_rot.x(0)));
66  }
67  else if ( x_pos ) {
68  pv = m_vol.placeVolume(c_vol,Position(x_pos.x(0),x_pos.y(0),x_pos.z(0)));
69  }
70  else {
71  pv = m_vol.placeVolume(c_vol);
72  }
73  c_vol.setRegion(lcdd, x_comp.regionStr());
74  c_vol.setLimitSet(lcdd, x_comp.limitsStr());
75  c_vol.setVisAttributes(lcdd, x_comp.visStr());
76  if ( x_comp.isSensitive() ) {
77  pv.addPhysVolID(_U(sensor),sensor_number++);
78  c_vol.setSensitiveDetector(sens);
79  sensitives[m_nam].push_back(pv);
80  }
81  }
82  }
83  for(xml_coll_t li(x_det,_U(layer)); li; ++li) {
84  xml_comp_t x_layer = li;
85  xml_comp_t x_barrel = x_layer.child(_U(barrel_envelope));
86  xml_comp_t x_layout = x_layer.child(_U(rphi_layout));
87  xml_comp_t z_layout = x_layer.child(_U(z_layout)); // Get the <z_layout> element.
88  int lay_id = x_layer.id();
89  string m_nam = x_layer.moduleStr();
90  string lay_nam = _toString(x_layer.id(),"layer%d");
91  Tube lay_tub (x_barrel.inner_r(),x_barrel.outer_r(),x_barrel.z_length()/2);
92  Volume lay_vol (lay_nam,lay_tub,air); // Create the layer envelope volume.
93  double phi0 = x_layout.phi0(); // Starting phi of first module.
94  double phi_tilt = x_layout.phi_tilt(); // Phi tilt of a module.
95  double rc = x_layout.rc(); // Radius of the module center.
96  int nphi = x_layout.nphi(); // Number of modules in phi.
97  double rphi_dr = x_layout.dr(); // The delta radius of every other module.
98  double phi_incr = (M_PI * 2) / nphi; // Phi increment for one module.
99  double phic = phi0; // Phi of the module center.
100  double z0 = z_layout.z0(); // Z position of first module in phi.
101  double nz = z_layout.nz(); // Number of modules to place in z.
102  double z_dr = z_layout.dr(); // Radial displacement parameter, of every other module.
103  Volume m_env = volumes[m_nam];
104  DetElement lay_elt(sdet,_toString(x_layer.id(),"layer%d"),lay_id);
105  Placements& sensVols = sensitives[m_nam];
106 
107  // Z increment for module placement along Z axis.
108  // Adjust for z0 at center of module rather than
109  // the end of cylindrical envelope.
110  double z_incr = nz > 1 ? (2.0 * z0) / (nz - 1) : 0.0;
111  // Starting z for module placement along Z axis.
112  double module_z = -z0;
113  int module = 1;
114 
115  // Loop over the number of modules in phi.
116  for (int ii = 0; ii < nphi; ii++) {
117  double dx = z_dr * std::cos(phic + phi_tilt); // Delta x of module position.
118  double dy = z_dr * std::sin(phic + phi_tilt); // Delta y of module position.
119  double x = rc * std::cos(phic); // Basic x module position.
120  double y = rc * std::sin(phic); // Basic y module position.
121 
122  // Loop over the number of modules in z.
123  for (int j = 0; j < nz; j++) {
124  string module_name = _toString(module,"module%d");
125  DetElement mod_elt(lay_elt,module_name,module);
126  // Module PhysicalVolume.
127  // Transform3D tr(RotationZYX(0,-((M_PI/2)-phic-phi_tilt),M_PI/2),Position(x,y,module_z));
128  //NOTE (Nikiforos, 26/08 Rotations needed to be fixed so that component1 (silicon) is on the outside
129  Transform3D tr(RotationZYX(0,((M_PI/2)-phic-phi_tilt),-M_PI/2),Position(x,y,module_z));
130 
131  pv = lay_vol.placeVolume(m_env,tr);
132  pv.addPhysVolID("module", module);
133  mod_elt.setPlacement(pv);
134  for(size_t ic=0; ic<sensVols.size(); ++ic) {
135  PlacedVolume sens_pv = sensVols[ic];
136  DetElement comp_elt(mod_elt,sens_pv.volume().name(),module);
137  comp_elt.setPlacement(sens_pv);
138  }
139 
141  module++;
142  // Adjust the x and y coordinates of the module.
143  x += dx;
144  y += dy;
145  // Flip sign of x and y adjustments.
146  dx *= -1;
147  dy *= -1;
148  // Add z increment to get next z placement pos.
149  module_z += z_incr;
150  }
151  phic += phi_incr; // Increment the phi placement of module.
152  rc += rphi_dr; // Increment the center radius according to dr parameter.
153  rphi_dr *= -1; // Flip sign of dr parameter.
154  module_z = -z0; // Reset the Z placement parameter for module.
155  }
156  // Create the PhysicalVolume for the layer.
157  pv = assembly.placeVolume(lay_vol); // Place layer in mother
158  pv.addPhysVolID("layer", lay_id); // Set the layer ID.
159  lay_elt.setAttributes(lcdd,lay_vol,x_layer.regionStr(),x_layer.limitsStr(),x_layer.visStr());
160  lay_elt.setPlacement(pv);
161  }
162  sdet.setAttributes(lcdd,assembly,x_det.regionStr(),x_det.limitsStr(),x_det.visStr());
163  assembly.setVisAttributes(lcdd.invisible());
164  pv = lcdd.pickMotherVolume(sdet).placeVolume(assembly);
165  pv.addPhysVolID("system", det_id); // Set the subdetector system ID.
166  pv.addPhysVolID("barrel", 0); // Flag this as a barrel subdetector.
167  sdet.setPlacement(pv);
168  return sdet;
169 }
