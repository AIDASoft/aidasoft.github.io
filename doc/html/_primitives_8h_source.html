39  inline unsigned int hash32(const char* key) {
40  unsigned int hash = 0;
41  const char* k = key;
42  for (; *k; k++) {
43  hash += *k;
44  hash += (hash << 10);
45  hash ^= (hash >> 6);
46  }
47  hash += (hash << 3);
48  hash ^= (hash >> 11); hash += (hash << 15);
49  return hash;
50  }
