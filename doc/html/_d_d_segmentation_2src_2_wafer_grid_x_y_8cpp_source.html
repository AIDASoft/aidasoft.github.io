58 Vector3D WaferGridXY::position(const CellID& cID) const {
59  _decoder->setValue(cID);
60  unsigned int _groupMGWaferIndex;
61  unsigned int _waferIndex;
62  Vector3D cellPosition;
63 
64  _groupMGWaferIndex = (*_decoder)[_identifierMGWaferGroup];
65  _waferIndex = (*_decoder)[_identifierWafer];
66 
67  if ( _waferOffsetX[_groupMGWaferIndex][_waferIndex] > 0 || _waferOffsetX[_groupMGWaferIndex][_waferIndex] < 0 )
68  {
69  cellPosition.X = binToPosition((*_decoder)[_xId].value(), _gridSizeX, _offsetX+_waferOffsetX[_groupMGWaferIndex][_waferIndex]);
70  }
71  else
72  {
73  cellPosition.X = binToPosition((*_decoder)[_xId].value(), _gridSizeX, _offsetX);
74  }
75 
76  if ( _waferOffsetY[_groupMGWaferIndex][_waferIndex] > 0 || _waferOffsetY[_groupMGWaferIndex][_waferIndex] < 0 )
77  {
78  cellPosition.Y = binToPosition((*_decoder)[_yId].value(), _gridSizeY, _offsetY+_waferOffsetY[_groupMGWaferIndex][_waferIndex]);
79  }
80  else
81  {
82  cellPosition.Y = binToPosition((*_decoder)[_yId].value(), _gridSizeY, _offsetY);
83  }
84 
85  return cellPosition;
