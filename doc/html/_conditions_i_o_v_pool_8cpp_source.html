48 void ConditionsIOVPool::selectRange(Condition::key_type key, const Condition::iov_type& req_validity, RangeConditions& result)
49 {
50  const IOV::Key range = req_validity.key();
51  for(Elements::const_iterator i=elements.begin(); i!=elements.end(); ++i) {
52  const IOV::Key& k = (*i).first;
53  if ( IOV::key_is_contained(k,range) )
54  // IOV test contained in key. Take it!
55  (*i).second->select(key, result);
56  else if ( IOV::key_overlaps_lower_end(k,range) )
57  // IOV overlap on test on the lower end of key
58  (*i).second->select(key, result);
59  else if ( IOV::key_overlaps_higher_end(k,range) )
60  // IOV overlap of test on the higher end of key
61  (*i).second->select(key, result);
62  }
63 }
