00068   void _findDef(const string& expression, vector<G4ParticleDefinition*>& results) {
00069     string exp = expression;   //'^'+expression+"$";
00070     G4ParticleTable* pt = G4ParticleTable::GetParticleTable();
00071     G4ParticleTable::G4PTblDicIterator* iter = pt->GetIterator();
00072     char msgbuf[128];
00073     regex_t reg;
00074     int ret = ::regcomp(&reg, exp.c_str(), 0);
00075     if (ret) {
00076       throw runtime_error(format("Geant4PhysicsList", "REGEX: Failed to compile particle name %s", exp.c_str()));
00077     }
00078     results.clear();
00079     iter->reset();
00080     while ((*iter)()) {
00081       G4ParticleDefinition* p = iter->value();
00082       ret = ::regexec(&reg, p->GetParticleName().c_str(), 0, NULL, 0);
00083       if (!ret)
00084         results.push_back(p);
00085       else if (ret == REG_NOMATCH)
00086         continue;
00087       else {
00088         ::regerror(ret, &reg, msgbuf, sizeof(msgbuf));
00089         ::regfree(&reg);
00090         throw runtime_error(format("Geant4PhysicsList", "REGEX: Failed to match particle name %s err=%s", exp.c_str(), msgbuf));
00091       }
00092     }
00093     ::regfree(&reg);
00094   }
