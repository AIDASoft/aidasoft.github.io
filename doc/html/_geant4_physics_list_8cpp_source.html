68  void _findDef(const string& expression, vector<G4ParticleDefinition*>& results) {
69  string exp = expression; //'^'+expression+"$";
70  G4ParticleTable* pt = G4ParticleTable::GetParticleTable();
71  G4ParticleTable::G4PTblDicIterator* iter = pt->GetIterator();
72  char msgbuf[128];
73  regex_t reg;
74  int ret = ::regcomp(&reg, exp.c_str(), 0);
75  if (ret) {
76  throw runtime_error(format("Geant4PhysicsList", "REGEX: Failed to compile particle name %s", exp.c_str()));
77  }
78  results.clear();
79  iter->reset();
80  while ((*iter)()) {
81  G4ParticleDefinition* p = iter->value();
82  ret = ::regexec(&reg, p->GetParticleName().c_str(), 0, NULL, 0);
83  if (!ret)
84  results.push_back(p);
85  else if (ret == REG_NOMATCH)
86  continue;
87  else {
88  ::regerror(ret, &reg, msgbuf, sizeof(msgbuf));
89  ::regfree(&reg);
90  throw runtime_error(format("Geant4PhysicsList", "REGEX: Failed to match particle name %s err=%s", exp.c_str(), msgbuf));
91  }
92  }
93  ::regfree(&reg);
94  }
95 }

252 void Geant4PhysicsList::constructProcesses(G4VUserPhysicsList* physics_pointer) {
253  debug("constructProcesses %p", physics_pointer);
254  for (PhysicsProcesses::const_iterator i = m_processes.begin(); i != m_processes.end(); ++i) {
255  const string& part_name = (*i).first;
256  const ParticleProcesses& procs = (*i).second;
257  vector<G4ParticleDefinition*> defs;
258  _findDef(part_name, defs);
259  if (defs.empty()) {
260  except("Particle:%s Cannot find the corresponding entry in the particle table.", part_name.c_str());
261  }
262  for (vector<G4ParticleDefinition*>::const_iterator id = defs.begin(); id != defs.end(); ++id) {
263  G4ParticleDefinition* particle = *id;
264  G4ProcessManager* mgr = particle->GetProcessManager();
265  for (ParticleProcesses::const_iterator ip = procs.begin(); ip != procs.end(); ++ip) {
266  const Process& p = (*ip);
267  G4VProcess* g4 = PluginService::Create<G4VProcess*>(p.name);
268  if (!g4) { // Error no factory for this process
269  except("Particle:%s -> [%s] Cannot create physics process %s",
270  part_name.c_str(), particle->GetParticleName().c_str(), p.name.c_str());
271  }
272  mgr->AddProcess(g4, p.ordAtRestDoIt, p.ordAlongSteptDoIt, p.ordPostStepDoIt);
273  info("Particle:%s -> [%s] added process %s with flags (%d,%d,%d)",
274  part_name.c_str(), particle->GetParticleName().c_str(), p.name.c_str(),
275  p.ordAtRestDoIt, p.ordAlongSteptDoIt, p.ordPostStepDoIt);
276  }
277  }
278  }

301 G4VUserPhysicsList* Geant4PhysicsListActionSequence::extensionList() {
302  G4VModularPhysicsList* physics = ( m_extends.empty() )
303  ? new EmptyPhysics()
304  : G4PhysListFactory().GetReferencePhysList(m_extends);
305  // Register all physics constructors with the physics list
306  constructPhysics(physics);
307  // Ensure the particles and processes declared are also invoked.

362 void Geant4PhysicsListActionSequence::constructDecays(G4VUserPhysicsList* physics_pointer) {
363  G4ParticleTable* pt = G4ParticleTable::GetParticleTable();
364  G4ParticleTable::G4PTblDicIterator* iter = pt->GetIterator();
365  // Add Decay Process
366  G4Decay* decay = new G4Decay();
367  info("ConstructDecays %p",physics_pointer);
368  iter->reset();
369  while ((*iter)()) {
370  G4ParticleDefinition* p = iter->value();
371  G4ProcessManager* mgr = p->GetProcessManager();
372  if (decay->IsApplicable(*p)) {
373  mgr->AddProcess(decay);
374  // set ordering for PostStepDoIt and AtRestDoIt
375  mgr->SetProcessOrdering(decay, idxPostStep);
376  mgr->SetProcessOrdering(decay, idxAtRest);
377  }
378  }
