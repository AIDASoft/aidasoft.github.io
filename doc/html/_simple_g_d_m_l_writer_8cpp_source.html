49  void getAngles(const Double_t* m, Double_t &phi, Double_t &theta, Double_t &psi) {
50  // Retreive Euler angles.
51  // Check if theta is 0 or 180.
52  if (TMath::Abs(1. - TMath::Abs(m[8])) < 1.e-9) {
53  theta = TMath::ACos(m[8]) * RAD_2_DEGREE;
54  phi = TMath::ATan2(-m[8] * m[1], m[0]) * RAD_2_DEGREE;
55  psi = 0.; // convention, phi+psi matters
56  return;
57  }
58  // sin(theta) != 0
59  phi = TMath::ATan2(m[2], -m[5]);
60  Double_t sphi = TMath::Sin(phi);
61  if (TMath::Abs(sphi) < 1.e-9)
62  theta = -TMath::ASin(m[5] / TMath::Cos(phi)) * RAD_2_DEGREE;
63  else
64  theta = TMath::ASin(m[2] / sphi) * RAD_2_DEGREE;
65  phi *= RAD_2_DEGREE;
66  psi = TMath::ATan2(m[6], m[7]) * RAD_2_DEGREE;
67  }
68 }

71 void* SimpleGDMLWriter::handleVolume(const string& name, const TGeoVolume* volume) const {
72  Volume vol(volume);
73  VisAttr vis = vol.visAttributes();
74  TGeoShape* shape = volume->GetShape();
75  TGeoMedium* medium = volume->GetMedium();
76  int num = volume->GetNdaughters();
77 
78  m_output << "\t\t<volume name=\"" << name << "\">" << endl;
79  m_output << "\t\t\t<solidref ref=\"" << shape->GetName() << "\"/>" << endl;
80  m_output << "\t\t\t<materialref ref=\""
81  << (medium ? medium->GetName() : "UnknownMaterial") << "\"/>" << endl;
82  if (vis.isValid()) {
83  m_output << "\t\t\t<visref ref=\"" << vis.name() << "\"/>" << endl;
84  }
85  if (num > 0) {
86  for (int i = 0; i < num; ++i) {
87  //TGeoNode* n = volume->GetNode(i);
88  TGeoNode* n = volume->GetNode(volume->GetNode(i)->GetName());
89  TGeoVolume* v = n->GetVolume();
90  TGeoMatrix* m = n->GetMatrix();
91  m_output << "\t\t\t<physvol>" << endl;
92  m_output << "\t\t\t\t<volumeref ref=\"" << v->GetName() << "\"/>" << endl;
93  if (m) {
94  if (m->IsTranslation()) {
95  m_output << "\t\t\t\t<positionref ref=\"" << n->GetName() << "_pos\"/>" << endl;
96  }
97  if (m->IsRotation()) {
98  m_output << "\t\t\t\t<rotationref ref=\"" << n->GetName() << "_rot\"/>" << endl;
99  }
100  }
101  m_output << "\t\t\t</physvol>" << endl;
102  }
103  }
104  m_output << "\t\t</volume>" << endl;
105  return 0;
106 }

160  const TGeoBoolNode* boolean = s->GetBoolNode();
161  TGeoBoolNode::EGeoBoolType oper = boolean->GetBooleanOperator();
162 
163  handleSolid(name + "_left", boolean->GetLeftShape());
164  handleSolid(name + "_right", boolean->GetRightShape());
165 
166  if (oper == TGeoBoolNode::kGeoSubtraction)

171  m_output << "\t\t<intersection name=\"" << nn << "\">" << endl;
172 
173  m_output << "\t\t\t<first ref=\"" << nn << "_left\"/>" << endl;
174  m_output << "\t\t\t<second ref=\"" << nn << "_right\"/>" << endl;

178 
179  if (oper == TGeoBoolNode::kGeoSubtraction)
180  m_output << "\t\t</subtraction>" << endl;
181  else if (oper == TGeoBoolNode::kGeoUnion)
182  m_output << "\t\t</union>" << endl;
183  else if (oper == TGeoBoolNode::kGeoIntersection)
184  m_output << "\t\t</intersection>" << endl;
185  }
186  else {
187  cerr << "Failed to handle unknwon solid shape:" << shape->IsA()->GetName() << endl;
188  }
189  }
190  return 0;
191 }

202 void* SimpleGDMLWriter::handleTransformation(const string& name, const TGeoMatrix* m) const {
203  if (m) {
204  if (m->IsTranslation()) {
205  const Double_t* f = m->GetTranslation();
206  m_output << indent << "\t\t<position ";
207  if (!name.empty())
208  m_output << "name=\"" << name << "_pos\" ";
209  m_output << "x=\"" << f[0] << "\" " << "y=\"" << f[1] << "\" " << "z=\"" << f[2] << "\" unit=\"cm\"/>" << endl;
210  }
211  if (m->IsRotation()) {
212  const Double_t* mat = m->GetRotationMatrix();
213  Double_t theta = 0.0, phi = 0.0, psi = 0.0;
214  getAngles(mat, theta, phi, psi);
215  m_output << indent << "\t\t<rotation ";
216  if (!name.empty())
217  m_output << "name=\"" << name << "_rot\" ";
218  m_output << "x=\"" << theta << "\" " << "y=\"" << psi << "\" " << "z=\"" << phi << "\" unit=\"deg\"/>" << endl;
219  }
220  }
221  return 0;
222 }

250 void SimpleGDMLWriter::handleVisualisation(const VisRefs& vis) const {
251  m_output << "\t<display>" << endl;
252  for (VisRefs::const_iterator i = vis.begin(); i != vis.end(); ++i) {
253  VisAttr v = Ref_t(*i);
254  if (v.isValid()) {
255  float r = 1., g = 1., b = 1., alpha = 1.;
256  TColor *color = gROOT->GetColor(v.color());
257  if (color) {
258  color->GetRGB(r, g, b);
259  alpha = color->GetAlpha();
260  }
261  const char* line_style = 0, *draw_style = 0;
262  switch (v.lineStyle()) {
263  case VisAttr::DASHED:
264  line_style = "broken";
265  break;
266  case VisAttr::SOLID:
267  default:
268  line_style = "unbroken";
269  break;
270  }
271  switch (v.drawingStyle()) {
272  case VisAttr::WIREFRAME:
273  draw_style = "wireframe";
274  break;
275  default:
276  break;
277  }
278 
279  m_output << "\t\t<vis name=\"" << v.name() << "\" ";
280  if (line_style)
281  m_output << "linestyle=\"" << line_style << "\" ";
282  if (draw_style)
283  m_output << "drawingStyle=\"" << draw_style << "\" ";
284  m_output << "show_daughters=\"" << (const char*) (v.showDaughters() ? "true" : "false") << "\" " << "visible=\""
285  << (const char*) (v.visible() ? "true" : "false") << "\" >" << endl << "\t\t\t<color R=\"" << r << "\" G=\"" << g
286  << "\" B=\"" << b << "\" alpha=\"" << alpha << "\" />" << endl << "\t\t</vis>" << endl;
287  }
288  }
289  m_output << "\t</display>" << endl;
