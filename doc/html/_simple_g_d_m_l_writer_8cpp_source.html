00049   void getAngles(const Double_t* m, Double_t &phi, Double_t &theta, Double_t &psi) {
00050     // Retreive Euler angles.
00051     // Check if theta is 0 or 180.
00052     if (TMath::Abs(1. - TMath::Abs(m[8])) < 1.e-9) {
00053       theta = TMath::ACos(m[8]) * RAD_2_DEGREE;
00054       phi = TMath::ATan2(-m[8] * m[1], m[0]) * RAD_2_DEGREE;
00055       psi = 0.;   // convention, phi+psi matters
00056       return;
00057     }
00058     // sin(theta) != 0
00059     phi = TMath::ATan2(m[2], -m[5]);
00060     Double_t sphi = TMath::Sin(phi);
00061     if (TMath::Abs(sphi) < 1.e-9)
00062       theta = -TMath::ASin(m[5] / TMath::Cos(phi)) * RAD_2_DEGREE;
00063     else
00064       theta = TMath::ASin(m[2] / sphi) * RAD_2_DEGREE;
00065     phi *= RAD_2_DEGREE;
00066     psi = TMath::ATan2(m[6], m[7]) * RAD_2_DEGREE;
00067   }
00068 }
