53 void Geant4ParticleGun::operator()(G4Event* event) {
54  if ( m_isotrop ) {
55  double mom = 0.0;
56  this->Geant4IsotropeGenerator::getParticleDirection(0,m_direction,mom);
57  }
58  else {
59  double r = m_direction.R(), eps = numeric_limits<float>::epsilon();
60  if ( r > eps ) {
61  m_direction.SetXYZ(m_direction.X()/r, m_direction.Y()/r, m_direction.Z()/r);
62  }
63  }
64 
65  if ( m_standalone ) {
66  generationInitialization(this,context());
67  }
68  this->Geant4ParticleGenerator::operator()(event);
69  print("Shoot [%d] %.3f GeV %s pos:(%.3f %.3f %.3f)[mm] dir:(%6.3f %6.3f %6.3f)",
70  m_shotNo, m_energy/CLHEP::GeV, m_particleName.c_str(),
71  m_position.X()/CLHEP::mm, m_position.Y()/CLHEP::mm, m_position.Z()/CLHEP::mm,
72  m_direction.X(),m_direction.Y(), m_direction.Z());
73  if ( m_print ) {
74  this->Geant4ParticleGenerator::printInteraction(m_mask);
75  }
76  ++m_shotNo;
